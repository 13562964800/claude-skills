    84→    fun loadSms() {
    85→        viewModelScope.launch {
    86→            _uiState.update { it.copy(loading = true) }
    87→            val rawList = smsReader.readCourtSms()
    88→            var addedCount = 0
    89→            for (raw in rawList) {
    90→                val hash = raw.body.hashCode()
    91→                if (repo.smsExists(hash)) continue
    92→                val parsed = smsParser.parse(raw.body)
    93→                val entity = CourtSmsEntity(
    94→                    sender = raw.sender,
    95→                    body = raw.body,
    96→                    bodyHash = hash,
    97→                    receivedAt = raw.date,
    98→                    caseNumber = parsed.caseNumber,
    99→                    courtName = parsed.courtName,
   100→                    hearingTime = parsed.hearingTime,
   101→                    partyNames = parsed.partyNames,
   102→                    webUrl = parsed.webUrl
   103→                )
   104→                val id = repo.insertSms(entity)
   105→                if (id > 0 && parsed.hearingTime != null && parsed.caseNumber != null) {
   106→                    val eventId = calendarSync.syncHearing(
   107→                        caseNumber = parsed.caseNumber,
   108→                        courtName = parsed.courtName,
   109→                        hearingTime = parsed.hearingTime,
   110→                        partyNames = parsed.partyNames,
   111→                        attachments = emptyList()
   112→                    )
   113→                    if (eventId != null) {
   114→                        repo.updateSms(entity.copy(calendarEventId = eventId))
   115→                    }
   116→                    addedCount++
   117→                }
   118→            }
   119→            _uiState.update { it.copy(loading = false, downloadCount = addedCount, message = "已读取 ${rawList.size} 条法院短信") }
   120→        }
   121→    }

   123→    fun downloadPdf(url: String, smsId: Long, fileName: String? = null, cookies: String? = null) {
   124→        viewModelScope.launch {
   125→            val result = pdfDownloader.download(url, fileName, cookies)
   126→            when (result) {
   127→                is DownloadResult.Success -> {
   128→                    // 使用MD5检查文件是否已下载（更可靠的去重方式）
   129→                    val existingPdf = repo.getPdfByMd5(result.fileMd5)
   130→                    if (existingPdf != null) {
   131→                        _downloadEvent.emit("文件已存在: ${result.fileName}")
   132→                        refreshData()
   133→                        return@launch
   134→                    }
   135→
   136→                    val isSubpoena = result.fileName.contains("传票")
   137→                    val pdf = PdfDocumentEntity(
   138→                        smsId = smsId,
   139→                        fileName = result.fileName,
   140→                        filePath = result.filePath,
   141→                        fileSize = result.fileSize,
   142→                        fileMd5 = result.fileMd5,
   143→                        ocrTitle = null,
   144→                        downloadedAt = System.currentTimeMillis(),
   145→                        isSubpoena = isSubpoena
   146→                    )
   147→                    val pdfId = repo.insertPdf(pdf)
   148→                    if (pdfId > 0) {
   149→                        _uiState.update { it.copy(downloadCount = it.downloadCount + 1, lastDownloadedPdfId = pdfId) }
   150→                        _downloadEvent.emit("已保存: ${result.fileName}")
   151→                        refreshData()
   152→                        if (isSubpoena) {
   153→                            parseAndSyncSubpoena(pdfId, result.filePath, smsId)
   154→                        }
   155→                    } else {
   156→                        _downloadEvent.emit("文件已存在: ${result.fileName}")
   157→                        refreshData()
   158→                    }
   159→                }
   160→                is DownloadResult.Error -> {
   161→                    _downloadEvent.emit("下载失败: ${result.message}")
   162→                }
   163→            }
   164→        }
   165→    }

   167→    private suspend fun parseAndSyncSubpoena(pdfId: Long, filePath: String, smsId: Long) {
   168→        val info = pdfParser.parseSubpoena(filePath) ?: return
   169→        val pdf = repo.getPdfById(pdfId) ?: return
   170→        val hearingDesc = buildString {
   171→            if (info.courtName != null) append("法院: ${info.courtName}\n")
   172→            if (info.partyNames != null) append("当事人: ${info.partyNames}\n")
   173→            if (info.location != null) append("地点: ${info.location}")
   174→        }
   175→        repo.updatePdf(pdf.copy(hearingInfo = hearingDesc))
   176→
   177→        if (info.hearingTime != null && info.caseNumber != null) {
   178→            // 创建PDF附件
   179→            val attachment = calendarSync.createAttachmentUri(filePath)
   180→
   181→            val eventId = calendarSync.syncHearing(
   182→                caseNumber = info.caseNumber,
   183→                courtName = info.courtName,
   184→                hearingTime = info.hearingTime,
   185→                partyNames = info.partyNames,
   186→                description = "地点: ${info.location ?: "见传票"}",
   187→                attachments = if (attachment != null) listOf(attachment) else emptyList()
   188→            )
   189→            if (eventId != null) {
   190→                repo.updatePdf(pdf.copy(hearingInfo = hearingDesc, calendarEventId = eventId))
   191→            }
   192→        }
   193→    }

   297→    fun syncCalendar() {
   298→        viewModelScope.launch {
   299→            val allSms = _uiState.value.smsWithPdfs
   300→            val needsSync = allSms.filter {
   301→                it.sms.hearingTime != null &&
   302→                it.sms.caseNumber != null &&
   303→                it.sms.calendarEventId == null
   304→            }
   305→
   306→            if (needsSync.isEmpty()) {
   307→                _uiState.update { it.copy(message = "没有需要同步的开庭信息") }
   308→                return@launch
   309→            }
   310→
   311→            _uiState.update {
   312→                it.copy(
   313→                    syncing = true,
   314→                    syncProgress = 0,
   315→                    syncTotal = needsSync.size,
   316→                    message = "开始同步 ${needsSync.size} 条开庭信息..."
   317→                )
   318→            }
   319→
   320→            var syncedCount = 0
   321→            val details = mutableListOf<String>()
   322→
   323→            needsSync.forEach { item ->
   324→                // 创建附件列表
   325→                val attachments = mutableListOf<com.court.assistant.calendar.CalendarAttachment>()
   326→
   327→                // 添加PDF附件
   328→                item.pdfs.forEach { pdf ->
   329→                    calendarSync.createAttachmentUri(pdf.filePath)?.let { attachments.add(it) }
   330→                }
   331→
   332→                // 如果没有PDF，创建短信文本附件
   333→                if (attachments.isEmpty()) {
   334→                    calendarSync.createSmsAttachment(
   335→                        caseNumber = item.sms.caseNumber!!,
   336→                        courtName = item.sms.courtName,
   337→                        body = item.sms.body,
   338→                        receivedAt = item.sms.receivedAt
   339→                    )?.let { attachments.add(it) }
   340→                }
   341→
   342→                val eventId = calendarSync.syncHearing(
   343→                    caseNumber = item.sms.caseNumber!!,
   344→                    courtName = item.sms.courtName,
   345→                    hearingTime = item.sms.hearingTime!!,
   346→                    partyNames = item.sms.partyNames,
   347→                    description = "地点: ${item.sms.location ?: "见传票"}",
   348→                    attachments = attachments
   349→                )
   350→
   351→                if (eventId != null) {
   352→                    syncedCount++
   353→                    repo.updateSms(item.sms.copy(calendarEventId = eventId))
   354→                    details.add("${item.sms.caseNumber} - ${item.sms.courtName} (${attachments.size}个附件)")
   355→                }
   356→
   357→                _uiState.update {
   358→                    it.copy(
   359→                        syncProgress = syncedCount,
   360→                        message = "正在同步... ($syncedCount/${needsSync.size})"
   361→                    )
   362→                }
   363→            }
   364→
   365→            _uiState.update {
   366→                it.copy(
   367→                    syncing = false,
   368→                    message = "同步完成！成功同步 $syncedCount 条开庭信息到日历"
   369→                )
   370→
   371→            _downloadEvent.emit("✅ 同步完成！\n成功同步 $syncedCount 条开庭信息")
   372→        }
   373→    }

   435→    /**
   436→     * 创建多个PDF的ZIP压缩包用于分享
   437→     * @param pdfIds 要打包的PDF文件ID列表
   438→     * @return ZIP文件路径，失败返回null
   439→     */
   440→    suspend fun createZipForPdf(pdfIds: List<Long>): String? {
   441→        return try {
   442→            val zipDir = java.io.File(getApplication<Application>().getExternalFilesDir(null), "Zips")
   443→            if (!zipDir.exists()) zipDir.mkdirs()
   444→
   445→            val timestamp = java.text.SimpleDateFormat("yyyyMMdd_HHmmss", java.util.Locale.CHINA)
   446→                .format(java.util.Date())
   447→            val zipFile = java.io.File(zipDir, "法院文书_$timestamp.zip")
   448→
   449→            java.util.zip.ZipOutputStream(java.io.FileOutputStream(zipFile)).use { zos ->
   450→                pdfIds.forEach { pdfId ->
   451→                    val pdf = repo.getPdfById(pdfId) ?: return@forEach
   452→                    val pdfFile = java.io.File(pdf.filePath)
   453→                    if (!pdfFile.exists()) return@forEach
   454→                    val entry = java.util.zip.ZipEntry(pdf.fileName)
   455→                    zos.putNextEntry(entry)
   456→                    pdfFile.inputStream().use { input ->
   457→                        input.copyTo(zos)
   458→                    }
   459→                    zos.closeEntry()
   460→                }
   461→            }
   462→            }
   463→            zipFile.absolutePath
   464→        } catch (e: Exception) {
   465→            e.printStackTrace()
   466→            null
   467→        }
   468→    }

   470→    /**
   471→     * 获取某个短信关联的所有PDF的ZIP压缩包
   472→     * @param smsId 短信ID
   473→     * @return ZIP文件路径，失败返回null
   474→     */
   475→    suspend fun createZipForSms(smsId: Long): String? {
   476→        val smsWithPdfs = _uiState.value.smsWithPdfs.find { it.sms.id == smsId }
   477→            ?: return null
   478→        val pdfIds = smsWithPdfs.pdfs.map { it.id }
   479→        return if (pdfIds.isNotEmpty()) {
   480→            val caseNumber = smsWithPdfs.sms.caseNumber ?: "案件"
   481→            val zipPath = createZipForPdf(pdfIds)
   482→            // 重命名ZIP文件为OCR标题（优先）或案号
   483→            zipPath?.let {
   484→                val oldFile = java.io.File(it)
   485→                val sms = _uiState.value.smsWithPdfs.find { it.sms.id == smsId }
   486→                val displayName = sms?.pdfs.firstNotNull()?.ocrTitle
   487→                    ?: sms?.pdfs.firstNotNull()?.fileName ?: "未知文件"
   488→                        .substringBeforeLast(".", "")
   489→                val newFile = java.io.File(oldFile.parentFile, displayName + ".zip")
   490→                oldFile.renameTo(newFile)
   491→                newFile.absolutePath
   492→            }
   493→        } else null
   494→    }
