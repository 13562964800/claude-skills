   375→    private suspend fun createZipFile(
   376→        docs: List<DocumentRecord>,
   377→        progress: ProgressDialog
   378→    ): File = withContext(Dispatchers.IO) {
   379→        val timestamp = System.currentTimeMillis()
   380→        val zipFile = File(requireContext().cacheDir, "court_docs_$timestamp.zip")
   381→
   382→        var processedCount = 0
   383→
   384→        ZipOutputStream(zipFile.outputStream()).use { zos ->
   385→            for (doc in docs) {
   386→                val file = File(doc.filePath)
   387→                if (!file.exists()) {
   388→                    processedCount++
   389→                    withContext(Dispatchers.Main) {
   390→                        progress.progress = processedCount
   391→                    }
   392→                    delay(10)
   393→                    continue
   394→                }
   395→
   396→                // 添加文件到ZIP
   397→                val entry = ZipEntry(doc.fileName)
   398→                zos.putNextEntry(entry)
   399→                file.inputStream().use { input ->
   400→                    input.copyTo(zos)
   401→                }
   402→                zos.closeEntry()
   403→
   404→                processedCount++
   405→                withContext(Dispatchers.Main) {
   406→                    progress.progress = processedCount
   407→                }
   408→                delay(50)
   409→            }
   410→        }
   411→
   412→        return@withContext zipFile
   413→    }
