    41→    fun syncSms() {
    42→        viewModelScope.launch {
    43→            _syncStatus.value = "同步中..."
    44→            try {
    45→                val count = withContext(Dispatchers.IO) {
    46→                    val latestTime = db.smsDao().getLatestTime() ?: 0
    47→                    Log.d(TAG, "syncSms: latestTime=$latestTime")
    48→                    val records = SmsReader.readAll(getApplication<App>().contentResolver, latestTime)
    49→                    Log.d(TAG, "syncSms: found ${records.size} records from 12368")
    50→                    var c = 0
    51→                    for (r in records) {
    52→                        Log.d(TAG, "syncSms: processing sms body=${r.body.take(80)} hasLink=${r.hasLink}")
    53→                        val urls = TextParser.extractUrls(r.body)
    54→                        Log.d(TAG, "syncSms: extracted ${urls.size} urls: $urls")
    55→                        val id = db.smsDao().insert(r)
    56→                        if (id > 0) {
    57→                            c++
    58→                            SmsProcessor.process(getApplication(), db, r.copy(id = id))
    59→                        }
    60→                    }
    61→                    c
    62→                }
    63→                _syncStatus.value = if (count > 0) "同步完成，新增 $count 条" else "无新短信"
    64→                _selectedSmsId.value = _selectedSmsId.value
    65→                refreshCalendarSyncMap()
    66→            } catch (e: Exception) {
    67→                Log.e(TAG, "syncSms error", e)
    68→                _syncStatus.value = "同步失败: ${e.message}"
    69→            }
    70→        }
    71→    }

   105→    /**
   106→     * 自动处理待下载的文档（parseStatus=4）
   107→     * 对于需要WebView手动交互的PDF链接，自动触发下载
   108→     */
   109→    fun autoProcessPendingDocs() {
   110→        viewModelScope.launch(Dispatchers.IO) {
   111→            // 首先清理卡在"处理中"状态的文档（超过5分钟未更新的）
   112→            val stuckDocs = db.documentDao().getStuckProcessingDocs()
   113→            for (stuck in stuckDocs) {
   114→                db.documentDao().update(stuck.copy(parseStatus = 4))
   115→                Log.d(TAG, "autoProcessPendingDocs: reset stuck doc id=${stuck.id}")
   116→            }
   117→
   118→            val pendingDocs = db.documentDao().getPendingWebviewDocs()
   119→            if (pendingDocs.isEmpty()) return@launch
   120→
   121→            Log.d(TAG, "autoProcessPendingDocs: found ${pendingDocs.size} pending docs")
   122→
   123→            // 依次处理每个待下载文档，每次只启动一个
   124→            for (doc in pendingDocs) {
   125→                // 检查是否已在处理中（parseStatus=5表示处理中）
   126→                if (doc.parseStatus == 5) {
   127→                    Log.d(TAG, "autoProcessPendingDocs: skip already processing ${doc.sourceUrl}")
   128→                    continue
   129→                }
   130→
   131→                // 标记为处理中（parseStatus=5）
   132→                db.documentDao().update(doc.copy(parseStatus = 5))
   133→
   134→                // 启动WebView处理
   135→                withContext(Dispatchers.Main) {
   136→                    val intent = Intent(getApplication(), WebViewActivity::class.java).apply {
   137→                        putExtra("url", doc.sourceUrl)
   138→                        putExtra("smsId", doc.smsId)
   139→                        putExtra("docId", doc.id)
   140→                        putExtra("autoMode", true) // 标记为自动模式
   141→                        addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   142→                    }
   143→                    getApplication<Application>().startActivity(intent)
   144→                }
   145→
   146→                // 等待当前WebView处理完成后再处理下一个
   147→                // 每个WebView最多需要90秒（超时时间），这里设置100秒确保安全
   148→                delay(100000)
   149→            }
   150→        }
   151→    }
