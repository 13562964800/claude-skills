   110→    private fun shareFile(path: String, name: String) {
   111→        try {
   112→            val file = File(path)
   113→            if (!file.exists()) { Toast.makeText(requireContext(), "文件不存在", Toast.LENGTH_SHORT).show(); return }
   114→            val uri = FileProvider.getUriForFile(requireContext(), "${requireContext().packageName}.fileprovider", file)
   115→            val intent = Intent(Intent.ACTION_SEND).apply {
   116→                type = "application/pdf"
   117→                putExtra(Intent.EXTRA_STREAM, uri)
   118→                addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
   119→            }
   120→            startActivity(Intent.createChooser(intent, "分享 $name"))
   121→        } catch (e: Exception) {
   122→            Toast.makeText(requireContext(), "分享失败: ${e.message}", Toast.LENGTH_SHORT).show()
   123→        }
   124→    }
   125→
   126→    private fun shareAsZip(files: List<File>) {
   127→        try {
   128→            val zipFile = File(requireContext().cacheDir, "court_docs_${System.currentTimeMillis()}.zip")
   129→            ZipOutputStream(FileOutputStream(zipFile)).use { zos ->
   130→                for (f in files) {
   131→                    zos.putNextEntry(ZipEntry(f.name))
   132→                    f.inputStream().use { it.copyTo(zos) }
   133→                    zos.closeEntry()
   134→                }
   135→            }
   136→            val uri = FileProvider.getUriForFile(requireContext(), "${requireContext().packageName}.fileprovider", zipFile)
   137→            val intent = Intent(Intent.ACTION_SEND).apply {
   138→                type = "application/zip"
   139→                putExtra(Intent.EXTRA_STREAM, uri)
   140→                addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
   141→            }
   142→            startActivity(Intent.createChooser(intent, "分享 ${files.size} 个文件"))
   143→        } catch (e: Exception) {
   144→            Toast.makeText(requireContext(), "打包失败: ${e.message}", Toast.LENGTH_SHORT).show()
   145→        }
   146→    }
   147→
   148→    private fun previewPdf(path: String) {
   149→        try {
   150→            val file = File(path)
   151→            if (!file.exists()) { Toast.makeText(requireContext(), "文件不存在", Toast.LENGTH_SHORT).show(); return }
   152→            val uri = FileProvider.getUriForFile(requireContext(), "${requireContext().packageName}.fileprovider", file)
   153→            val intent = Intent(Intent.ACTION_VIEW).apply {
   154→                setDataAndType(uri, "application/pdf")
   155→                addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
   156→            }
   157→            startActivity(intent)
   158→        } catch (e: Exception) {
   159→            Toast.makeText(requireContext(), "无法打开PDF，请安装PDF阅读器", Toast.LENGTH_SHORT).show()
   160→        }
   161→    }
