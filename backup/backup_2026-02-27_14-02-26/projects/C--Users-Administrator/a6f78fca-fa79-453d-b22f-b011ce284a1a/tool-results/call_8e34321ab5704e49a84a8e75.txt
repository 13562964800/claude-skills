   331→/**
   332→ * 下载进度卡片组件
   333→ */
   334→@Composable
   335→private fun DownloadProgressCard(
   336→    task: DownloadTask,
   337→    onCancel: () -> Unit,
   338→    formatFileSize: (Long) -> String,
   339→    formatSpeed: (Long) -> String,
   340→    formatRemainingTime: (Long) -> String,
   341→    getStatusText: (DownloadStatus) -> String
   342→) {
   343→    Card(
   344→        modifier = Modifier.fillMaxWidth(),
   345→        colors = CardDefaults.cardColors(
   346→            containerColor = when (task.status) {
   347→                DownloadStatus.FAILED, DownloadStatus.CANCELLED -> MaterialTheme.colorScheme.errorContainer
   348→                else -> MaterialTheme.colorScheme.primaryContainer
   349→            }
   350→        )
   351→    ) {
   352→        Column(modifier = Modifier.padding(16.dp)) {
   353→            // 标题和状态
   354→            Row(
   355→                modifier = Modifier.fillMaxWidth(),
   356→                horizontalArrangement = Arrangement.SpaceBetween,
   357→                verticalAlignment = Alignment.CenterVertically
   358→            ) {
   359→                Column(Modifier.weight(1f)) {
   360→                    Text(
   361→                        task.title,
   362→                        style = MaterialTheme.typography.titleSmall,
   363→                        maxLines = 1,
   364→                        overflow = TextOverflow.Ellipsis
   365→                    )
   366→                    Text(
   367→                        getStatusText(task.status),
   368→                        style = MaterialTheme.typography.bodySmall,
   369→                        color = when (task.status) {
   370→                            DownloadStatus.FAILED, DownloadStatus.CANCELLED -> MaterialTheme.colorScheme.error
   371→                            else -> MaterialTheme.colorScheme.onPrimaryContainer
   372→                        }
   373→                    )
   374→                }
   375→
   376→                // 取消按钮
   377→                if (task.status == DownloadStatus.DOWNLOADING || task.status == DownloadStatus.FETCHING_INFO) {
   378→                    IconButton(onClick = onCancel) {
   379→                        Icon(Icons.Default.Close, "取消", tint = MaterialTheme.colorScheme.error)
   380→                    }
   381→                }
   382→            }
   383→
   384→            Spacer(Modifier.height(12.dp))
   385→
   386→            // 进度条
   387→            if (task.status == DownloadStatus.DOWNLOADING) {
   388→                Column {
   389→                    Row(
   390→                        modifier = Modifier.fillMaxWidth(),
   391→                        horizontalArrangement = Arrangement.SpaceBetween
   392→                    ) {
   393→                        Text(
   394→                            "${task.progress}%",
   395→                            style = MaterialTheme.typography.bodySmall
   396→                        )
   397→                        if (task.speed > 0) {
   398→                            Text(
   399→                                formatSpeed(task.speed),
   400→                                style = MaterialTheme.typography.bodySmall,
   401→                                color = MaterialTheme.colorScheme.primary
   402→                            )
   403→                        }
   404→                    }
   405→
   406→                    Spacer(Modifier.height(4.dp))
   407→
   408→                    LinearProgressIndicator(
   409→                        progress = { task.progress / 100f },
   410→                        modifier = Modifier.fillMaxWidth(),
   411→                        color = MaterialTheme.colorScheme.primary
   412→                    )
   413→
   414→                    // 文件大小和剩余时间
   415→                    if (task.totalBytes > 0) {
   416→                        Spacer(Modifier.height(4.dp))
   417→                        Row(
   418→                            modifier = Modifier.fillMaxWidth(),
   419→                            horizontalArrangement = Arrangement.SpaceBetween
   420→                        ) {
   421→                            Text(
   422→                                "${formatFileSize(task.bytesDownloaded)} / ${formatFileSize(task.totalBytes)}",
   423→                                style = MaterialTheme.typography.bodySmall,
   424→                                color = MaterialTheme.colorScheme.onPrimaryContainer
   425→                            )
   426→                            if (task.remainingTime > 0 && task.speed > 0) {
   427→                                Text(
   428→                                    "剩余 ${formatRemainingTime(task.remainingTime)}",
   429→                                    style = MaterialTheme.typography.bodySmall,
   430→                                    color = MaterialTheme.colorScheme.onPrimaryContainer
   431→                                )
   432→                            }
   433→                        }
   434→                    }
   435→                }
   436→            } else if (task.status == DownloadStatus.FETCHING_INFO) {
   437→                Row(
   438→                    modifier = Modifier.fillMaxWidth(),
   439→                    verticalAlignment = Alignment.CenterVertically
   440→                ) {
   441→                    CircularProgressIndicator(
   442→                        modifier = Modifier.size(20.dp),
   443→                        strokeWidth = 2.dp
   444→                    )
   445→                    Spacer(Modifier.width(8.dp))
   446→                    Text(
   447→                        "正在解析视频信息...",
   448→                        style = MaterialTheme.typography.bodySmall,
   449→                        color = MaterialTheme.colorScheme.onPrimaryContainer
   450→                    )
   451→                }
   452→            } else if (task.status == DownloadStatus.FAILED) {
   453→                Row(
   454→                    modifier = Modifier.fillMaxWidth(),
   455→                    verticalAlignment = Alignment.CenterVertically
   456→                ) {
   457→                    Icon(
   458→                        Icons.Default.Error,
   459→                        contentDescription = null,
   460→                        tint = MaterialTheme.colorScheme.error,
   461→                        modifier = Modifier.size(20.dp)
   462→                    )
   463→                    Spacer(Modifier.width(8.dp))
   464→                    Text(
   465→                        task.errorMessage ?: "下载失败",
   466→                        style = MaterialTheme.typography.bodySmall,
   467→                        color = MaterialTheme.colorScheme.error
   468→                    )
   469→                }
   470→            } else if (task.status == DownloadStatus.COMPLETED) {
   471→                Row(
   472→                    modifier = Modifier.fillMaxWidth(),
   473→                    verticalAlignment = Alignment.CenterVertically
   474→                ) {
   475→                    Icon(
   476→                        Icons.Default.CheckCircle,
   477→                        contentDescription = null,
   478→                        tint = MaterialTheme.colorScheme.primary,
   479→                        modifier = Modifier.size(20.dp)
   480→                    )
   481→                    Spacer(Modifier.width(8.dp))
   482→                    Text(
   483→                        "下载完成",
   484→                        style = MaterialTheme.typography.bodySmall,
   485→                        color = MaterialTheme.colorScheme.primary
   486→                    )
   487→                }
   488→            }
   489→        }
   490→    }
   491→}
