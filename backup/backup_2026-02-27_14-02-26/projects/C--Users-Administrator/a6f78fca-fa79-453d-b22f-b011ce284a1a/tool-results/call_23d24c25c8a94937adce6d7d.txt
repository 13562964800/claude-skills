    55→    /**
    56→     * 平台检测 - 参考 Electron 版本的 detectPlatform 函数
    57→     * 支持更多平台和短链接
    58→     */
    59→    fun detectPlatform(url: String): VideoPlatform {
    60→        val lowerUrl = url.lowercase()
    61→        return when {
    62→            // YouTube - 支持主域名和短链接
    63→            lowerUrl.contains("youtube.com") || lowerUrl.contains("youtu.be") -> VideoPlatform.YOUTUBE
    64→            // Bilibili - 支持主域名和 b23.tv 短链接
    65→            lowerUrl.contains("bilibili.com") || lowerUrl.contains("b23.tv") -> VideoPlatform.BILIBILI
    66→            // 抖音 - 支持多种域名
    67→            lowerUrl.contains("douyin.com") || lowerUrl.contains("iesdouyin.com") -> VideoPlatform.DOUYIN
    68→            // TikTok
    69→            lowerUrl.contains("tiktok.com") -> VideoPlatform.TIKTOK
    70→            // 微博
    71→            lowerUrl.contains("weibo.com") -> VideoPlatform.WEIBO
    72→            // 小红书
    73→            lowerUrl.contains("xiaohongshu.com") || lowerUrl.contains("xhslink.com") -> VideoPlatform.XIAOHONGSHU
    74→            // 直接视频链接
    75→            lowerUrl.contains(".mp4") || lowerUrl.contains(".m3u8") || lowerUrl.contains(".mpd") -> VideoPlatform.DIRECT
    76→            // 未知平台
    77→            else -> VideoPlatform.UNKNOWN
    78→        }
    79→    }

    81→    /**
    82→     * 获取视频下载链接 - 主入口函数
    83→     * 参考 Electron 版本的 extract-info IPC 处理
    84→     */
    85→    suspend fun getVideoInfo(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
    86→        withTimeout(DEFAULT_TIMEOUT) {
    87→            try {
    88→                val cleanUrl = cleanUrl(url)
    89→                Log.d(TAG, "Processing URL: $cleanUrl")
    90→
    91→                val platform = detectPlatform(cleanUrl)
    92→                Log.d(TAG, "Detected platform: $platform")
    93→
    94→                when (platform) {
    95→                    VideoPlatform.YOUTUBE -> fetchYoutubeInfo(cleanUrl)
    96→                    VideoPlatform.BILIBILI -> fetchBilibiliInfo(cleanUrl)
    97→                    VideoPlatform.DOUYIN -> fetchDouyinInfo(cleanUrl)
    98→                    VideoPlatform.TIKTOK -> fetchTiktokInfo(cleanUrl)
    99→                    VideoPlatform.DIRECT -> Result.success(
   100→                        VideoDownloadInfo(
   101→                            title = "直接视频链接",
   102→                            downloadUrl = cleanUrl,
   103→                            cover = null,
   104→                            duration = 0L,
   105→                            author = null,
   106→                            platform = VideoPlatform.DIRECT
   107→                        )
   108→                    )
   109→                    else -> Result.failure(
   110→                        Exception(
   111→                            "❌ 不支持的平台\n\n当前链接: $cleanUrl\n\n支持的平台:\n• YouTube (youtube.com, youtu.be)\n• B站 (bilibili.com, b23.tv)\n• 抖音 (douyin.com)\n• TikTok (tiktok.com)\n\n建议: 请使用「导入本地视频」功能"
   112→                        )
   113→                    )
   114→                }
   115→            } catch (e: Exception) {
   116→                Log.e(TAG, "Error fetching video info", e)
   117→                Result.failure(e)
   118→            }
   119→        }
   120→    }

   123→     * 清理 URL - 移除多余空格和无效字符
   124→     * 参考 Electron 版本的 cleanUrl 函数
   125→     */
   126→    private fun cleanUrl(url: String): String {
   127→        val trimmed = url.trim()
   128→        // 提取第一个有效的 HTTP/HTTPS 链接
   129→        val match = Regex("(https?://[^\\s\"'<>]+)").find(trimmed)
   130→        return match?.value ?: trimmed
   131→    }

   139→    private suspend fun fetchYoutubeInfo(url: String): Result<VideoDownloadInfo> {
   140→        Log.d(TAG, "Fetching YouTube info for: $url")
   141→
   142→        // 策略1: 尝试 InnerTube API
   143→        val innertubeResult = fetchYouTubeInnerTubeAPI(url)
   144→        if (innertubeResult.isSuccess) {
   145→            Log.d(TAG, "InnerTube API success")
   146→            return innertubeResult
   147→        }
   148→        Log.w(TAG, "InnerTube API failed: ${innertubeResult.exceptionOrNull()?.message}")
   149→
   150→        // 策略2: 尝试第三方API（备用）
   151→        return fetchYouTubeThirdPartyAPI(url)
   152→    }

   154→    /**
   155→     * 使用 YouTube InnerTube API 解析
   156→     * InnerTube 是 YouTube 的内部 RESTful API
   157→     */
   158→    private suspend fun fetchYouTubeInnerTubeAPI(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
   159→        try {
   160→            val videoId = extractYoutubeVideoId(url)
   161→            if (videoId == null) {
   162→                return@withContext Result.failure(Exception("无法提取视频ID"))
   163→            }
   164→            Log.d(TAG, "YouTube video ID: $videoId")
   165→
   166→            // InnerTube API 请求
   167→            val requestBody = JSONObject().apply {
   168→                put("context", JSONObject().apply {
   169→                    put("client", JSONObject().apply {
   170→                        put("clientName", "ANDROID")
   171→                        put("clientVersion", "19.29.37")
   172→                    })
   173→                })
   174→                put("videoId", videoId)
   175→            }.toString()
   176→
   177→            val request = Request.Builder()
   178→                .url("https://www.youtube.com/youtubei/v1/player?key=AIzaSyAO_FJ2SlqU8Q4STEHLGCilw_Y9_11qcW8")
   179→                .header("Content-Type", "application/json")
   180→                .header("User-Agent", "com.google.android.youtube/19.29.37")
   181→                .post(requestBody.toRequestBody("application/json".toMediaType()))
   182→                .build()
   183→
   184→            val response = client.newCall(request).execute()
   185→            val body = response.body?.string() ?: return@withContext Result.failure(IOException("空响应"))
   186→            val json = JSONObject(body)
   187→
   188→            val videoDetails = json.optJSONObject("videoDetails")
   189→            val title = videoDetails?.optString("title") ?: "YouTube视频"
   190→            val author = videoDetails?.optString("author") ?: ""
   191→            val lengthSeconds = videoDetails?.optLong("lengthSeconds") ?: 0L
   192→
   193→            val streamingData = json.optJSONObject("streamingData")
   194→            val formats = streamingData?.optJSONArray("formats")
   195→
   196→            if (formats != null && formats.length() > 0) {
   197→                val format = formats.getJSONObject(0)
   198→                val videoUrl = format.optString("url")
   199→                val thumbnails = videoDetails?.optJSONObject("thumbnail")?.optJSONArray("thumbnails")
   200→                val cover = thumbnails?.optJSONObject(0)?.optString("url") ?: ""
   201→
   202→                return@withContext Result.success(VideoDownloadInfo(
   203→                    title = title,
   204→                    downloadUrl = videoUrl,
   205→                    cover = cover,
   206→                    duration = lengthSeconds,
   207→                    author = author,
   208→                    platform = VideoPlatform.YOUTUBE
   209→                ))
   210→            }
   211→
   212→            Result.failure(Exception("无法获取视频流"))
   213→
   214→        } catch (e: Exception) {
   215→            Log.e(TAG, "InnerTube API error", e)
   216→            Result.failure(e)
   217→        }
   218→    }

   333→    /**
   334→     * 提取 B站 BV 号
   335→     * 支持 BV 号和 b23.tv 短链接
   336→     */
   337→    private fun extractBVID(url: String): String? {
   338→        val bvPattern = Regex("(BV[a-zA-Z0-9]{10})")
   339→        val match = bvPattern.find(url)
   340→        return match?.groupValues?.get(1)
   341→    }

   343→    /**
   344→     * 提取 YouTube 视频 ID
   345→     */
   346→    private fun extractYoutubeVideoId(url: String): String? {
   347→        // 支持 youtube.com/watch?v= 和 youtu.be/ 格式
   348→        val patterns = listOf(
   349→            Regex("(?:youtube\\.com/watch\\?v=|youtu\\.be/)([a-zA-Z0-9_-]{11})"),
   350→            Regex("youtube\\.com/embed/([a-zA-Z0-9_-]{11})")
   351→        )
   352→
   353→        for (pattern in patterns) {
   354→            val match = pattern.find(url)
   355→            if (match != null) {
   356→                return match.groupValues[1]
   357→            }
   358→        }
   359→        return null
   360→    }

   362→    /**
   363→     * 获取 B站视频信息 - 改进版，优先使用官方API
   364→     * 基于 GitHub 研究：bilibili-api Kotlin 库实现
   365→     *
   366→     * 策略：
   367→     * 1. 优先使用 B站官方 API
   368→     * 2. 官方API失败后，尝试第三方API作为备用
   369→     */
   370→    private suspend fun fetchBilibiliInfo(url: String): Result<VideoDownloadInfo> {
   371→        Log.d(TAG, "Fetching Bilibili info for: $url")
   372→
   373→        // 策略1: 尝试官方API
   374→        val officialResult = fetchBilibiliOfficialAPI(url)
   375→        if (officialResult.isSuccess) {
   376→            Log.d(TAG, "Official API success")
   377→            return officialResult
   378→        }
   379→        Log.w(TAG, "Official API failed: ${officialResult.exceptionOrNull()?.message}")

   381→        // 策略2: 尝试第三方API（备用）
   382→        return fetchBilibiliThirdPartyAPI(url)

   385→    /**
   386→     * 使用 B站官方 API 解析
   387→     * 参考 bilibili-api (Kotlin): https://github.com/czp3009/bilibili-api
   388→     *
   389→     * API 端点：
   390→     * - 视频信息: https://api.bilibili.com/x/web-interface/view?bvid={bvid}
   391→     * - 播放地址: https://api.bilibili.com/x/player/playurl?bvid={bvid}&cid={cid}
   392→     */
   393→    private suspend fun fetchBilibiliOfficialAPI(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
   394→        try {
   395→            // 1. 提取 BV 号
   396→            val bvid = extractBVID(url)
   397→            if (bvid == null) {
   398→                return@withContext Result.failure(Exception("无法提取BV号"))
   399→            }
   400→            Log.d(TAG, "BV ID: $bvid")
   401→
   402→            // 获取Cookie（如果有配置）
   403→            val cookie = getCookieForPlatform(url)
   404→            if (cookie != null) {
   405→                Log.d(TAG, "Using configured cookies for Bilibili API")
   406→            }
   407→
   408→            // 2. 获取视频信息（获取 cid）
   409→            val infoUrl = "https://api.bilibili.com/x/web-interface/view?bvid=$bvid"
   410→            val infoRequestBuilder = Request.Builder()
   411→                .url(infoUrl)
   412→                .header("User-Agent", getUserAgent())
   413→                .header("Referer", "https://www.bilibili.com/")
   414→
   415→            // 添加Cookie
   416→            if (cookie != null) {
   417→                infoRequestBuilder.header("Cookie", cookie)
   418→            }
   419→
   420→            val infoRequest = infoRequestBuilder.build()
   421→
   422→            val infoResponse = client.newCall(infoRequest).execute()
   423→            val infoBody = infoResponse.body?.string() ?: return@withContext Result.failure(IOException("空响应"))
   424→            val infoJson = JSONObject(infoBody)
   425→
   426→            if (infoJson.optInt("code", -1) != 0) {
   427→                val errorMsg = infoJson.optString("message")
   428→                // 检查是否是需要Cookie的错误
   429→                if (errorMsg.contains("登录") || errorMsg.contains("会员") || errorMsg.contains("权限")) {
   430→                    return@withContext Result.failure(
   431→                        Exception("❌ 需要登录或会员权限\n\n错误信息: $errorMsg\n\n解决方案：\n• 在设置中配置B站Cookie（SESSDATA）\n• 或使用「导入本地视频」功能")
   432→                    )
   433→                }
   434→                return@withContext Result.failure(Exception("API错误: $errorMsg"))
   435→            }
   436→
   437→            val data = infoJson.getJSONObject("data")
   438→            val cid = data.optLong("cid")
   439→            val title = data.optString("title")
   440→            val cover = data.optString("pic").replace("\\/\\\\/", "/")
   441→            val duration = data.optInt("duration")
   442→            val author = data.optJSONObject("owner")?.optString("name") ?: "未知UP主"
   443→
   444→            Log.d(TAG, "Video info: $title, cid=$cid")
   445→
   446→            // 3. 获取播放URL - 使用fnval=16获取DASH格式（更稳定）
   447→            val playUrl = "https://api.bilibili.com/x/player/playurl?bvid=$bvid&cid=$cid&qn=80&fnval=16&fnver=0&fourk=1"
   448→            val playRequestBuilder = Request.Builder()
   449→                .url(playUrl)
   450→                .header("User-Agent", getUserAgent())
   451→                .header("Referer", "https://www.bilibili.com/")
   452→
   453→            // 添加Cookie
   454→            if (cookie != null) {
   455→                playRequestBuilder.header("Cookie", cookie)
   456→            }
   457→
   458→            val playRequest = playRequestBuilder.build()
   459→
   460→            val playResponse = client.newCall(playRequest).execute()
   461→            val playBody = playResponse.body?.string() ?: return@withContext Result.failure(IOException("空响应"))
   462→            val playJson = JSONObject(playBody)
   463→
   464→            if (playJson.optInt("code", -1) != 0) {
   465→                val errorMsg = playJson.optString("message")
   466→                if (errorMsg.contains("登录") || errorMsg.contains("会员") || errorMsg.contains("权限")) {
   467→                    return@withContext Result.failure(
   468→                        Exception("❌ 需要登录或会员权限\n\n错误信息: $errorMsg\n\n解决方案：\n• 在设置中配置B站Cookie（SESSDATA）\n• 或使用「导入本地视频」功能")
   469→                    )
   470→                }
   471→                return@withContext Result.failure(Exception("获取播放URL失败: $errorMsg"))
   472→            }
   473→
   474→            val playData = playJson.getJSONObject("data")
   475→            val durl = playData.optJSONArray("durl")
   476→            val videoUrl = if (durl != null && durl.length() > 0) {
   477→                durl.getJSONObject(0).optString("url")
   478→            } else {
   479→                // 尝试 DASH 格式
   480→                playData.optJSONObject("dash")?.optJSONArray("video")?.getJSONObject(0)?.optString("baseUrl")
   481→            }
   482→
   483→            if (videoUrl != null) {
   484→                Log.d(TAG, "Got video URL from official API, cookies used: ${cookie != null}")
   485→                // 无论是否配置Cookie，都将Cookie传递给VideoDownloadInfo
   486→                // 下载时会优先使用配置的Cookie
   487→                return@withContext Result.success(VideoDownloadInfo(
   488→                    title = title,
   489→                    downloadUrl = videoUrl,
   490→                    cover = cover,
   491→                    duration = duration.toLong(),
   492→                    author = author,
   493→                    cookies = cookie,
   494→                    platform = VideoPlatform.BILIBILI
   495→                ))
   496→            }
   497→
   498→            Result.failure(Exception("无法获取播放地址"))
   499→
   500→        } catch (e: Exception) {
   501→            Log.e(TAG, "Official API error", e)
   502→            Result.failure(e)
   503→        }
   504→    }

   506→    /**
   507→     * 使用第三方API解析（备用方案）
   508→     */

   577→                return@withContext Result.failure(Exception("API返回HTML，可能已失效"))
   578→            }
   579→
   580→            val json = JSONObject(responseBody)
   581→
   582→            // 尝试多种返回格式解析
   583→
   584→            // 格式1: {data: {url: "...", title: "..."}}
   585→            if (json.has("data")) {
   586→                val data = json.getJSONObject("data")
   587→                val videoUrl = data.optJSONArray("url")?.optString(0)
   588→                    ?: data.optString("url")
   589→                    ?: data.optString("video_url")
   590→                    ?: data.optString("playurl")
   591→                    ?: data.optString("download_url")
   592→
   593→                val title = data.optString("title")
   594→                    ?: data.optString("name")
   595→                    ?: "B站视频"
   596→
   597→                if (videoUrl.isNotEmpty() && videoUrl.startsWith("http")) {

   600→                        title = title,
   601→                        downloadUrl = videoUrl,
   602→                        cover = data.optString("pic")
   603→                            ?: data.optString("cover")
   604→                            ?: data.optString("img"),
   605→                        duration = data.optLong("duration", 0L) / 1000,
   606→                        author = data.optString("author")
   607→                            ?: data.optString("uploader")
   608→                            ?: data.optString("owner")?.let { ownerJson ->
   609→                                try { JSONObject(ownerJson).optString("name") } catch(e: Exception) { null }
   610→                            }
   611→                    ))
   612→                }
   613→            }
   614→
   615→            // 格式2: {url: "...", title: "..."}
   616→            val directUrl = json.optString("url")
   617→                ?: json.optString("video_url")
   618→                ?: json.optString("playurl")
   619→                ?: json.optString("download_url")
   620→
   621→            if (directUrl.startsWith("http")) {
   622→                val title = json.optString("title") ?: "B站视频"
   623→                Log.d(TAG, "[${api.name}] Got direct URL: $directUrl")
   624→                return@withContext Result.success(VideoDownloadInfo(
   625→                    title = title,
   626→                    downloadUrl = directUrl,
   627→                    cover = json.optString("pic") ?: json.optString("cover"),
   628→                    duration = json.optLong("duration", 0L) / 1000,
   629→                    author = json.optString("author")
   630→                ))
   631→            }
   632→
   633→            // 格式3: {code: 200, data: {url: "..."}} 或 {code: 200, url: "..."}
   634→            if (json.optInt("code") == 200 || json.optInt("code") == 1) {
   635→                val nestedData = json.optJSONObject("data")
   636→                val videoUrl = nestedData?.optString("url")
   637→                    ?: nestedData?.optString("video_url")
   638→                    ?: json.optString("url")
   639→                    ?: json.optString("video_url")

   641→                if (videoUrl?.startsWith("http") == true) {

   643→                        ?: json.optString("title")
   644→                        ?: "B站视频"
   645→                    return@withContext Result.success(VideoDownloadInfo(
   646→                        title = title,
   647→                        downloadUrl = videoUrl,
   648→                        cover = nestedData?.optString("pic") ?: json.optString("pic"),
   649→                        duration = nestedData?.optLong("duration", 0L)?.div(1000) ?: 0L
   650→                    ))
   651→                }
   652→            }
   653→
   654→            // 格式4: {msg: "...", code: ...} - 可能包含错误信息
   655→            val msg = json.optString("msg") ?: json.optString("message")
   656→            if (msg.isNotEmpty()) {
   657→                return@withContext Result.failure(Exception("API返回: $msg"))
   658→            }
   659→
   660→            // 无法解析
   661→            return@withContext Result.failure(Exception("无法解析API响应格式"))
   662→
   663→        } catch (e: Exception) {
   664→            Log.e(TAG, "[${api.name}] Request failed", e)
   665→            Result.failure(e)
