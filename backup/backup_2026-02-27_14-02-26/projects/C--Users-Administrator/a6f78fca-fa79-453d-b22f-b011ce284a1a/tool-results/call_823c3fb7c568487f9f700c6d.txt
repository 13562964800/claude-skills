    49→) {
    50→    val uiState by viewModel.uiState.collectAsState()

    54→    // 注册转录服务广播接收器
    55→    val transcriptionReceiver = remember {
    56→        object : BroadcastReceiver() {
    57→            override fun onReceive(context: Context?, intent: Intent?) {
    58→                when (intent?.action) {
    59→                    TranscriptionService.ACTION_PROGRESS -> {
    60→                        val segment = intent.getIntExtra(TranscriptionService.EXTRA_SEGMENT, 0)
    61→                        val total = intent.getIntExtra(TranscriptionService.EXTRA_TOTAL, 0)
    62→                        val progress = intent.getIntExtra(TranscriptionService.EXTRA_PROGRESS, 0)
    63→                        val message = intent.getStringExtra(TranscriptionService.EXTRA_MESSAGE) ?: ""
    64→                        val remaining = intent.getLongExtra(TranscriptionService.EXTRA_REMAINING, 0)
    65→                        viewModel.onTranscriptionProgress(segment, total, progress, message, remaining)
    66→                    }
    67→                    TranscriptionService.ACTION_COMPLETED -> {
    68→                        val success = intent.getBooleanExtra(TranscriptionService.EXTRA_SUCCESS, false)
    69→                        val detectedLang = intent.getStringExtra(TranscriptionService.EXTRA_DETECTED_LANG) ?: ""
    70→                        val subtitleCount = intent.getIntExtra(TranscriptionService.EXTRA_SUBTITLE_COUNT, 0)
    71→                        val error = intent.getStringExtra(TranscriptionService.EXTRA_ERROR)
    72→                        viewModel.onTranscriptionCompleted(videoPath, success, detectedLang, subtitleCount, error)
    73→                    }
    74→                }
    75→            }
    76→        }
    77→    }
    78→
    79→    // 注册和注销广播接收器
    80→    DisposableEffect(Unit) {
    81→        val progressFilter = IntentFilter(TranscriptionService.ACTION_PROGRESS)
    82→        val completedFilter = IntentFilter(TranscriptionService.ACTION_COMPLETED)
    83→        context.registerReceiver(transcriptionReceiver, progressFilter)
    84→        context.registerReceiver(transcriptionReceiver, completedFilter)
    85→
    86→        onDispose {

    89→    }

    98→
    99→    // Load subtitles

   112→    }

   118→        }
   119→    }

   121→    DisposableEffect(Unit) {

   133→                }
   134→            )
   135→        }
