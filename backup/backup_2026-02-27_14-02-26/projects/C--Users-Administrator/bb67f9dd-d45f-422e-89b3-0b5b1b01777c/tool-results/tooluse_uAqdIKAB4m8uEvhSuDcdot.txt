   316→// VTT解析 (处理YouTube自动字幕的双行格式)
   317→function parseVTT(vtt) {
   318→  const raw = [];
   319→  for (const block of vtt.split(/\n\n+/)) {
   320→    const lines = block.trim().split('\n');
   321→    for (let i = 0; i < lines.length; i++) {
   322→      const m = lines[i].match(/(\d{2}):(\d{2}):(\d{2})\.(\d{3})\s*-->\s*(\d{2}):(\d{2}):(\d{2})\.(\d{3})/);
   323→      if (!m) continue;
   324→      const start = +m[1]*3600 + +m[2]*60 + +m[3] + +m[4]/1000;
   325→      const end = +m[5]*3600 + +m[6]*60 + +m[7] + +m[8]/1000;
   326→      // 跳过极短条目(YouTube自动字幕的过渡帧)
   327→      if (end - start < 0.05) break;
   328→      // 取最后一行非空文本(YouTube双行字幕中新内容在最后一行)
   329→      const textLines = lines.slice(i+1).map(l => l.replace(/<[^>]+>/g, '').trim()).filter(Boolean);
   330→      const text = textLines.length > 0 ? textLines[textLines.length - 1] : '';
   331→      if (text) raw.push({ start, end, text });
   332→      break;
   333→    }
   334→  }
   335→  // 合并连续相同文本，去重
   336→  const result = [];
   337→  for (const s of raw) {
   338→    if (result.length > 0 && result[result.length-1].text === s.text) {
   339→      result[result.length-1].end = s.end; // 延长时间
   340→    } else {
   341→      result.push({ ...s });
   342→    }
   343→  }
   344→  return result;
   345→}

   347→// 字幕同步 (用setInterval确保可靠触发)
   348→let syncTimer = null;
   349→function startSubtitleSync() {
   350→  if (syncTimer) clearInterval(syncTimer);
   351→  syncTimer = setInterval(() => {
   352→    if (!subtitles.length || currentMode !== 'learn' || videoPlayer.paused) return;
   353→    const t = videoPlayer.currentTime;
   354→    // 二分查找当前时间对应的字幕
   355→    let lo = 0, hi = subtitles.length - 1, idx = -1;
   356→    while (lo <= hi) {
   357→      const mid = (lo + hi) >> 1;
   358→      if (t >= subtitles[mid].start && t <= subtitles[mid].end) { idx = mid; break; }
   359→      if (t < subtitles[mid].start) hi = mid - 1;
   360→      else lo = mid + 1;
   361→    }
   362→    if (idx < 0) {
   363→      // 在间隙中：找下一条即将出现的字幕（0.5秒内），或保持上一条
   364→      for (let i = 0; i < subtitles.length; i++) {
   365→        if (subtitles[i].start > t) {
   366→          if (subtitles[i].start - t < 0.5) { idx = i; }
   367→          else if (i > 0) { idx = i - 1; }
   368→          break;
   369→        }
   370→      }
   371→      if (idx < 0) idx = subtitles.length - 1; // 超过最后一条
   372→    }
   373→    if (idx !== currentSubIndex) {
   374→      currentSubIndex = idx;
   375→      currentSentence = subtitles[idx].text;
   376→      renderSubtitle(currentSentence);
   377→      aiAnalyzeBtn.style.display = 'inline-block';
   378→      aiResult.style.display = 'none';
   379→      highlightSubtitleList(idx);
   380→    }
   381→    // 卡拉OK式逐词点亮
   382→    if (idx >= 0 && idx === currentSubIndex) {
   383→      const sub = subtitles[idx];
   384→      const words = subtitleText.querySelectorAll('.word');
   385→      if (words.length > 0) {
   386→        if (wordTimings) {
   387→          // 精准模式：用逐词时间戳，按文本匹配对齐
   388→          const subWords = wordTimings.filter(wt => wt.s >= sub.start - 0.1 && wt.s <= sub.end + 0.3);
   389→          if (subWords.length > 0) {
   390→            // 建立显示词 → wordTiming 的精准映射
   391→            let wtIdx = 0;
   392→            words.forEach((w) => {
   393→              const displayed = w.textContent.trim().toLowerCase().replace(/[^\p{L}\p{M}\p{N}'-]/gu, '');
   394→              // 向前搜索匹配的 wordTiming
   395→              let matched = null;
   396→              for (let k = wtIdx; k < subWords.length; k++) {
   397→                const wtWord = subWords[k].w.trim().toLowerCase().replace(/[^\p{L}\p{M}\p{N}'-]/gu, '');
   398→                if (wtWord === displayed || displayed.startsWith(wtWord) || wtWord.startsWith(displayed)) {
   399→                  matched = subWords[k];
   400→                  wtIdx = k + 1;
   401→                  break;
   402→                }
   403→              }
   404→              if (matched) {
   405→                w.classList.toggle('word-lit', t >= matched.s && t < matched.e);
   406→                w.classList.toggle('word-past', t >= matched.e);
   407→              } else {
   408→                // 无匹配时不高亮
   409→                w.classList.remove('word-lit', 'word-past');
   410→              }
   411→            });
   412→          } else {
   413→            fallbackHighlight(words, t, sub);
   414→          }
   415→        } else {
   416→          fallbackHighlight(words, t, sub);
   417→        }
   418→      }
   419→    }
   420→  }, 30);
   421→}

   424→function fallbackHighlight(words, t, sub) {
   425→  const elapsed = t - sub.start;
   426→  const dur = sub.end - sub.start;
   427→  if (dur <= 0 || elapsed < 0) return;
   428→  const lens = Array.from(words).map(w => Math.max(w.textContent.length, 1));
   429→  const total = lens.reduce((a, b) => a + b, 0);
   430→  let cum = 0, activeIdx = words.length - 1;
   431→  for (let i = 0; i < lens.length; i++) {
   432→    cum += (lens[i] / total) * dur;
   433→    if (elapsed < cum) { activeIdx = i; break; }
   434→  }
   435→  words.forEach((w, i) => {
   436→    w.classList.toggle('word-lit', i === activeIdx);
   437→    w.classList.toggle('word-past', i < activeIdx);
   438→  });
   439→}

   473→function renderSubtitle(text) {
   474→  subtitleText.innerHTML = '';
   475→  wordInlinePanel.style.display = 'none';
   476→  const displayedWords = text.split(/(\s+)/).filter(w => !/^\s*$/.test(w));
   477→  // 预计算当前字幕对应的 wordTimings 映射
   478→  const sub = subtitles[currentSubIndex];
   479→  let wtMap = [];
   480→  if (wordTimings && sub) {
   481→    const subWords = wordTimings.filter(wt => wt.s >= sub.start - 0.1 && wt.s <= sub.end + 0.3);
   482→    let wtIdx = 0;
   483→    for (const dw of displayedWords) {
   484→      const clean = dw.trim().toLowerCase().replace(/[^\p{L}\p{M}\p{N}'-]/gu, '');
   485→      let matched = null;
   486→      for (let k = wtIdx; k < subWords.length; k++) {
   487→        const wtWord = subWords[k].w.trim().toLowerCase().replace(/[^\p{L}\p{M}\p{N}'-]/gu, '');
   488→        if (wtWord === clean || clean.startsWith(wtWord) || wtWord.startsWith(clean)) {
   489→          matched = subWords[k];
   490→          wtIdx = k + 1;
   491→          break;
   492→        }
   493→      }
   494→      wtMap.push(matched);
   495→    }
   496→  }
   497→
   498→  let wordIdx = 0;
   499→  for (const w of text.split(/(\s+)/)) {
   500→    if (/^\s+$/.test(w)) { subtitleText.appendChild(document.createTextNode(' ')); continue; }
   501→    const span = document.createElement('span');
   502→    span.className = 'word'; span.textContent = w;
   503→    const wt = wtMap[wordIdx] || null;
   504→    span.addEventListener('click', (e) => {
   505→      e.stopPropagation();
   506→      // 点击单词跳转到该词的精准时间
   507→      if (wt) {
   508→        videoPlayer.currentTime = wt.s;
   509→        videoPlayer.pause();
   510→      } else {
   511→        videoPlayer.pause();
   512→      }
   513→      const sel = window.getSelection().toString().trim();
   514→      if (sel && sel.includes(' ')) {
   515→        showWordInline(sel);
   516→      } else {
   517→        const clean = cleanWord(w);
   518→        if (clean) showWordInline(clean);
   519→      }
   520→    });
   521→    subtitleText.appendChild(span);
   522→    wordIdx++;
   523→  }
   524→}

   734→        `<div class="example">${e}</div>`

  1134→function drawFallbackBg(ctx, W, H, title) {
  1135→  // 丰富的渐变背景
  1136→  const colors = [
  1137→    ['#667eea', '#764ba2'], ['#f093fb', '#f5576c'], ['#4facfe', '#00f2fe'],
  1138→    ['#43e97b', '#38f9d7'], ['#fa709a', '#fee140'], ['#a18cd1', '#fbc2eb']
  1139→  ];
  1140→  const pair = colors[Math.abs(title.charCodeAt(0)) % colors.length];
  1141→  const grad = ctx.createLinearGradient(0, 0, W, H);
  1142→  grad.addColorStop(0, pair[0]); grad.addColorStop(1, pair[1]);
  1143→  ctx.fillStyle = grad; ctx.fillRect(0, 0, W, H);
  1144→
  1145→  // 装饰元素 - 大圆、小圆、波浪
  1146→  ctx.globalAlpha = 0.12;
  1147→  for (let i = 0; i < 8; i++) {
  1148→    const cx = Math.random() * W, cy = Math.random() * H * 0.7, r = 40 + Math.random() * 100;
  1149→    ctx.fillStyle = '#fff';
  1150→    ctx.beginPath(); ctx.arc(cx, cy, r, 0, Math.PI * 2); ctx.fill();
  1151→  }
  1152→  // 底部波浪
  1153→  ctx.fillStyle = 'rgba(255,255,255,0.15)';
  1154→  ctx.beginPath(); ctx.moveTo(0, H * 0.65);
  1155→  for (let x = 0; x <= W; x += 20) {
  1156→    ctx.lineTo(x, H * 0.65 + Math.sin(x * 0.015) * 30 + Math.sin(x * 0.008) * 20);
  1157→  }
  1158→  ctx.lineTo(W, H); ctx.lineTo(0, H); ctx.fill();
  1159→  ctx.globalAlpha = 1;
  1160→}

  1162→function rrect(ctx, x, y, w, h, r) {
  1163→  ctx.beginPath(); ctx.moveTo(x + r, y); ctx.lineTo(x + w - r, y);
  1164→  ctx.quadraticCurveTo(x + w, y, x + w, y + r); ctx.lineTo(x + w, y + h - r);
  1165→  ctx.quadraticCurveTo(x + w, y + h, x + w - r, y + h); ctx.lineTo(x + r, y + h);
  1166→  ctx.quadraticCurveTo(x, y + h, x, y + h - r); ctx.lineTo(x, y + r);
  1167→  ctx.quadraticCurveTo(x, y, x + r, y); ctx.closePath();
  1168→}

  1208→function speakLecture(text) {
  1209→  speechSynthesis.cancel();
  1210→  // 分段朗读，每段不超过200字
  1211→  const chunks = [];
  1212→  let remaining = text;
  1213→  while (remaining.length > 0) {
  1214→    if (remaining.length <= 200) { chunks.push(remaining); break; }
  1215→    let cut = remaining.lastIndexOf('。', 200);
  1216→    if (cut < 50) cut = remaining.lastIndexOf('.', 200);
  1217→    if (cut < 50) cut = remaining.lastIndexOf('，', 200);
  1218→    if (cut < 50) cut = 200;
  1219→    chunks.push(remaining.slice(0, cut + 1));
  1220→    remaining = remaining.slice(cut + 1);
  1221→  }
  1222→  let i = 0;
  1223→  function next() {
  1224→    if (i >= chunks.length) return;
  1225→    const u = new SpeechSynthesisUtterance(chunks[i]);
  1226→    u.lang = 'zh-CN'; u.rate = 0.9;
  1227→    u.onend = () => { i++; next(); };
  1228→    speechSynthesis.speak(u);
  1229→  }
  1230→  next();
  1231→}
