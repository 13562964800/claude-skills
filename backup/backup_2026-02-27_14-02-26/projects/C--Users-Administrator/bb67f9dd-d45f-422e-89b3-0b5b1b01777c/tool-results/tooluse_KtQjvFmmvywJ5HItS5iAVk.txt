    17→function baiduTranslate(text, fromLang) {
    18→  return new Promise((resolve) => {
    19→    const cfg = loadTranslateConfig();
    20→    const appid = cfg.baidu_appid || '', key = cfg.baidu_key || '';
    21→    if (!appid || !key) { resolve(null); return; }
    22→    const salt = String(Math.floor(Math.random() * 90000) + 10000);
    23→    const sign = crypto.createHash('md5').update(appid + text + salt + key).digest('hex');
    24→    const params = new URLSearchParams({ q: text, from: BAIDU_LANG_MAP[fromLang] || 'auto', to: 'zh', appid, salt, sign });
    25→    https.get('https://fanyi-api.baidu.com/api/trans/vip/translate?' + params, (res) => {
    26→      let d = ''; res.on('data', c => d += c);
    27→      res.on('end', () => {
    28→        try {
    29→          const json = JSON.parse(d);
    30→          if (json.trans_result) resolve(json.trans_result.map(r => r.dst).join('\n'));
    31→          else resolve(null);
    32→        } catch(e) { resolve(null); }
    33→      });
    34→    }).on('error', () => resolve(null));
    35→  });
    36→}

    78→// 通用AI调用 (OpenAI兼容格式, Claude单独处理)
    79→function callAI(prompt) {
    80→  return new Promise((resolve) => {
    81→    const cfg = loadAIConfig();
    82→    if (!cfg.apiKey) { resolve(null); return; }
    83→
    84→    const preset = AI_PROVIDERS[cfg.provider] || {};
    85→    const baseUrl = cfg.baseUrl || preset.baseUrl || '';
    86→    const model = cfg.model || preset.model || '';
    87→    const maxTokens = cfg.maxTokens || 1024;
    88→    const temperature = cfg.temperature !== null && cfg.temperature !== undefined ? cfg.temperature : undefined;
    89→    if (!baseUrl) { resolve(null); return; }
    90→
    91→    const isClaude = cfg.provider === 'claude' || (model && model.toLowerCase().includes('claude'));
    92→    const urlObj = new URL(isClaude ? baseUrl + '/messages' : baseUrl + '/chat/completions');
    93→    const isHttps = urlObj.protocol === 'https:';
    94→
    95→    let body, headers;
    96→    if (isClaude) {
    97→      const bodyObj = { model, max_tokens: maxTokens, messages: [{ role: 'user', content: prompt }] };
    98→      if (temperature !== undefined) bodyObj.temperature = temperature;
    99→      body = JSON.stringify(bodyObj);
   100→      headers = {
   101→        'Content-Type': 'application/json',
   102→        'x-api-key': cfg.apiKey,
   103→        'anthropic-version': '2023-06-01'
   104→      };
   105→    } else {
   106→      const bodyObj = {
   107→        model, max_tokens: maxTokens,
   108→        messages: [
   109→          { role: 'system', content: '你是多语言学习助手。用中文回答。翻译要准确，不要音译。简洁直接。' },
   110→          { role: 'user', content: prompt }
   111→        ]
   112→      };
   113→      if (temperature !== undefined) bodyObj.temperature = temperature;
   114→      body = JSON.stringify(bodyObj);
   115→      headers = {
   116→        'Content-Type': 'application/json',
   117→        'Authorization': 'Bearer ' + cfg.apiKey
   118→      };
   119→    }
   120→
   121→    const opts = {
   122→      hostname: urlObj.hostname, port: urlObj.port || (isHttps ? 443 : 80),
   123→      path: urlObj.pathname, method: 'POST', headers
   124→    };
   125→
   126→    const client = isHttps ? https : http;
   127→    const req = client.request(opts, (res) => {
   128→      let data = '';
   129→      res.on('data', d => data += d);
   130→      res.on('end', () => {
   131→        try {
   132→          const json = JSON.parse(data);
   133→          if (isClaude) {
   134→            resolve(json.content?.[0]?.text || null);
   135→          } else {
   136→            resolve(json.choices?.[0]?.message?.content || null);
   137→          }
   138→        } catch(e) { resolve(null); }
   139→      });
   140→    });
   141→    req.on('error', () => resolve(null));
   142→    req.setTimeout(15000, () => { req.destroy(); resolve(null); });
   143→    req.write(body);
   144→    req.end();
   145→  });
   146→}

   240→// 下载视频+字幕
   241→ipcMain.handle('download-video', async (_, url, lang) => {
   242→  const subLang = LANGUAGES[lang]?.sub || 'en';
   243→  return new Promise((resolve) => {
   244→    const infoArgs = ['--js-runtimes', 'node', '--dump-json', '--no-playlist', url];
   245→    if (fs.existsSync(path.join(FFMPEG_DIR, 'ffmpeg.exe'))) infoArgs.unshift('--ffmpeg-location', FFMPEG_DIR);
   246→    const infoProc = spawn(YTDLP, infoArgs);
   247→    let infoOut = '';
   248→    infoProc.stdout.on('data', d => infoOut += d);
   249→    infoProc.on('close', (code) => {
   250→      let title = 'video_' + Date.now();
   251→      if (code === 0) { try { title = JSON.parse(infoOut).title || title; } catch(e) {} }
   252→      const safeName = title.replace(/[<>:"/\\|?*]/g, '_').substring(0, 60);
   253→      const outTemplate = path.join(DOWNLOADS, safeName + '.%(ext)s');
   254→      const args = [
   255→        '--js-runtimes', 'node',
   256→        '--ffmpeg-location', FFMPEG_DIR,
   257→        '-f', 'bestvideo[height<=720]+bestaudio/best[height<=720]/best',
   258→        '--merge-output-format', 'mp4',
   259→        '--write-sub', '--write-auto-sub', '--sub-lang', subLang,
   260→        '--sub-format', 'vtt', '--convert-subs', 'vtt',
   261→        '--no-playlist', '-o', outTemplate, url
   262→      ];
   263→      const proc = spawn(YTDLP, args);
   264→      let stderr = '', maxPercent = 0, streamCount = 0;
   265→      proc.stdout.on('data', (data) => {
   266→        const output = data.toString();
   267→        if (output.includes('Destination:')) streamCount++;
   268→        const matches = output.match(/(\d+\.?\d*)%/g);
   269→        if (matches) {
   270→          let percent = parseFloat(matches[matches.length - 1]);
   271→          if (streamCount >= 2) percent = 50 + percent / 2;
   272→          else if (streamCount === 1) percent = percent / 2;
   273→          if (percent > maxPercent) { maxPercent = percent; mainWindow.webContents.send('download-progress', { percent }); }
   274→        }
   275→        if (output.includes('Merging')) mainWindow.webContents.send('download-progress', { percent: 99 });
   276→      });
   277→      proc.stderr.on('data', d => stderr += d);
   278→      proc.on('close', (code2) => {
   279→        const videoFile = path.join(DOWNLOADS, safeName + '.mp4');
   280→        const videoExists = fs.existsSync(videoFile);
   281→        if (code2 !== 0 && !videoExists) { resolve({ success: false, error: stderr || '下载失败' }); return; }
   282→        let subFile = null;
   283→        const possibleSubs = [
   284→          path.join(DOWNLOADS, safeName + '.' + subLang + '.vtt'),
   285→          path.join(DOWNLOADS, safeName + '.vtt'),
   286→        ];
   287→        try {
   288→          for (const f of fs.readdirSync(DOWNLOADS)) {
   289→            if (f.startsWith(safeName) && f.endsWith('.vtt')) possibleSubs.push(path.join(DOWNLOADS, f));
   290→          }
   291→        } catch(e) {}
   292→        for (const p of possibleSubs) { if (fs.existsSync(p)) { subFile = p; break; } }
   293→        resolve({ success: true, videoPath: fs.existsSync(videoFile) ? videoFile : null, subtitlePath: subFile, title: safeName });
   294→      });
   295→      proc.on('error', () => resolve({ success: false, error: 'yt-dlp未找到' }));
   296→    });
   297→  });
   298→});

   372→            phonetic: entry.phonetic || '',
   373→            audio: entry.phonetics?.find(p => p.audio)?.audio || '',

   375→            examples: examples.slice(0, 5)

   772→// Whisper API 转录（支持所有语言，自动检测）
   773→function transcribeWhisper(audioPath) {
   774→  return new Promise((resolve) => {
   775→    const cfg = loadAIConfig();
   776→    if (!cfg.apiKey) { resolve(null); return; }
   777→
   778→    // Groq 和 OpenAI 支持 Whisper
   779→    let hostname, apiPath, apiKey;
   780→    if (cfg.provider === 'groq') {
   781→      hostname = 'api.groq.com'; apiPath = '/openai/v1/audio/transcriptions'; apiKey = cfg.apiKey;
   782→    } else if (cfg.provider === 'openai') {
   783→      hostname = 'api.openai.com'; apiPath = '/v1/audio/transcriptions'; apiKey = cfg.apiKey;
   784→    } else {
   785→      // 其他供应商不支持 Whisper，尝试用 OpenAI 兼容格式
   786→      const baseUrl = cfg.baseUrl || AI_PROVIDERS[cfg.provider]?.baseUrl || '';
   787→      if (!baseUrl) { resolve(null); return; }
   788→      try {
   789→        const u = new URL(baseUrl);
   790→        hostname = u.hostname; apiPath = u.pathname.replace(/\/+$/, '') + '/audio/transcriptions'; apiKey = cfg.apiKey;
   791→      } catch(e) { resolve(null); return; }
   792→    }
   793→
   794→    const boundary = '----FormBoundary' + Date.now();
   795→    const audioData = fs.readFileSync(audioPath);
   796→    const model = cfg.provider === 'groq' ? 'whisper-large-v3' : 'whisper-1';
   797→
   798→    // multipart/form-data（不传language，让Whisper自动检测）
   799→    const parts = [];
   800→    parts.push(`--${boundary}\r\nContent-Disposition: form-data; name="file"; filename="audio.wav"\r\nContent-Type: audio/wav\r\n\r\n`);
   801→    parts.push(audioData);
   802→    parts.push(`\r\n--${boundary}\r\nContent-Disposition: form-data; name="model"\r\n\r\n${model}`);
   803→    parts.push(`\r\n--${boundary}\r\nContent-Disposition: form-data; name="response_format"\r\n\r\nverbose_json`);
   804→    parts.push(`\r\n--${boundary}--\r\n`);
   805→
   806→    const body = Buffer.concat(parts.map(p => typeof p === 'string' ? Buffer.from(p) : p));
   807→
   808→    const req = https.request({
   809→      hostname, port: 443, path: apiPath, method: 'POST',
   810→      headers: {
   811→        'Content-Type': `multipart/form-data; boundary=${boundary}`,
   812→        'Content-Length': body.length,
   813→        'Authorization': 'Bearer ' + apiKey
   814→      }
   815→    }, (res) => {
   816→      let data = '';
   817→      res.on('data', d => data += d);
   818→      res.on('end', () => {
   819→        try {
   820→          const json = JSON.parse(data);
   821→          if (json.text) {
   822→            const words = json.words || json.segments || [];
   823→            const timestamp = words.map(w => [Math.round((w.start || 0) * 1000), Math.round((w.end || 0) * 1000)]);
   824→            resolve({ success: true, text: json.text, timestamp, detectedLang: json.language || null });
   825→          } else { resolve(null); }
   826→        } catch(e) { resolve(null); }
   827→      });
   828→    });
   829→    req.on('error', () => resolve(null));
   830→    req.setTimeout(300000, () => { req.destroy(); resolve(null); });
   831→    req.write(body);
   832→    req.end();
   833→  });
   834→}

   855→    if (cfg.groqWhisperEnabled === false) return false;
   856→    const tc = loadTranslateConfig();
   857→    if (cfg.groqWhisperKey || tc.groq_key) return true;
   858→    return cfg.apiKey && cfg.provider === 'groq';
   859→  })();
   860→
   861→  sendSubtitleStatus('正在分析视频...');
   862→  const ffprobe = path.join(FFMPEG_DIR, 'ffprobe.exe');
   863→  const ffmpeg = path.join(FFMPEG_DIR, 'ffmpeg.exe');
   864→
   865→  const duration = await new Promise((resolve) => {
   866→    const p = spawn(ffprobe, ['-v', 'quiet', '-print_format', 'json', '-show_format', videoPath]);
   867→    let out = '';
   868→    p.stdout.on('data', d => out += d);
   869→    p.on('close', () => {
   870→      try { resolve(parseFloat(JSON.parse(out).format.duration) || 0); } catch(e) { resolve(0); }
   871→    });
   872→    p.on('error', () => resolve(0));
   873→  });
   874→  if (!duration) return null;
   875→
   876→  const durationMin = Math.ceil(duration / 60);
   877→  const segCount = Math.ceil(duration / SEGMENT_DURATION);
   878→  const fgSegCount = Math.ceil(Math.min(duration, FOREGROUND_LIMIT) / SEGMENT_DURATION);
   879→  const tmpDir = path.join(DOWNLOADS, '_tmp_segments');
   880→  if (!fs.existsSync(tmpDir)) fs.mkdirSync(tmpDir, { recursive: true });
   881→
   882→  // 转录指定范围的段落，返回 {vttText, allText}
   883→  async function transcribeRange(startSeg, endSeg) {
   884→    let allText = '', vtt = '', wordTimings = [];
   885→    let segTimes = [];
   886→    const totalSegs = endSeg - startSeg;
   887→    for (let i = startSeg; i < endSeg; i++) {
   888→      const segStart = Date.now();
   889→      const startSec = i * SEGMENT_DURATION;
   890→      const segFile = path.join(tmpDir, `seg_${i}.wav`);
   891→      const done = i - startSeg;
   892→      const remaining = endSeg - i;
   893→
   894→      let etaStr = '';
   895→      if (segTimes.length > 0) {
   896→        const avgMs = segTimes.reduce((a,b)=>a+b,0) / segTimes.length;
   897→        const etaMin = Math.ceil(avgMs * remaining / 60000);
   898→        const etaSec = Math.ceil(avgMs * remaining / 1000);
   899→        etaStr = etaSec < 120 ? `，预计剩余 ${etaSec} 秒` : `，预计剩余 ${etaMin} 分钟`;
   900→      }
   901→      const progressStr = `[${done}/${totalSegs}]`;
   902→
   903→      sendSubtitleStatus(`${progressStr} 提取音频 ${i + 1}/${segCount}${etaStr}`);
   904→      const extractOk = await new Promise((resolve) => {
   905→        const p = spawn(ffmpeg, ['-y', '-ss', String(startSec), '-t', String(SEGMENT_DURATION), '-i', videoPath, '-vn', '-ar', '16000', '-ac', '1', '-f', 'wav', segFile]);
   906→        p.on('close', code => resolve(code === 0));
   907→        p.on('error', () => resolve(false));
   908→      });
   909→      if (!extractOk) continue;
   910→
   911→      let result;
   912→
   913→      const statusBase = `${progressStr} 转录 ${i + 1}/${segCount}${etaStr}`;
   914→
   915→      // 检测语言后更新前端
   916→      function handleLangDetect(result) {
   917→        if (result && result.success && result.detectedLang && !detectedLanguage) {
   918→          const rawLang = result.detectedLang.toLowerCase();
   919→          detectedLanguage = LANG_NAME_TO_CODE[rawLang] || rawLang;
   920→          currentLangGlobal = detectedLanguage;
   921→          const langName = LANGUAGES[detectedLanguage]?.name || result.detectedLang;
   922→          sendSubtitleStatus(`${statusBase} - 检测到${langName}`);
   923→          if (mainWindow && !mainWindow.isDestroyed()) {
   924→            mainWindow.webContents.send('lang-detected', detectedLanguage);
   925→          }
   926→        }
   927→      }
   928→
   929→      // 优先级1: Groq Whisper API
   930→      if (groqKeyAvailable) {
   931→        sendSubtitleStatus(`${statusBase} [Groq]`);
   932→        result = await transcribeGroqPython(segFile, null);
   933→        handleLangDetect(result);
   934→      }
   935→      // 优先级2: 本地 whisper（支持所有语言）
   936→      if (!result || !result.success) {
   937→        sendSubtitleStatus(`${statusBase} [本地Whisper]`);
   938→        result = await transcribeLocalWhisper(segFile, null);
   939→        handleLangDetect(result);
   940→      }
   941→      // 优先级3: FunASR（仅中文）
   942→      if (!result || !result.success) {
   943→        if (!detectedLanguage || detectedLanguage === 'zh' || detectedLanguage === 'chinese') {
   944→          sendSubtitleStatus(`${statusBase} [FunASR]`);
   945→          const ok = await ensureFunasr();
   946→          if (ok) result = await transcribeFile(segFile);
   947→        }
   948→      }
   949→      try { fs.unlinkSync(segFile); } catch(e) {}
   950→      segTimes.push(Date.now() - segStart);
   951→
   952→      if (!result || !result.success || !result.text) continue;
   953→
   954→      const text = result.text;
   955→      const timestamps = result.timestamp || [];
   956→      const apiWords = result.words || []; // Groq word-level timestamps
   957→
   958→      if (apiWords.length > 0) {
   959→        // === 精准模式：直接使用 Whisper 返回的逐词时间戳 ===
   960→        // 按句子分组（遇到句末标点切分）
   961→        let sentences = [], curWords = [];
   962→        for (const aw of apiWords) {
   963→          curWords.push(aw);
   964→          const w = aw.w.trim();
   965→          if (/[.!?。！？]$/.test(w) && curWords.length > 0) {
   966→            sentences.push([...curWords]);
   967→            curWords = [];
   968→          }
   969→        }
   970→        if (curWords.length > 0) sentences.push(curWords);
   971→
   972→        for (const sentWords of sentences) {
   973→          const sentText = sentWords.map(w => w.w).join(' ').replace(/\s+/g, ' ').trim();
   974→          const sStart = sentWords[0].s / 1000;
   975→          const sEnd = sentWords[sentWords.length - 1].e / 1000;
   976→          vtt += `${fmtVTT(startSec + sStart)} --> ${fmtVTT(startSec + sEnd)}\n${sentText}\n\n`;
   977→
   978→          // 逐词时间戳：直接用 API 返回的精准时间，词文本也直接用 API 的
   979→          for (const aw of sentWords) {
   980→            wordTimings.push({
   981→              w: aw.w.trim(),
   982→              s: +(startSec + aw.s / 1000).toFixed(3),
   983→              e: +(startSec + aw.e / 1000).toFixed(3)
   984→            });
   985→          }
   986→        }
   987→      } else if (timestamps.length > 0) {
   988→        // === 回退模式：segment-level timestamps，按字符比例估算 ===
   989→        const noSpace = text.replace(/\s/g, '');
   990→        let sentences = [];
   991→        let lastPos = 0, charCount = 0;
   992→        for (let j = 0; j < text.length; j++) {
   993→          if (/\s/.test(text[j])) continue;
   994→          charCount++;
   995→          if (/[.!?。！？]/.test(text[j])) {
   996→            const s = text.slice(lastPos, j + 1).trim();
   997→            const charEnd = charCount;
   998→            const charStart = charEnd - s.replace(/\s/g, '').length;
   999→            if (s) sentences.push({ text: s, charStart, charEnd });
  1000→            lastPos = j + 1;
  1001→          }
  1002→        }
  1003→        if (lastPos < text.length) {
  1004→          const s = text.slice(lastPos).trim();
  1005→          if (s) sentences.push({ text: s, charStart: charCount - s.replace(/\s/g, '').length, charEnd: charCount });
  1006→        }
  1007→        if (!sentences.length) sentences.push({ text, charStart: 0, charEnd: noSpace.length });
  1008→
  1009→        const totalChars = noSpace.length;
  1010→        const totalTs = timestamps.length;
  1011→
  1012→        for (const sent of sentences) {
  1013→          const tsStart = Math.floor(sent.charStart / totalChars * totalTs);
  1014→          const tsEnd = Math.min(Math.floor(sent.charEnd / totalChars * totalTs), totalTs - 1);
  1015→          const startMs = timestamps[tsStart] ? timestamps[tsStart][0] : 0;
  1016→          const endMs = timestamps[tsEnd] ? timestamps[tsEnd][1] : startMs + 1000;
  1017→          const start = fmtVTT(startSec + startMs / 1000);
  1018→          const end = fmtVTT(startSec + endMs / 1000);
  1019→          vtt += `${start} --> ${end}\n${sent.text}\n\n`;
  1020→
  1021→          const words = sent.text.split(/(\s+)/).filter(w => !/^\s*$/.test(w));
  1022→          let charPos = sent.charStart;
  1023→          for (const w of words) {
  1024→            const wLen = w.replace(/\s/g, '').length;
  1025→            const wTsStart = Math.floor(charPos / totalChars * totalTs);
  1026→            const wTsEnd = Math.min(Math.floor((charPos + wLen) / totalChars * totalTs), totalTs - 1);
  1027→            const wStartMs = timestamps[wTsStart] ? timestamps[wTsStart][0] : 0;
  1028→            const wEndMs = timestamps[wTsEnd] ? timestamps[wTsEnd][1] : wStartMs + 200;
  1029→            wordTimings.push({
  1030→              w, s: +(startSec + wStartMs / 1000).toFixed(3), e: +(startSec + wEndMs / 1000).toFixed(3)
  1031→            });
  1032→            charPos += wLen;
  1033→          }
  1034→        }
  1035→      } else {
  1036→        const sentences = (text.match(/[^.!?。！？]+[.!?。！？]+/g) || [text]).map(s => s.trim()).filter(s => s.length > 0);
  1037→        if (!sentences.length) { allText += text; continue; }
  1038→        const segDur = Math.min(SEGMENT_DURATION, duration - startSec);
  1039→        const totalLen = sentences.reduce((a, s) => a + s.length, 0);
  1040→        let t = startSec;
  1041→        for (const s of sentences) {
  1042→          const seg = (s.length / totalLen) * segDur;
  1043→          const start = fmtVTT(t);
  1044→          t += seg;
  1045→          const end = fmtVTT(t);
  1046→          vtt += `${start} --> ${end}\n${s}\n\n`;
  1047→        }
  1048→      }
  1049→      allText += text;
  1050→    }
  1051→
  1052→    // 根据用户设置决定是否翻译字幕
  1053→    const aiCfgTrans = loadAIConfig();
  1054→    const translateMode = aiCfgTrans.subtitleTranslateMode || 'none';
  1055→    const needTranslate = translateMode !== 'none' && detectedLanguage && detectedLanguage !== 'zh' && detectedLanguage !== 'chinese';
  1056→
  1057→    if (needTranslate && vtt) {
  1058→      sendSubtitleStatus('正在翻译为中文...');
  1059→      const lines = vtt.split('\n');
  1060→      const textLines = [];
  1061→      const textIndices = [];
  1062→      for (let li = 0; li < lines.length; li++) {
  1063→        const line = lines[li];
  1064→        if (line.trim() && !line.includes('-->') && !/^\d+$/.test(line.trim())) {
  1065→          textLines.push(line.trim());
  1066→          textIndices.push(li);
  1067→        }
  1068→      }
  1069→      const translated = {};
  1070→
  1071→      if (translateMode === 'ai') {
  1072→        // AI 大模型批量翻译
  1073→        const BATCH = 20;
  1074→        let pos = 0;
  1075→        while (pos < textLines.length) {
  1076→          const chunk = textLines.slice(pos, pos + BATCH);
  1077→          const start = pos;
  1078→          pos += chunk.length;
  1079→          const numbered = chunk.map((t, i) => `${i + 1}. ${t}`).join('\n');
  1080→          sendSubtitleStatus(`正在翻译为中文... (${Math.min(pos, textLines.length)}/${textLines.length})`);
  1081→          const langName = LANGUAGES[detectedLanguage]?.name || detectedLanguage;
  1082→          const result = await callAI(`将以下${langName}字幕逐行译为中文，保持行号格式"序号. 中文"：\n${numbered}`);
  1083→          if (result) {
  1084→            for (const rl of result.trim().split('\n')) {
  1085→              const m = rl.match(/^(\d+)\.\s*(.+)/);
  1086→              if (m) {
  1087→                const idx = parseInt(m[1]) - 1;
  1088→                if (idx >= 0 && idx < chunk.length) translated[textIndices[start + idx]] = m[2].trim();
  1089→              }
  1090→            }
  1091→          }
  1092→        }
  1093→      } else if (translateMode === 'baidu') {
  1094→        // 百度翻译批量
  1095→        const MAX_CHARS = 4000;
  1096→        let pos = 0;
  1097→        while (pos < textLines.length) {
  1098→          let chunk = [], chunkLen = 0, start = pos;
  1099→          while (pos < textLines.length && chunkLen + textLines[pos].length < MAX_CHARS) {
  1100→            chunk.push(textLines[pos]); chunkLen += textLines[pos].length + 1; pos++;
  1101→          }
  1102→          if (!chunk.length) { chunk.push(textLines[pos]); pos++; }
  1103→          sendSubtitleStatus(`正在翻译为中文... (${Math.min(pos, textLines.length)}/${textLines.length})`);
  1104→          const result = await baiduTranslate(chunk.join('\n'), detectedLanguage || 'auto');
  1105→          if (result) {
  1106→            const resultLines = result.split('\n');
  1107→            for (let j = 0; j < chunk.length; j++) {
  1108→              if (resultLines[j]) translated[textIndices[start + j]] = resultLines[j];
  1109→            }
  1110→          }
  1111→          await new Promise(r => setTimeout(r, 1100));
  1112→        }
  1113→      }
  1114→
  1115→      // 重建VTT，在原文后插入中文
  1116→      let newVtt = '';
  1117→      for (let li = 0; li < lines.length; li++) {
  1118→        newVtt += lines[li] + '\n';
  1119→        if (translated[li]) newVtt += translated[li] + '\n';
  1120→      }
  1121→      vtt = newVtt;
  1122→    }
  1123→
  1124→    return { vttText: vtt, allText, wordTimings };
  1125→  }
  1126→
  1127→  // 前台转录前40分钟
  1128→  const fg = await transcribeRange(0, fgSegCount);
  1129→
  1130→  if (!fg.allText) {
  1131→    try { fs.rmSync(tmpDir, { recursive: true }); } catch(e) {}
  1132→    sendSubtitleStatus('语音识别失败');
  1133→    return null;
  1134→  }
  1135→
  1136→  // 先写入前台部分
  1137→  let fullVtt = 'WEBVTT\n\n' + fg.vttText;
  1138→  fs.writeFileSync(vttPath, fullVtt, 'utf-8');
  1139→  const wtsPath = vttPath.replace(/\.vtt$/, '.wordts.json');
  1140→  let allWordTimings = fg.wordTimings || [];
  1141→  fs.writeFileSync(wtsPath, JSON.stringify(allWordTimings), 'utf-8');
  1142→
  1143→  if (fgSegCount < segCount) {
  1144→    sendSubtitleStatus(`前 ${Math.ceil(FOREGROUND_LIMIT/60)} 分钟字幕已就绪，剩余部分后台转录中...`);
  1145→    // 后台继续转录剩余部分
  1146→    (async () => {
  1147→      const bg = await transcribeRange(fgSegCount, segCount);
  1148→      if (bg.vttText) {
  1149→        fullVtt += bg.vttText;
  1150→        fs.writeFileSync(vttPath, fullVtt, 'utf-8');
  1151→        allWordTimings = allWordTimings.concat(bg.wordTimings || []);
  1152→        fs.writeFileSync(wtsPath, JSON.stringify(allWordTimings), 'utf-8');
  1153→      }
  1154→      try { fs.rmSync(tmpDir, { recursive: true }); } catch(e) {}
  1155→      sendSubtitleStatus(`全部转录完成，共 ${segCount} 段`);
  1156→      // 通知前端刷新字幕
  1157→      if (mainWindow && !mainWindow.isDestroyed()) {
  1158→        mainWindow.webContents.send('subtitle-updated', vttPath);
  1159→      }
  1160→    })();
  1161→  } else {
  1162→    try { fs.rmSync(tmpDir, { recursive: true }); } catch(e) {}
  1163→    sendSubtitleStatus(`转录完成，共 ${segCount} 段`);
  1164→  }
  1165→
  1166→  return vttPath;
  1167→});
  1168→
  1169→function fmtVTT(sec) {
  1170→  const h = Math.floor(sec / 3600), m = Math.floor((sec % 3600) / 60), s = sec % 60;
  1171→  return String(h).padStart(2,'0') + ':' + String(m).padStart(2,'0') + ':' + s.toFixed(3).padStart(6,'0');
  1172→}
