   170→private fun parseContentDispositionFileName(contentDisposition: String?): String? {
   171→    if (contentDisposition == null) return null
   172→    Regex("filename\*=(?:UTF-8''|utf-8'')(.+)").find(contentDisposition)?.let {
   173→        return try { java.net.URLDecoder.decode(it.groupValues[1], "UTF-8") } catch (_: Exception) { null }
   174→    }
   175→    Regex("filename=\"?([^\"]+)\"?").find(contentDisposition)?.let {
   176→        return it.groupValues[1].trim()
   177→    }
   178→    return null
   179→}
