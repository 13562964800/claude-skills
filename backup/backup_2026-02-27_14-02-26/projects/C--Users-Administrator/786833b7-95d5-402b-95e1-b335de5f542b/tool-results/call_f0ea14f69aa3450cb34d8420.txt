    17→data class SubpoenaInfo(
    18→    val caseNumber: String?,
    19→    val courtName: String?,
    20→    val hearingTime: Long?,
    21→    val partyNames: String?,
    22→    val location: String?
    23→)
    24→
    25→class PdfParser(private val context: Context) {
    26→
    27→    suspend fun parseSubpoena(filePath: String): SubpoenaInfo? = withContext(Dispatchers.IO) {
    28→        val text = extractText(filePath)
    29→        if (text.isNullOrBlank()) return@withContext null
    30→        parseSubpoenaText(text)
    31→    }
    32→
    33→    private suspend fun extractText(filePath: String): String? {
    34→        // 先尝试 iText 提取
    35→        val iTextResult = extractWithIText(filePath)
    36→        if (!iTextResult.isNullOrBlank() && iTextResult.any { it.code > 0x4E00 }) {
    37→            return iTextResult
    38→        }
    39→        // 回退到 OCR
    40→        return extractWithOcr(filePath)
    41→    }
    42→
    43→    private fun extractWithIText(filePath: String): String? {
    44→        return try {
    45→            val reader = com.itextpdf.kernel.pdf.PdfReader(filePath)
    46→            val doc = com.itextpdf.kernel.pdf.PdfDocument(reader)
    47→            val sb = StringBuilder()
    48→            for (i in 1..doc.numberOfPages) {
    49→                val text = com.itextpdf.kernel.pdf.canvas.parser.PdfTextExtractor.getTextFromPage(doc.getPage(i))
    50→                sb.appendLine(text)
    51→            }
    52→            doc.close()
    53→            sb.toString()
    54→        } catch (_: Exception) { null }
    55→    }
    56→
    57→    private suspend fun extractWithOcr(filePath: String): String? {
    58→        val file = File(filePath)
    59→        if (!file.exists()) return null
    60→        val fd = ParcelFileDescriptor.open(file, ParcelFileDescriptor.MODE_READ_ONLY)
    61→        val renderer = PdfRenderer(fd)
    62→        val sb = StringBuilder()
    63→        val recognizer = TextRecognition.getClient(ChineseTextRecognizerOptions.Builder().build())
    64→
    65→        for (i in 0 until renderer.pageCount) {
    66→            val page = renderer.openPage(i)
    67→            val bitmap = Bitmap.createBitmap(page.width * 2, page.height * 2, Bitmap.Config.ARGB_8888)
    68→            page.render(bitmap, null, null, PdfRenderer.Page.RENDER_MODE_FOR_DISPLAY)
    69→            page.close()
    70→
    71→            val text = suspendCancellableCoroutine<String?> { cont ->
    72→                val image = InputImage.fromBitmap(bitmap, 0)
    73→                recognizer.process(image)
    74→                    .addOnSuccessListener { result -> cont.resume(result.text) }
    75→                    .addOnFailureListener { cont.resume(null) }
    76→            }
    77→            if (text != null) sb.appendLine(text)
    78→            bitmap.recycle()
    79→        }
    80→        renderer.close()
    81→        fd.close()
    82→        return sb.toString().takeIf { it.isNotBlank() }
    83→    }
    84→
    85→    private fun parseSubpoenaText(text: String): SubpoenaInfo {
    86→        val caseNumber = Regex("[（(]\d{4}[）)][^\s，。]+号").find(text)?.value
    87→        val courtName = Regex("(.+?法院)").find(text)?.groupValues?.get(1)
    88→        val partyNames = Regex("(?:被告|原告|当事人)[：:]?\s*(\S+)").find(text)?.groupValues?.get(1)
    89→        val location = Regex("(?:应到地点|地[点址])[：:]?\s*(\S+)").find(text)?.groupValues?.get(1)
    90→
    91→        var hearingTime: Long? = null
    92→        val timeMatch = Regex("(?:应到时间|开庭时间|定于)[：:]?\s*(\d{4})[年-](\d{1,2})[月-](\d{1,2})[日]?\s*(\d{1,2})[时点:：](\d{0,2})").find(text)
    93→        if (timeMatch != null) {
    94→            val cal = Calendar.getInstance().apply {
    95→                set(timeMatch.groupValues[1].toInt(),
    96→                    timeMatch.groupValues[2].toInt() - 1,
    97→                    timeMatch.groupValues[3].toInt(),
    98→                    timeMatch.groupValues[4].toInt(),
    99→                    timeMatch.groupValues[5].takeIf { it.isNotEmpty() }?.toInt() ?: 0, 0)
   100→                set(Calendar.MILLISECOND, 0)
   101→            }
   102→            hearingTime = cal.timeInMillis
   103→        }
   104→
   105→        return SubpoenaInfo(caseNumber, courtName, hearingTime, partyNames, location)
   106→    }
   107→}
