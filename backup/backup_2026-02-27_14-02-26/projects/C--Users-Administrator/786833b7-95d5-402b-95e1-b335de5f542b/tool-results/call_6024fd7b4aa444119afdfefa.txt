   176→
   177→private fun renderPdfPages(filePath: String): List<Bitmap> {
   178→    val file = File(filePath)
   179→    if (!file.exists()) return emptyList()
   180→    val fd = ParcelFileDescriptor.open(file, ParcelFileDescriptor.MODE_READ_ONLY)
   181→    val renderer = PdfRenderer(fd)
   182→    val bitmaps = mutableListOf<Bitmap>()
   183→    for (i in 0 until renderer.pageCount) {
   184→        val page = renderer.openPage(i)
   185→        val bitmap = Bitmap.createBitmap(page.width * 2, page.height * 2, Bitmap.Config.ARGB_8888)
   186→        page.render(bitmap, null, null, PdfRenderer.Page.RENDER_MODE_FOR_DISPLAY)
   187→        page.close()
   188→        bitmaps.add(bitmap)
   189→    }
   190→    renderer.close()
   191→    fd.close()
   192→    return bitmaps
   193→}
