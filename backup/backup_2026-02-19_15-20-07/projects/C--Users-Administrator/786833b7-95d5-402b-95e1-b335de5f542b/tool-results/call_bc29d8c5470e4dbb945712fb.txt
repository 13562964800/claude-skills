   252→@Composable
   253→fun PdfCard(pdf: PdfDocumentEntity, onClick: () -> Unit) {
   254→    val dateFormat = remember { SimpleDateFormat("MM-dd HH:mm", Locale.CHINA) }
   255→    Surface(
   256→        modifier = Modifier
   257→            .fillMaxWidth()
   258→            .clip(RoundedCornerShape(6.dp))
   259→            .clickable(onClick = onClick),
   260→        color = ParchmentBg,
   261→        shape = RoundedCornerShape(6.dp)
   262→    ) {
   263→        Row(
   264→            modifier = Modifier.padding(10.dp),
   265→            verticalAlignment = Alignment.CenterVertically
   266→        ) {
   267→            Text(
   268→                if (pdf.isSubpoena) "📋" else "📄",
   269→                style = MaterialTheme.typography.titleLarge
   270→            )
   271→            Spacer(modifier = Modifier.width(8.dp))
   272→            Column(modifier = Modifier.weight(1f)) {
   273→                Text(
   274→                    pdf.fileName,
   275→                    style = MaterialTheme.typography.bodyMedium,
   276→                    color = InkBlack,
   277→                    maxLines = 1,
   278→                    overflow = TextOverflow.Ellipsis
   279→                )
   280→                Text(
   281→                    "${dateFormat.format(Date(pdf.downloadedAt))} · ${formatSize(pdf.fileSize)}",
   282→                    style = MaterialTheme.typography.labelSmall,
   283→                    color = CharcoalGray
   284→                )
   285→            }
   286→            if (pdf.isSubpoena && pdf.hearingInfo != null) {
   287→                Text("已同步日历", style = MaterialTheme.typography.labelSmall, color = JadeGreen)
   288→            }
   289→        }
   290→    }
   291→}
   292→
   293→private fun formatSize(bytes: Long): String {
   294→    return when {
   295→        bytes < 1024 -> "${bytes}B"
   296→        bytes < 1024 * 1024 -> "${bytes / 1024}KB"
   297→        else -> String.format("%.1fMB", bytes / (1024.0 * 1024.0))
   298→    }
   299→}
