    46→            contract = androidx.activity.result.contract.ActivityResultContracts.CreateDocument("text/csv")
    47→        ) { uri ->
    48→            uri?.let {
    49→                context.contentResolver.openOutputStream(it)?.use { output ->
    50→                    val exportPath = viewModel.exportData()

    52→                        java.io.File(exportPath).inputStream().use { input ->
    53→                            input.copyTo(output)
    54→                        }
    55→                        android.widget.Toast.makeText(context, "已导出数据", android.widget.Toast.LENGTH_SHORT)

    58→                }

    67→            confirmButton = {

    69→                    onClick = {

    84→
    85→    if (showAboutDialog) {
    86→        AlertDialog(

    94→                    Spacer(modifier = Modifier.height(8.dp))

    98→                        color = CinnabarRed
    99→                    )
   100→                    Spacer(modifier = Modifier.height(4.dp))
   101→                    Text("版本 1.0.0", style = MaterialTheme.typography.bodyMedium)

   110→                    Text(
   111→                        "本应用帮助您管理法院送达短信、下载和预览法律文书，并自动同步开庭信息到系统日历。",

   113→                        textAlign = TextAlign.Center
   114→                    )
   115→                }

   124→
   125→    Scaffold(
