   353→    /**
   354→     * 创建ZIP文件
   355→     */
   356→    private suspend fun createZipFile(
   357→        docs: List<DocumentRecord>,
   358→        progress: ProgressDialog
   359→    ): File = withContext(Dispatchers.IO) {
   360→        val timestamp = System.currentTimeMillis()
   361→        val zipFile = File(requireContext().cacheDir, "court_docs_$timestamp.zip")
   362→
   363→        var processedCount = 0
   364→
   365→        ZipOutputStream(zipFile.outputStream()).use { zos ->
   366→            for (doc in docs) {
   367→                val file = File(doc.filePath)
   368→                if (!file.exists()) {
   369→                    processedCount++
   370→                    withContext(Dispatchers.Main) {
   371→                        progress.progress = processedCount
   372→                    }
   373→                    delay(10)
   374→                    continue
   375→                }
   376→
   377→                val entry = ZipEntry(doc.fileName)
   378→                zos.putNextEntry(entry)
   379→                file.inputStream().use { input ->
   380→                    input.copyTo(zos)
   381→                }
   382→                zos.closeEntry()
   383→
   384→                processedCount++
   385→                withContext(Dispatchers.Main) {
   386→                    progress.progress = processedCount
   387→                }
   388→                delay(50)
   389→            }
   390→        }
   391→
   392→        return@withContext zipFile
   393→    }
