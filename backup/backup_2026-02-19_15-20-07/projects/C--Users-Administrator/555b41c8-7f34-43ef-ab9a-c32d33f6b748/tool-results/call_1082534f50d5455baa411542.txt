    25→        // 回写 courtName/partyName 到短信记录

    29→
    30→        // 同步开庭信息到日历
    31→        if (info.hearingTime > 0) {

    33→            Log.d(TAG, "SmsProcessor: calendar synced=$synced")
    34→        }
    35→
    36→        // 提取URL
    37→        val urls = TextParser.extractUrls(sms.body)

    40→        if (urls.isNotEmpty()) {
    41→            // 直接启动WebView处理所有PDF下载

    45→                        putExtra("url", url)

    47→                        putExtra("autoMode", true) // 自动模式
    48→                        addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
    49→                    }
    50→                    // 延迟启动，避免多个WebView同时打开
    51→                    delay(500)

    53→                        context.startActivity(intent)

    58→                }

    60→        }
    61→
    62→        // 重新读取，避免覆盖 PdfParser 设置的 hasSummons
    63→        val updated = db.smsDao().getById(sms.id) ?: sms
    64→        db.smsDao().update(updated.copy(
    65→            parseStatus = 1,
