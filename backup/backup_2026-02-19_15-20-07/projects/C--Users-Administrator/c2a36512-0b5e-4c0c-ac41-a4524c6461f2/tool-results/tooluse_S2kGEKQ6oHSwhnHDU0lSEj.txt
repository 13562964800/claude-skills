    89	export function injectHistoryImagesIntoMessages(
    90	  messages: AgentMessage[],
    91	  historyImagesByIndex: Map<number, ImageContent[]>,
    92	): boolean {
    93	  if (historyImagesByIndex.size === 0) return false;
    94	  let didMutate = false;
    95	
    96	  for (const [msgIndex, images] of historyImagesByIndex) {
    97	    // Bounds check: ensure index is valid before accessing
    98	    if (msgIndex < 0 || msgIndex >= messages.length) continue;
    99	    const msg = messages[msgIndex];
   100	    if (msg && msg.role === "user") {
   101	      // Convert string content to array format if needed
   102	      if (typeof msg.content === "string") {
   103	        msg.content = [{ type: "text", text: msg.content }];
   104	        didMutate = true;
   105	      }
   106	      if (Array.isArray(msg.content)) {
   107	        // Check for existing image content to avoid duplicates across turns
   108	        const existingImageData = new Set(
   109	          msg.content
   110	            .filter(
   111	              (c): c is ImageContent =>
   112	                c != null &&
   113	                typeof c === "object" &&
   114	                c.type === "image" &&
   115	                typeof c.data === "string",
   116	            )
   117	            .map((c) => c.data),
   118	        );
   119	        for (const img of images) {
   120	          // Only add if this image isn't already in the message
   121	          if (!existingImageData.has(img.data)) {
   122	            msg.content.push(img);
   123	            didMutate = true;
   124	          }
   125	        }
   126	      }
   127	    }
   128	  }
   129	
   130	  return didMutate;
   131	}

   569	          runAbortController.abort(reason);
   570	        }
   571	        void activeSession.abort();
   572	      };
   573	      const abortable = <T>(promise: Promise<T>): Promise<T> => {
   574	        const signal = runAbortController.signal;
   575	        if (signal.aborted) {
   576	          return Promise.reject(makeAbortError(signal));
   577	        }
   578	        return new Promise<T>((resolve, reject) => {
   579	          const onAbort = () => {
   580	            signal.removeEventListener("abort", onAbort);
   581	            reject(makeAbortError(signal));
   582	          };
   583	          signal.addEventListener("abort", onAbort, { once: true });
   584	          promise.then(
   585	            (value) => {
   586	              signal.removeEventListener("abort", onAbort);
   587	              resolve(value);
   588	            },
   589	            (err) => {
   590	              signal.removeEventListener("abort", onAbort);
   591	              reject(err);
   592	            },
   593	          );
   594	        });
   595	      };
   596	
   597	      const subscription = subscribeEmbeddedPiSession({
   598	        session: activeSession,
   599	        runId: params.runId,
   600	        verboseLevel: params.verboseLevel,
   601	        reasoningMode: params.reasoningLevel ?? "off",
   602	        toolResultFormat: params.toolResultFormat,
   603	        shouldEmitToolResult: params.shouldEmitToolResult,
   604	        shouldEmitToolOutput: params.shouldEmitToolOutput,
   605	        onToolResult: params.onToolResult,
   606	        onReasoningStream: params.onReasoningStream,
   607	        onBlockReply: params.onBlockReply,
   608	        onBlockReplyFlush: params.onBlockReplyFlush,
   609	        blockReplyBreak: params.blockReplyBreak,
   610	        blockReplyChunking: params.blockReplyChunking,
   611	        onPartialReply: params.onPartialReply,
   612	        onAssistantMessageStart: params.onAssistantMessageStart,
   613	        onAgentEvent: params.onAgentEvent,
   614	        enforceFinalTag: params.enforceFinalTag,
   615	      });
   616	
   617	      const {
   618	        assistantTexts,
   619	        toolMetas,
   620	        unsubscribe,
   621	        waitForCompactionRetry,
   622	        getMessagingToolSentTexts,
   623	        getMessagingToolSentTargets,
   624	        didSendViaMessagingTool,
   625	        getLastToolError,
   626	      } = subscription;
   627	
   628	      const queueHandle: EmbeddedPiQueueHandle = {
   629	        queueMessage: async (text: string) => {
   630	          await activeSession.steer(text);
   631	        },
   632	        isStreaming: () => activeSession.isStreaming,
   633	        isCompacting: () => subscription.isCompacting(),
   634	        abort: abortRun,
   635	      };
   636	      setActiveEmbeddedRun(params.sessionId, queueHandle);
   637	
   638	      let abortWarnTimer: NodeJS.Timeout | undefined;
   639	      const isProbeSession = params.sessionId?.startsWith("probe-") ?? false;
   640	      const abortTimer = setTimeout(
   641	        () => {
   642	          if (!isProbeSession) {
   643	            log.warn(
   644	              `embedded run timeout: runId=${params.runId} sessionId=${params.sessionId} timeoutMs=${params.timeoutMs}`,
   645	            );
   646	          }
   647	          abortRun(true);
   648	          if (!abortWarnTimer) {
   649	            abortWarnTimer = setTimeout(() => {
   650	              if (!activeSession.isStreaming) return;
   651	              if (!isProbeSession) {
   652	                log.warn(
   653	                  `embedded run abort still streaming: runId=${params.runId} sessionId=${params.sessionId}`,
   654	                );
   655	              }
   656	            }, 10_000);
   657	          }
   658	        },
   659	        Math.max(1, params.timeoutMs),
   660	      );
   661	
   662	      let messagesSnapshot: AgentMessage[] = [];
   663	      let sessionIdUsed = activeSession.sessionId;
   664	      const onAbort = () => {
   665	        const reason = params.abortSignal ? getAbortReason(params.abortSignal) : undefined;
   666	        const timeout = reason ? isTimeoutError(reason) : false;
   667	        abortRun(timeout, reason);
   668	      };
   669	      if (params.abortSignal) {
   670	        if (params.abortSignal.aborted) {
   671	          onAbort();
   672	        } else {
   673	          params.abortSignal.addEventListener("abort", onAbort, {
   674	            once: true,
   675	          });
   676	        }
   677	      }
   678	
   679	      // Get hook runner once for both before_agent_start and agent_end hooks
   680	      const hookRunner = getGlobalHookRunner();
   681	
   682	      let promptError: unknown = null;
   683	      try {
   684	        const promptStartedAt = Date.now();
   685	
   686	        // Run before_agent_start hooks to allow plugins to inject context
   687	        let effectivePrompt = params.prompt;
   688	        if (hookRunner?.hasHooks("before_agent_start")) {
   689	          try {
   690	            const hookResult = await hookRunner.runBeforeAgentStart(
   691	              {
   692	                prompt: params.prompt,
   693	                messages: activeSession.messages,
   694	              },
   695	              {
   696	                agentId: params.sessionKey?.split(":")[0] ?? "main",
   697	                sessionKey: params.sessionKey,
   698	                workspaceDir: params.workspaceDir,
   699	                messageProvider: params.messageProvider ?? undefined,
   700	              },
   701	            );
   702	            if (hookResult?.prependContext) {
   703	              effectivePrompt = `${hookResult.prependContext}\n\n${params.prompt}`;
   704	              log.debug(
   705	                `hooks: prepended context to prompt (${hookResult.prependContext.length} chars)`,
   706	              );
   707	            }
   708	          } catch (hookErr) {
   709	            log.warn(`before_agent_start hook failed: ${String(hookErr)}`);
   710	          }
   711	        }
   712	
   713	        log.debug(`embedded run prompt start: runId=${params.runId} sessionId=${params.sessionId}`);
   714	        cacheTrace?.recordStage("prompt:before", {
   715	          prompt: effectivePrompt,
   716	          messages: activeSession.messages,
   717	        });
   718	
   719	        // Repair orphaned trailing user messages so new prompts don't violate role ordering.
   720	        const leafEntry = sessionManager.getLeafEntry();
   721	        if (leafEntry?.type === "message" && leafEntry.message.role === "user") {
   722	          if (leafEntry.parentId) {
   723	            sessionManager.branch(leafEntry.parentId);
   724	          } else {
   725	            sessionManager.resetLeaf();
   726	          }
   727	          const sessionContext = sessionManager.buildSessionContext();
   728	          activeSession.agent.replaceMessages(sessionContext.messages);
   729	          log.warn(
   730	            `Removed orphaned user message to prevent consecutive user turns. ` +
   731	              `runId=${params.runId} sessionId=${params.sessionId}`,
   732	          );
   733	        }
   734	
   735	        try {
   736	          // Detect and load images referenced in the prompt for vision-capable models.
   737	          // This eliminates the need for an explicit "view" tool call by injecting
   738	          // images directly into the prompt when the model supports it.
   739	          // Also scans conversation history to enable follow-up questions about earlier images.
   740	          const imageResult = await detectAndLoadPromptImages({
   741	            prompt: effectivePrompt,
   742	            workspaceDir: effectiveWorkspace,
   743	            model: params.model,
   744	            existingImages: params.images,
   745	            historyMessages: activeSession.messages,
   746	            maxBytes: MAX_IMAGE_BYTES,
   747	            // Enforce sandbox path restrictions when sandbox is enabled
   748	            sandboxRoot: sandbox?.enabled ? sandbox.workspaceDir : undefined,
   749	          });
   750	
   751	          // Inject history images into their original message positions.
   752	          // This ensures the model sees images in context (e.g., "compare to the first image").
   753	          const didMutate = injectHistoryImagesIntoMessages(
   754	            activeSession.messages,
   755	            imageResult.historyImagesByIndex,
   756	          );
   757	          if (didMutate) {
   758	            // Persist message mutations (e.g., injected history images) so we don't re-scan/reload.
   759	            activeSession.agent.replaceMessages(activeSession.messages);
   760	          }
   761	
   762	          cacheTrace?.recordStage("prompt:images", {
   763	            prompt: effectivePrompt,
   764	            messages: activeSession.messages,
   765	            note: `images: prompt=${imageResult.images.length} history=${imageResult.historyImagesByIndex.size}`,
   766	          });
   767	
   768	          const shouldTrackCacheTtl =
   769	            params.config?.agents?.defaults?.contextPruning?.mode === "cache-ttl" &&
   770	            isCacheTtlEligibleProvider(params.provider, params.modelId);
   771	          if (shouldTrackCacheTtl) {
   772	            appendCacheTtlTimestamp(sessionManager, {
   773	              timestamp: Date.now(),
   774	              provider: params.provider,
   775	              modelId: params.modelId,
   776	            });
   777	          }
   778	
   779	          // Only pass images option if there are actually images to pass
   780	          // This avoids potential issues with models that don't expect the images parameter
   781	          if (imageResult.images.length > 0) {
   782	            await abortable(activeSession.prompt(effectivePrompt, { images: imageResult.images }));
   783	          } else {
   784	            await abortable(activeSession.prompt(effectivePrompt));
   785	          }
   786	        } catch (err) {
   787	          promptError = err;
   788	        } finally {
   789	          log.debug(
   790	            `embedded run prompt end: runId=${params.runId} sessionId=${params.sessionId} durationMs=${Date.now() - promptStartedAt}`,
   791	          );
   792	        }
   793	
   794	        try {
   795	          await waitForCompactionRetry();
   796	        } catch (err) {
   797	          if (isAbortError(err)) {
   798	            if (!promptError) promptError = err;
   799	          } else {
   800	            throw err;
   801	          }
   802	        }
   803	
   804	        messagesSnapshot = activeSession.messages.slice();
   805	        sessionIdUsed = activeSession.sessionId;
   806	        cacheTrace?.recordStage("session:after", {
   807	          messages: messagesSnapshot,
   808	          note: promptError ? "prompt error" : undefined,
   809	        });
   810	        anthropicPayloadLogger?.recordUsage(messagesSnapshot, promptError);
   811	
   812	        // Run agent_end hooks to allow plugins to analyze the conversation
   813	        // This is fire-and-forget, so we don't await
   814	        if (hookRunner?.hasHooks("agent_end")) {
   815	          hookRunner
   816	            .runAgentEnd(
   817	              {
   818	                messages: messagesSnapshot,
   819	                success: !aborted && !promptError,
   820	                error: promptError ? describeUnknownError(promptError) : undefined,
   821	                durationMs: Date.now() - promptStartedAt,
   822	              },
   823	              {
   824	                agentId: params.sessionKey?.split(":")[0] ?? "main",
   825	                sessionKey: params.sessionKey,
   826	                workspaceDir: params.workspaceDir,
   827	                messageProvider: params.messageProvider ?? undefined,
   828	              },
   829	            )
   830	            .catch((err) => {
   831	              log.warn(`agent_end hook failed: ${err}`);
   832	            });
   833	        }
   834	      } finally {
   835	        clearTimeout(abortTimer);
   836	        if (abortWarnTimer) clearTimeout(abortWarnTimer);
   837	        unsubscribe();
   838	        clearActiveEmbeddedRun(params.sessionId, queueHandle);
   839	        params.abortSignal?.removeEventListener?.("abort", onAbort);
   840	      }
   841	
   842	      const lastAssistant = messagesSnapshot
   843	        .slice()
   844	        .reverse()
   845	        .find((m) => (m as AgentMessage)?.role === "assistant") as AssistantMessage | undefined;
   846	
   847	      const toolMetasNormalized = toolMetas
   848	        .filter(
   849	          (entry): entry is { toolName: string; meta?: string } =>
   850	            typeof entry.toolName === "string" && entry.toolName.trim().length > 0,
   851	        )
   852	        .map((entry) => ({ toolName: entry.toolName, meta: entry.meta }));
   853	
   854	      return {
   855	        aborted,
   856	        timedOut,
   857	        promptError,
   858	        sessionIdUsed,
   859	        systemPromptReport,
   860	        messagesSnapshot,
   861	        assistantTexts,
   862	        toolMetas: toolMetasNormalized,
   863	        lastAssistant,
   864	        lastToolError: getLastToolError?.(),
   865	        didSendViaMessagingTool: didSendViaMessagingTool(),
   866	        messagingToolSentTexts: getMessagingToolSentTexts(),
   867	        messagingToolSentTargets: getMessagingToolSentTargets(),
   868	        cloudCodeAssistFormatError: Boolean(
   869	          lastAssistant?.errorMessage && isCloudCodeAssistFormatError(lastAssistant.errorMessage),
   870	        ),
   871	        // Client tool call detected (OpenResponses hosted tools)
   872	        clientToolCall: clientToolCallDetected ?? undefined,
   873	      };
   874	    } finally {
   875	      // Always tear down the session (and release the lock) before we leave this attempt.
   876	      sessionManager?.flushPendingToolResults?.();
   877	      session?.dispose();
   878	      await sessionLock.release();
   879	    }
   880	  } finally {
   881	    restoreSkillEnv?.();
   882	    process.chdir(prevCwd);
   883	  }
   884	}
