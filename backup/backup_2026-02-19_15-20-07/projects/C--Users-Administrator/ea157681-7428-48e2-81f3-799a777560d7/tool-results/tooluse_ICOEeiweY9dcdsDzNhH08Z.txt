   232→// VTT解析 (处理YouTube自动字幕的双行格式)
   233→function parseVTT(vtt) {
   234→  const raw = [];
   235→  for (const block of vtt.split(/\n\n+/)) {
   236→    const lines = block.trim().split('\n');
   237→    for (let i = 0; i < lines.length; i++) {
   238→      const m = lines[i].match(/(\d{2}):(\d{2}):(\d{2})\.(\d{3})\s*-->\s*(\d{2}):(\d{2}):(\d{2})\.(\d{3})/);
   239→      if (!m) continue;
   240→      const start = +m[1]*3600 + +m[2]*60 + +m[3] + +m[4]/1000;
   241→      const end = +m[5]*3600 + +m[6]*60 + +m[7] + +m[8]/1000;
   242→      // 跳过极短条目(YouTube自动字幕的过渡帧)
   243→      if (end - start < 0.05) break;
   244→      // 取最后一行非空文本(YouTube双行字幕中新内容在最后一行)
   245→      const textLines = lines.slice(i+1).map(l => l.replace(/<[^>]+>/g, '').trim()).filter(Boolean);
   246→      const text = textLines.length > 0 ? textLines[textLines.length - 1] : '';
   247→      if (text) raw.push({ start, end, text });
   248→      break;
   249→    }
   250→  }
   251→  // 合并连续相同文本，去重
   252→  const result = [];
   253→  for (const s of raw) {
   254→    if (result.length > 0 && result[result.length-1].text === s.text) {
   255→      result[result.length-1].end = s.end; // 延长时间
   256→    } else {
   257→      result.push({ ...s });
   258→    }
   259→  }
   260→  return result;
   261→}

   263→// 字幕同步 (用setInterval确保可靠触发)
   264→let syncTimer = null;
   265→function startSubtitleSync() {
   266→  if (syncTimer) clearInterval(syncTimer);
   267→  syncTimer = setInterval(() => {
   268→    if (!subtitles.length || currentMode !== 'learn' || videoPlayer.paused) return;
   269→    const t = videoPlayer.currentTime;
   270→    // 二分查找当前时间对应的字幕
   271→    let lo = 0, hi = subtitles.length - 1, idx = -1;
   272→    while (lo <= hi) {
   273→      const mid = (lo + hi) >> 1;
   274→      if (t >= subtitles[mid].start && t <= subtitles[mid].end) { idx = mid; break; }
   275→      if (t < subtitles[mid].start) hi = mid - 1;
   276→      else lo = mid + 1;
   277→    }
   278→    if (idx < 0) {
   279→      // 找最近的前一条字幕
   280→      for (let i = subtitles.length - 1; i >= 0; i--) {
   281→        if (subtitles[i].start <= t) { idx = i; break; }
   282→      }
   283→      if (idx < 0) idx = 0;
   284→    }
   285→    if (idx !== currentSubIndex) {
   286→      currentSubIndex = idx;
   287→      currentSentence = subtitles[idx].text;
   288→      renderSubtitle(currentSentence);
   289→      aiAnalyzeBtn.style.display = 'inline-block';
   290→      aiResult.style.display = 'none';
   291→      highlightSubtitleList(idx);
   292→    }
   293→  }, 100);
   294→}
   295→videoPlayer.addEventListener('play', startSubtitleSync);
   296→videoPlayer.addEventListener('seeked', () => { currentSubIndex = -1; });

   302→    if (/^\s+$/.test(w)) { subtitleText.appendChild(document.createTextNode(' ')); continue; }
   303→    const span = document.createElement('span');
   304→    span.className = 'word'; span.textContent = w;
   305→    span.addEventListener('click', (e) => {
   306→      e.stopPropagation();
   307→      const clean = cleanWord(w);
   308→      if (clean) showWordInline(clean);
   309→    });
   310→    subtitleText.appendChild(span);
   311→  }
   312→}

   442→async function showWordInline(word) {
   443→  wordInlinePanel.style.display = 'block';
   444→  inlineWord.textContent = word;
   445→  inlinePhonetic.textContent = '';
   446→  inlineTranslation.textContent = '查询中...';
   447→  inlineDefs.innerHTML = '';
   448→  inlineExamples.innerHTML = '';
   449→  inlineDefsSection.style.display = 'none';
   450→  inlineExamplesSection.style.display = 'none';
   451→  speak(word);
   452→
   453→  if (currentLang === 'en') {
   454→    const [ecdict, online, aiTrans] = await Promise.all([
   455→      ipcRenderer.invoke('lookup-word', word),
   456→      ipcRenderer.invoke('fetch-examples', word),
   457→      ipcRenderer.invoke('ai-translate-word', word, currentSentence)
   458→    ]);
   459→    if (ecdict?.phonetic) inlinePhonetic.textContent = '/' + ecdict.phonetic + '/';
   460→    else if (online?.phonetic) inlinePhonetic.textContent = online.phonetic;
   461→    if (aiTrans) {
   462→      inlineTranslation.textContent = aiTrans;
   463→    } else if (ecdict?.translation) {
   464→      inlineTranslation.textContent = ecdict.translation;
   465→    } else {
   466→      const tr = await ipcRenderer.invoke('translate-word', word, 'en');
   467→      inlineTranslation.textContent = tr?.translation || '未找到释义';
   468→    }
   469→    if (online) {
   470→      if (online.definitions?.length) {
   471→        inlineDefsSection.style.display = 'block';
   472→        inlineDefs.innerHTML = online.definitions.map(d =>
   473→          `<div class="def-item"><span class="pos">${d.partOfSpeech || ''}</span>${d.definition}</div>`
   474→        ).join('');
   475→      }
   476→      if (online.examples?.length) {
   477→        inlineExamplesSection.style.display = 'block';
   478→        inlineExamples.innerHTML = online.examples.map(e =>
   479→          `<div class="example">${e}</div>`
   480→        ).join('');
   481→      }
   482→    }
   483→  } else {
   484→    const [tr, aiTrans, online] = await Promise.all([
   485→      ipcRenderer.invoke('translate-word', word, currentLang),
   486→      ipcRenderer.invoke('ai-translate-word', word, currentSentence),
   487→      ipcRenderer.invoke('fetch-examples', word, currentLang)
   488→    ]);
   489→    if (online?.phonetic) inlinePhonetic.textContent = online.phonetic;
   490→    inlineTranslation.textContent = aiTrans || tr?.translation || '未找到释义';
   491→    if (online?.definitions?.length) {
   492→      inlineDefsSection.style.display = 'block';
   493→      inlineDefs.innerHTML = online.definitions.map(d =>
   494→        `<div class="def-item"><span class="pos">${d.partOfSpeech || ''}</span>${d.definition}</div>`
   495→      ).join('');
   496→    }
   497→    if (online?.examples?.length) {
   498→      inlineExamplesSection.style.display = 'block';
   499→      inlineExamples.innerHTML = online.examples.map(e =>
   500→        `<div class="example">${e}</div>`
   501→      ).join('');
   502→    }
   503→  }
   504→}
