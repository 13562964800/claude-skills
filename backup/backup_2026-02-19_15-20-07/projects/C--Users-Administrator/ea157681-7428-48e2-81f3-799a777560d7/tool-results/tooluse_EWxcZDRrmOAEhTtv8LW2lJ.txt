    40вҶ’function callAI(prompt) {
    41вҶ’  return new Promise((resolve) => {
    42вҶ’    const cfg = loadAIConfig();
    43вҶ’    if (!cfg.apiKey) { resolve(null); return; }
    44вҶ’
    45вҶ’    const preset = AI_PROVIDERS[cfg.provider] || {};
    46вҶ’    const baseUrl = cfg.baseUrl || preset.baseUrl || '';
    47вҶ’    const model = cfg.model || preset.model || '';
    48вҶ’    if (!baseUrl) { resolve(null); return; }
    49вҶ’
    50вҶ’    const isClaude = cfg.provider === 'claude';
    51вҶ’    const urlObj = new URL(isClaude ? baseUrl + '/messages' : baseUrl + '/chat/completions');
    52вҶ’    const isHttps = urlObj.protocol === 'https:';
    53вҶ’
    54вҶ’    let body, headers;
    55вҶ’    if (isClaude) {
    56вҶ’      body = JSON.stringify({
    57вҶ’        model, max_tokens: 1024,
    58вҶ’        messages: [{ role: 'user', content: prompt }]
    59вҶ’      });
    60вҶ’      headers = {
    61вҶ’        'Content-Type': 'application/json',
    62вҶ’        'x-api-key': cfg.apiKey,
    63вҶ’        'anthropic-version': '2023-06-01'
    64вҶ’      };
    65вҶ’    } else {
    66вҶ’      body = JSON.stringify({
    67вҶ’        model, max_tokens: 1024,
    68вҶ’        messages: [
    69вҶ’          { role: 'system', content: 'дҪ жҳҜдёҖдёӘдё“дёҡзҡ„еӨҡиҜӯиЁҖиҜҚе…ёе’ҢиҜӯиЁҖеӯҰд№ еҠ©жүӢгҖӮжүҖжңүйҮҠд№үгҖҒзҝ»иҜ‘гҖҒи§ЈйҮҠеҝ…йЎ»дҪҝз”ЁеҮҶзЎ®зҡ„дёӯж–ҮгҖӮеҜ№иӢұиҜӯеҚ•иҜҚпјҢиҜ·з»ҷеҮәеҮҶзЎ®зҡ„дёӯж–ҮеҜ№еә”иҜҚд№үпјҢдёҚиҰҒйҹіиҜ‘гҖӮ' },
    70вҶ’          { role: 'user', content: prompt }
    71вҶ’        ]
    72вҶ’      });
    73вҶ’      headers = {
    74вҶ’        'Content-Type': 'application/json',
    75вҶ’        'Authorization': 'Bearer ' + cfg.apiKey
    76вҶ’      };
    77вҶ’    }
    78вҶ’
    79вҶ’    const opts = {
    80вҶ’      hostname: urlObj.hostname, port: urlObj.port || (isHttps ? 443 : 80),
    81вҶ’      path: urlObj.pathname, method: 'POST', headers
    82вҶ’    };
    83вҶ’
    84вҶ’    const client = isHttps ? https : http;
    85вҶ’    const req = client.request(opts, (res) => {
    86вҶ’      let data = '';
    87вҶ’      res.on('data', d => data += d);
    88вҶ’      res.on('end', () => {
    89вҶ’        try {
    90вҶ’          const json = JSON.parse(data);
    91вҶ’          if (isClaude) {
    92вҶ’            resolve(json.content?.[0]?.text || null);
    93вҶ’          } else {
    94вҶ’            resolve(json.choices?.[0]?.message?.content || null);
    95вҶ’          }
    96вҶ’        } catch(e) { resolve(null); }
    97вҶ’      });
    98вҶ’    });
    99вҶ’    req.on('error', () => resolve(null));
   100вҶ’    req.setTimeout(15000, () => { req.destroy(); resolve(null); });
   101вҶ’    req.write(body);
   102вҶ’    req.end();
   103вҶ’  });
   104вҶ’}

   168вҶ’ipcMain.handle('download-video', async (_, url, lang) => {
   169вҶ’  const subLang = LANGUAGES[lang]?.sub || 'en';
   170вҶ’  return new Promise((resolve) => {
   171вҶ’    const infoArgs = ['--dump-json', '--no-playlist', url];
   172вҶ’    if (fs.existsSync(path.join(FFMPEG_DIR, 'ffmpeg.exe'))) infoArgs.unshift('--ffmpeg-location', FFMPEG_DIR);
   173вҶ’    const infoProc = spawn(YTDLP, infoArgs);
   174вҶ’    let infoOut = '';
   175вҶ’    infoProc.stdout.on('data', d => infoOut += d);
   176вҶ’    infoProc.on('close', (code) => {
   177вҶ’      let title = 'video_' + Date.now();
   178вҶ’      if (code === 0) { try { title = JSON.parse(infoOut).title || title; } catch(e) {} }
   179вҶ’      const safeName = title.replace(/[<>:"/\\|?*]/g, '_').substring(0, 60);
   180вҶ’      const outTemplate = path.join(DOWNLOADS, safeName + '.%(ext)s');
   181вҶ’      const args = [
   182вҶ’        '--ffmpeg-location', FFMPEG_DIR,
   183вҶ’        '-f', 'bestvideo[height<=720]+bestaudio/best[height<=720]/best',
   184вҶ’        '--merge-output-format', 'mp4',
   185вҶ’        '--write-sub', '--write-auto-sub', '--sub-lang', subLang,
   186вҶ’        '--sub-format', 'vtt', '--convert-subs', 'vtt',
   187вҶ’        '--no-playlist', '-o', outTemplate, url
   188вҶ’      ];
   189вҶ’      const proc = spawn(YTDLP, args);
   190вҶ’      let stderr = '', maxPercent = 0, streamCount = 0;
   191вҶ’      proc.stdout.on('data', (data) => {
   192вҶ’        const output = data.toString();
   193вҶ’        if (output.includes('Destination:')) streamCount++;
   194вҶ’        const matches = output.match(/(\d+\.?\d*)%/g);
   195вҶ’        if (matches) {
   196вҶ’          let percent = parseFloat(matches[matches.length - 1]);
   197вҶ’          if (streamCount >= 2) percent = 50 + percent / 2;
   198вҶ’          else if (streamCount === 1) percent = percent / 2;
   199вҶ’          if (percent > maxPercent) { maxPercent = percent; mainWindow.webContents.send('download-progress', { percent }); }
   200вҶ’        }
   201вҶ’        if (output.includes('Merging')) mainWindow.webContents.send('download-progress', { percent: 99 });
   202вҶ’      });
   203вҶ’      proc.stderr.on('data', d => stderr += d);
   204вҶ’      proc.on('close', (code2) => {
   205вҶ’        if (code2 !== 0) { resolve({ success: false, error: stderr || 'дёӢиҪҪеӨұиҙҘ' }); return; }
   206вҶ’        const videoFile = path.join(DOWNLOADS, safeName + '.mp4');
   207вҶ’        let subFile = null;
   208вҶ’        const possibleSubs = [
   209вҶ’          path.join(DOWNLOADS, safeName + '.' + subLang + '.vtt'),
   210вҶ’          path.join(DOWNLOADS, safeName + '.vtt'),
   211вҶ’        ];
   212вҶ’        try {
   213вҶ’          for (const f of fs.readdirSync(DOWNLOADS)) {
   214вҶ’            if (f.startsWith(safeName) && f.endsWith('.vtt')) possibleSubs.push(path.join(DOWNLOADS, f));
   215вҶ’          }
   216вҶ’        } catch(e) {}
   217вҶ’        for (const p of possibleSubs) { if (fs.existsSync(p)) { subFile = p; break; } }
   218вҶ’        resolve({ success: true, videoPath: fs.existsSync(videoFile) ? videoFile : null, subtitlePath: subFile, title: safeName });
   219вҶ’      });
   220вҶ’      proc.on('error', () => resolve({ success: false, error: 'yt-dlpжңӘжүҫеҲ°' }));
   221вҶ’    });

   442вҶ’const SEGMENT_DURATION = 600; // 10еҲҶй’ҹдёҖж®ө
   443вҶ’
   444вҶ’ipcMain.handle('generate-subtitle', async (_, videoPath) => {
   445вҶ’  const ready = await ensureFunasr();
   446вҶ’  if (!ready) return null;
   447вҶ’
   448вҶ’  sendSubtitleStatus('жӯЈеңЁеҲҶжһҗи§Ҷйў‘...');
   449вҶ’  const ffprobe = path.join(FFMPEG_DIR, 'ffprobe.exe');
   450вҶ’  const ffmpeg = path.join(FFMPEG_DIR, 'ffmpeg.exe');
   451вҶ’
   452вҶ’  const duration = await new Promise((resolve) => {
   453вҶ’    const p = spawn(ffprobe, ['-v', 'quiet', '-print_format', 'json', '-show_format', videoPath]);
   454вҶ’    let out = '';
   455вҶ’    p.stdout.on('data', d => out += d);
   456вҶ’    p.on('close', () => {
   457вҶ’      try { resolve(parseFloat(JSON.parse(out).format.duration) || 0); } catch(e) { resolve(0); }
   458вҶ’    });
   459вҶ’    p.on('error', () => resolve(0));
   460вҶ’  });
   461вҶ’  if (!duration) return null;
   462вҶ’
   463вҶ’  const durationMin = Math.ceil(duration / 60);
   464вҶ’  const segCount = Math.ceil(duration / SEGMENT_DURATION);
   465вҶ’  const tmpDir = path.join(DOWNLOADS, '_tmp_segments');
   466вҶ’  if (!fs.existsSync(tmpDir)) fs.mkdirSync(tmpDir, { recursive: true });
   467вҶ’
   468вҶ’  let allText = '';
   469вҶ’  let vtt = 'WEBVTT\n\n';
   470вҶ’
   471вҶ’  for (let i = 0; i < segCount; i++) {
   472вҶ’    const startSec = i * SEGMENT_DURATION;
   473вҶ’    const segFile = path.join(tmpDir, `seg_${i}.wav`);
   474вҶ’
   475вҶ’    sendSubtitleStatus(`жӯЈеңЁжҸҗеҸ–йҹійў‘зүҮж®ө ${i + 1}/${segCount}пјҲи§Ҷйў‘ ${durationMin} еҲҶй’ҹпјү`);
   476вҶ’    const extractOk = await new Promise((resolve) => {
   477вҶ’      const p = spawn(ffmpeg, ['-y', '-ss', String(startSec), '-t', String(SEGMENT_DURATION), '-i', videoPath, '-vn', '-ar', '16000', '-ac', '1', '-f', 'wav', segFile]);
   478вҶ’      p.on('close', code => resolve(code === 0));
   479вҶ’      p.on('error', () => resolve(false));
   480вҶ’    });
   481вҶ’    if (!extractOk) continue;
   482вҶ’
   483вҶ’    sendSubtitleStatus(`жӯЈеңЁиҪ¬еҪ•зүҮж®ө ${i + 1}/${segCount}пјҲи§Ҷйў‘ ${durationMin} еҲҶй’ҹпјү`);
   484вҶ’
   485вҶ’    // зЎ®дҝқжңҚеҠЎиҝҳеңЁпјҲеҸҜиғҪиў«дёҠдёҖж®өOOMжқҖдәҶпјү
   486вҶ’    const ok = await ensureFunasr();
   487вҶ’    if (!ok) { try { fs.unlinkSync(segFile); } catch(e) {} continue; }
   488вҶ’
   489вҶ’    const result = await transcribeFile(segFile);
   490вҶ’    try { fs.unlinkSync(segFile); } catch(e) {}
   491вҶ’
   492вҶ’    if (!result || !result.success || !result.text) continue;
   493вҶ’
   494вҶ’    // дҪҝз”Ё funasr иҝ”еӣһзҡ„зңҹе®һж—¶й—ҙжҲі
   495вҶ’    const text = result.text;
   496вҶ’    const timestamps = result.timestamp || [];
   497вҶ’
   498вҶ’    if (timestamps.length > 0) {
   499вҶ’      // жҢүж ҮзӮ№еҲҶеҸҘпјҢз”Ёж—¶й—ҙжҲізІҫзЎ®е®ҡдҪҚжҜҸеҸҘзҡ„иө·жӯўж—¶й—ҙ
   500вҶ’      const noSpace = text.replace(/\s/g, '');
   501вҶ’      const sentenceEnds = /[.!?гҖӮпјҒпјҹ]/g;
   502вҶ’      // еңЁеҺҹж–ҮдёӯжҢүж ҮзӮ№еҲҶеҸҘпјҢдҝқз•ҷз©әж ј
   503вҶ’      let sentences = [];
   504вҶ’      let lastPos = 0;
   505вҶ’      let charCount = 0;
   506вҶ’      for (let i = 0; i < text.length; i++) {
   507вҶ’        if (/\s/.test(text[i])) continue;
   508вҶ’        charCount++;
   509вҶ’        if (/[.!?гҖӮпјҒпјҹ]/.test(text[i])) {
   510вҶ’          const s = text.slice(lastPos, i + 1).trim();
   511вҶ’          const charEnd = charCount;
   512вҶ’          const charStart = charEnd - s.replace(/\s/g, '').length;
   513вҶ’          if (s) sentences.push({ text: s, charStart, charEnd });
   514вҶ’          lastPos = i + 1;
   515вҶ’        }
   516вҶ’      }
   517вҶ’      if (lastPos < text.length) {
   518вҶ’        const s = text.slice(lastPos).trim();
   519вҶ’        if (s) sentences.push({ text: s, charStart: charCount - s.replace(/\s/g, '').length, charEnd: charCount });
   520вҶ’      }
   521вҶ’      if (!sentences.length) sentences.push({ text, charStart: 0, charEnd: noSpace.length });
   522вҶ’
   523вҶ’      const totalChars = noSpace.length;
   524вҶ’      const totalTs = timestamps.length;
   525вҶ’
   526вҶ’      for (const sent of sentences) {
   527вҶ’        const tsStart = Math.floor(sent.charStart / totalChars * totalTs);
   528вҶ’        const tsEnd = Math.min(Math.floor(sent.charEnd / totalChars * totalTs), totalTs - 1);
   529вҶ’        const startMs = timestamps[tsStart] ? timestamps[tsStart][0] : 0;
   530вҶ’        const endMs = timestamps[tsEnd] ? timestamps[tsEnd][1] : startMs + 1000;
   531вҶ’        const start = fmtVTT(startSec + startMs / 1000);
   532вҶ’        const end = fmtVTT(startSec + endMs / 1000);
   533вҶ’        vtt += `${start} --> ${end}\n${sent.text}\n\n`;
   534вҶ’      }
   535вҶ’    } else {
   536вҶ’      // ж— ж—¶й—ҙжҲіеӣһйҖҖпјҡжҢүж–Үжң¬й•ҝеәҰдј°з®—
   537вҶ’      const sentences = (text.match(/[^.!?гҖӮпјҒпјҹ]+[.!?гҖӮпјҒпјҹ]+/g) || [text]).map(s => s.trim()).filter(s => s.length > 0);
   538вҶ’      if (!sentences.length) { allText += text; continue; }
   539вҶ’      const segDur = Math.min(SEGMENT_DURATION, duration - startSec);
   540вҶ’      const totalLen = sentences.reduce((a, s) => a + s.length, 0);
   541вҶ’      let t = startSec;
   542вҶ’      for (const s of sentences) {
   543вҶ’        const seg = (s.length / totalLen) * segDur;
   544вҶ’        const start = fmtVTT(t);
   545вҶ’        t += seg;
   546вҶ’        const end = fmtVTT(t);
   547вҶ’        vtt += `${start} --> ${end}\n${s}\n\n`;
   548вҶ’      }
   549вҶ’    }
   550вҶ’    allText += text;
   551вҶ’  }
   552вҶ’
   553вҶ’  // жё…зҗҶдёҙж—¶зӣ®еҪ•
   554вҶ’  try { fs.rmSync(tmpDir, { recursive: true }); } catch(e) {}
   555вҶ’
   556вҶ’  if (!allText) { sendSubtitleStatus('иҜӯйҹіиҜҶеҲ«еӨұиҙҘ'); return null; }
   557вҶ’
   558вҶ’  const vttPath = videoPath.replace(/\.[^.]+$/, '.vtt');
   559вҶ’  fs.writeFileSync(vttPath, vtt, 'utf-8');
   560вҶ’  sendSubtitleStatus(`иҪ¬еҪ•е®ҢжҲҗпјҢе…ұ ${segCount} ж®ө`);
   561вҶ’  return vttPath;
   562вҶ’});

   564вҶ’function fmtVTT(sec) {
   565вҶ’  const h = Math.floor(sec / 3600), m = Math.floor((sec % 3600) / 60), s = sec % 60;
   566вҶ’  return String(h).padStart(2,'0') + ':' + String(m).padStart(2,'0') + ':' + s.toFixed(3).padStart(6,'0');
   567вҶ’}
