    65→                    TranscriptionService.ACTION_PROGRESS -> {
    66→                        val message = intent.getStringExtra(TranscriptionService.EXTRA_MESSAGE) ?: ""

    68→                        // 从消息中提取当前转录的视频名称
    69→                        val currentVideo = message.substringBefore(")").trim()
    70→                        viewModel.updateTranscriptionQueue(currentVideo, uiState.transcriptionQueue, progress, message)
    71→                    }

    73→                        val queueArray = intent.getStringArrayExtra(TranscriptionService.EXTRA_QUEUE_LIST)
    74→                        val queueList = queueArray?.toList() ?: emptyList()
    75→                        viewModel.updateTranscriptionQueue(uiState.transcribingVideo, queueList, uiState.transcriptionProgress, uiState.transcriptionMessage)
    76→                    }
    77→                    TranscriptionService.ACTION_COMPLETED -> {
    78→                        val success = intent.getBooleanExtra(TranscriptionService.EXTRA_SUCCESS, false)
    79→                        val videoPath = intent.getStringExtra(TranscriptionService.EXTRA_VIDEO_PATH)
    80→                        val vttPath = intent.getStringExtra(TranscriptionService.EXTRA_VTT_PATH)

    82→                            viewModel.transcriptionCompleted(videoPath, vttPath)
    83→                        } else {
    84→                            viewModel.transcriptionCompleted()
    85→                        }
    86→                    }
