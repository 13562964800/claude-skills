    49→        /**
    50→         * API 健康状态数据类
    51→         */
    52→        data class ApiHealthStatus(
    53→            val isAvailable: Boolean,
    54→            val responseTime: Long,
    55→            val lastCheckTime: Long,
    56→            val errorMessage: String? = null
    57→        )
    58→
    59→        /**
    60→         * 检查 API 健康状态
    61→         * @param apiUrl API 端点 URL
    62→         * @return 如果缓存有效且 API 可用返回 true，否则返回 true（默认可用，需实际检测）
    63→         */
    64→        fun isApiHealthy(apiUrl: String): Boolean {
    65→            val cached = apiHealthCache[apiUrl]
    66→            if (cached != null && System.currentTimeMillis() - cached.lastCheckTime < HEALTH_CACHE_DURATION) {
    67→                return cached.isAvailable
    68→            }
    69→            return true // 没有缓存时默认可用
    70→        }
    71→
    72→        /**
    73→         * 更新 API 健康状态
    74→         * @param apiUrl API 端点 URL
    75→         * @param isAvailable 是否可用
    76→         * @param responseTime 响应时间（毫秒）
    77→         * @param error 错误信息（可选）
    78→         */
    79→        fun updateApiHealth(apiUrl: String, isAvailable: Boolean, responseTime: Long = 0, error: String? = null) {
    80→            apiHealthCache[apiUrl] = ApiHealthStatus(isAvailable, responseTime, System.currentTimeMillis(), error)
    81→            Log.d(TAG, "API health updated: $apiUrl -> available=$isAvailable, responseTime=${responseTime}ms")
    82→        }
    83→
    84→        /**
    85→         * 获取最佳 API（按响应时间排序，优先使用最快的可用 API）
    86→         * @param apis API 配置列表
    87→         * @param getUrl 从配置中获取 URL 的函数
    88→         * @return 排序后的 API 列表
    89→         */
    90→        fun <T> getBestApi(apis: List<T>, getUrl: (T) -> String): List<T> {
    91→            return apis.sortedBy { api ->
    92→                val url = getUrl(api)
    93→                val status = apiHealthCache[url]
    94→                if (status == null) 0L else if (status.isAvailable) status.responseTime else Long.MAX_VALUE
    95→            }
    96→        }

   174→    /**
   175→     * 平台检测 - 参考 Electron 版本的 detectPlatform 函数
   176→     * 支持更多平台和短链接
   177→     */
   178→    fun detectPlatform(url: String): VideoPlatform {
   179→        val lowerUrl = url.lowercase()
   180→        return when {
   181→            // YouTube - 支持主域名和短链接
   182→            lowerUrl.contains("youtube.com") || lowerUrl.contains("youtu.be") -> VideoPlatform.YOUTUBE
   183→            // Bilibili - 支持主域名和 b23.tv 短链接
   184→            lowerUrl.contains("bilibili.com") || lowerUrl.contains("b23.tv") -> VideoPlatform.BILIBILI
   185→            // 抖音 - 支持多种域名
   186→            lowerUrl.contains("douyin.com") || lowerUrl.contains("iesdouyin.com") -> VideoPlatform.DOUYIN
   187→            // TikTok
   188→            lowerUrl.contains("tiktok.com") -> VideoPlatform.TIKTOK
   189→            // 微博
   190→            lowerUrl.contains("weibo.com") -> VideoPlatform.WEIBO
   191→            // 小红书
   192→            lowerUrl.contains("xiaohongshu.com") || lowerUrl.contains("xhslink.com") -> VideoPlatform.XIAOHONGSHU
   193→            // 微信视频号 - 只匹配视频号专用域名
   194→            lowerUrl.contains("channels.weixin.qq.com") -> VideoPlatform.WECHAT
   195→            // 直接视频链接
   196→            lowerUrl.contains(".mp4") || lowerUrl.contains(".m3u8") || lowerUrl.contains(".mpd") -> VideoPlatform.DIRECT
   197→            // 未知平台
   198→            else -> VideoPlatform.UNKNOWN
   199→        }
   200→    }

   202→    /**
   203→     * 获取视频下载链接 - 主入口函数
   204→     * 参考 Electron 版本的 extract-info IPC 处理
   205→     */
   206→    suspend fun getVideoInfo(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
   207→        try {
   208→            withTimeout(DEFAULT_TIMEOUT) {
   209→                val cleanUrl = cleanUrl(url)
   210→                Log.d(TAG, "Processing URL: $cleanUrl")
   211→
   212→                val platform = detectPlatform(cleanUrl)
   213→                Log.d(TAG, "Detected platform: $platform")
   214→
   215→                when (platform) {
   216→                    VideoPlatform.YOUTUBE -> fetchYoutubeInfo(cleanUrl)
   217→                    VideoPlatform.BILIBILI -> fetchBilibiliInfo(cleanUrl)
   218→                    VideoPlatform.DOUYIN -> fetchDouyinInfo(cleanUrl)
   219→                    VideoPlatform.TIKTOK -> fetchTiktokInfo(cleanUrl)
   220→                    VideoPlatform.XIAOHONGSHU -> fetchXiaohongshuInfo(cleanUrl)
   221→                    VideoPlatform.WECHAT -> fetchWechatChannelsInfo(cleanUrl)
   222→                    VideoPlatform.DIRECT -> Result.success(
   223→                        VideoDownloadInfo(
   224→                            title = "直接视频链接",
   225→                            downloadUrl = cleanUrl,
   226→                            cover = null,
   227→                            duration = 0L,
   228→                            author = null,
   229→                            platform = VideoPlatform.DIRECT
   230→                        )
   231→                    )
   232→                    else -> Result.failure(
   233→                        Exception(
   234→                            "❌ 不支持的平台\n\n当前链接: $cleanUrl\n\n支持的平台:\n• YouTube (youtube.com, youtu.be)\n• B站 (bilibili.com, b23.tv)\n• 抖音 (douyin.com)\n• TikTok (tiktok.com)\n• 小红书 (xiaohongshu.com, xhslink.com)\n• 微信视频号 (channels.weixin.qq.com)\n\n建议: 请使用「导入本地视频」功能"
   235→                        )
   236→                    )
   237→                }
   238→            }
   239→        } catch (e: Throwable) {
   240→            Log.e(TAG, "Error fetching video info", e)
   241→            val message = if (e is kotlinx.coroutines.TimeoutCancellationException) {
   242→                "解析超时，请检查网络连接后重试"
   243→            } else {
   244→                "获取视频信息失败: ${e.message}"
   245→            }
   246→            Result.failure(Exception(message, e))
   247→        }
   248→    }

   252→     * 参考 Electron 版本的 cleanUrl 函数

   254→    private fun cleanUrl(url: String): String {
   255→        val trimmed = url.trim()
   256→        // 提取第一个有效的 HTTP/HTTPS 链接
   257→        val match = Regex("(https?://[^\\s\"'<>]+)").find(trimmed)
   258→        return match?.value ?: trimmed
   259→    }

   261→    /**
   262→     * 获取 YouTube 视频信息 - 改进版，使用 InnerTube API
   263→     * 基于 GitHub 研究：
   264→     * - ViMusic: https://github.com/vfsfitvnm/ViMusic (InnerTube 实现)
   265→     * - NewPipeExtractor: https://github.com/TeamNewPipe/NewPipeExtractor
   266→     */
   267→    private suspend fun fetchYoutubeInfo(url: String): Result<VideoDownloadInfo> {
   268→        Log.d(TAG, "Fetching YouTube info for: $url")
   269→
   270→        // 策略1: 尝试 NewPipeExtractor（v0.26.0 已修复低版本 Android 兼容性）
   271→        val newPipeResult = fetchYoutubeNewPipe(url)
   272→        if (newPipeResult.isSuccess) {
   273→            Log.d(TAG, "NewPipeExtractor success")
   274→            return newPipeResult
   275→        }
   276→        Log.w(TAG, "NewPipeExtractor failed: ${newPipeResult.exceptionOrNull()?.message}")
   277→
   278→        // 策略2: 尝试 InnerTube API（备用）
   279→        val innertubeResult = fetchYouTubeInnerTubeAPI(url)
   280→        if (innertubeResult.isSuccess) {
   281→            Log.d(TAG, "InnerTube API success")
   282→            return innertubeResult
   283→        }
   284→        Log.w(TAG, "InnerTube API failed: ${innertubeResult.exceptionOrNull()?.message}")
   285→
   286→        // 策略3: 尝试第三方API（最后备用）
   287→        return fetchYouTubeThirdPartyAPI(url)
   288→    }

   290→    /**
   291→     * 使用 NewPipeExtractor 解析 YouTube 视频
   292→     * 这是专门为 Android 设计的库，不依赖 YouTube API
   293→     */
   294→    private suspend fun fetchYoutubeNewPipe(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
   295→        try {
   296→            val videoId = extractYoutubeVideoId(url)
   297→                ?: return@withContext Result.failure(Exception("无法提取 YouTube 视频 ID"))
   298→
   299→            Log.d(TAG, "Using NewPipeExtractor for YouTube: $videoId")
   300→
   301→            // 使用标准 YouTube URL 格式
   302→            val standardUrl = "https://www.youtube.com/watch?v=$videoId"
   303→            val streamInfo = org.schabi.newpipe.extractor.stream.StreamInfo.getInfo(standardUrl)
   304→
   305→            // 优先使用混合流（视频+音频合一），文件更小下载更快
   306→            // 限制最高 720p，平衡画质和下载速度
   307→            val videoStream = streamInfo.videoStreams
   308→                .filter { !it.isVideoOnly }
   309→                .filter { (it.getResolution()?.replace("p", "")?.toIntOrNull() ?: 0) <= 720 }
   310→                .maxByOrNull { it.getResolution()?.replace("p", "")?.toIntOrNull() ?: 0 }
   311→
   312→            // 如果没有 720p 以下的混合流，放宽到所有混合流
   313→            val fallbackStream = videoStream ?: streamInfo.videoStreams
   314→                .filter { !it.isVideoOnly }
   315→                .maxByOrNull { it.getResolution()?.replace("p", "")?.toIntOrNull() ?: 0 }
   316→
   317→            // 只有在完全没有混合流时才用 DASH
   318→            val videoOnlyStream = if (fallbackStream == null) {
   319→                streamInfo.videoOnlyStreams
   320→                    .filter { (it.getResolution()?.replace("p", "")?.toIntOrNull() ?: 0) <= 720 }
   321→                    .maxByOrNull { it.getResolution()?.replace("p", "")?.toIntOrNull() ?: 0 }
   322→                    ?: streamInfo.videoOnlyStreams
   323→                        .maxByOrNull { it.getResolution()?.replace("p", "")?.toIntOrNull() ?: 0 }
   324→            } else null
   325→
   326→            val audioStream = if (videoOnlyStream != null) {
   327→                streamInfo.audioStreams.maxByOrNull { it.averageBitrate }
   328→            } else null
   329→
   330→            val bestVideo = fallbackStream ?: videoOnlyStream
   331→            val bestAudio = if (fallbackStream != null) null else audioStream
   332→
   333→            Log.d(TAG, "NewPipe streams - video: ${streamInfo.videoStreams.size}, videoOnly: ${streamInfo.videoOnlyStreams.size}, audio: ${streamInfo.audioStreams.size}")
   334→            Log.d(TAG, "NewPipe selected: ${bestVideo?.getResolution()}, mixed=${fallbackStream != null}, URL: ${bestVideo?.content?.take(100)}")
   335→
   336→            if (bestVideo != null) {
   337→                Result.success(VideoDownloadInfo(
   338→                    title = streamInfo.name ?: "YouTube 视频",
   339→                    downloadUrl = bestVideo.content ?: "",
   340→                    cover = streamInfo.thumbnails?.firstOrNull()?.url,
   341→                    duration = streamInfo.duration,
   342→                    author = streamInfo.uploaderName,
   343→                    platform = VideoPlatform.YOUTUBE,
   344→                    audioUrl = bestAudio?.content
   345→                ))
   346→            } else {
   347→                Result.failure(Exception("无法获取视频流"))
   348→            }
   349→        } catch (e: Throwable) {
   350→            // 捕获 Throwable 而非 Exception，因为 NewPipe 可能抛出 NoSuchMethodError 等
   351→            Log.e(TAG, "NewPipeExtractor error", e)
   352→            Result.failure(Exception("NewPipe解析失败: ${e.message}", e))
   353→        }
   354→    }

   356→    /**
   357→     * 使用 YouTube InnerTube API 解析
   358→     * 使用 ANDROID 客户端身份请求 googleapis 端点
   359→     */
   360→    private suspend fun fetchYouTubeInnerTubeAPI(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
   361→        try {
   362→            val videoId = extractYoutubeVideoId(url)
   363→            if (videoId == null) {
   364→                return@withContext Result.failure(Exception("无法提取视频ID"))
   365→            }
   366→            Log.d(TAG, "YouTube video ID: $videoId")
   367→
   368→            // InnerTube API 请求 - 使用 TV 嵌入式播放器客户端（不需要 PoToken）
   369→            val requestBody = JSONObject().apply {
   370→                put("context", JSONObject().apply {
   371→                    put("client", JSONObject().apply {
   372→                        put("clientName", "TVHTML5_SIMPLY_EMBEDDED_PLAYER")
   373→                        put("clientVersion", "2.0")
   374→                        put("hl", "en")
   375→                        put("gl", "US")
   376→                    })
   377→                    put("thirdParty", JSONObject().apply {
   378→                        put("embedUrl", "https://www.youtube.com/")
   379→                    })
   380→                })
   381→                put("videoId", videoId)
   382→                put("contentCheckOk", true)
   383→                put("racyCheckOk", true)
   384→            }.toString()
   385→
   386→            val request = Request.Builder()
   387→                .url("https://youtubei.googleapis.com/youtubei/v1/player?key=AIzaSyA8eiZmM1FaDVjRy-df2KTyQ_vz_yYM39w")
   388→                .header("Content-Type", "application/json")
   389→                .header("User-Agent", "Mozilla/5.0")
   390→                .header("X-YouTube-Client-Name", "85")
   391→                .header("X-YouTube-Client-Version", "2.0")
   392→                .post(requestBody.toRequestBody("application/json".toMediaType()))
   393→                .build()
   394→
   395→            val response = client.newCall(request).execute()
   396→            val body = response.body?.string() ?: return@withContext Result.failure(IOException("空响应"))
   397→            val json = JSONObject(body)
   398→
   399→            // 检查播放状态
   400→            val playabilityStatus = json.optJSONObject("playabilityStatus")
   401→            val status = playabilityStatus?.optString("status")
   402→            if (status != "OK") {
   403→                val reason = playabilityStatus?.optString("reason") ?: "视频不可用"
   404→                return@withContext Result.failure(Exception("YouTube: $reason"))
   405→            }
   406→
   407→            val videoDetails = json.optJSONObject("videoDetails")
   408→            val title = videoDetails?.optString("title") ?: "YouTube视频"
   409→            val author = videoDetails?.optString("author") ?: ""
   410→            val lengthSeconds = videoDetails?.optLong("lengthSeconds") ?: 0L
   411→
   412→            val streamingData = json.optJSONObject("streamingData")
   413→
   414→            // 优先使用 adaptiveFormats（分离的音视频流，画质更好）
   415→            val adaptiveFormats = streamingData?.optJSONArray("adaptiveFormats")
   416→            var bestVideoUrl: String? = null
   417→            var bestAudioUrl: String? = null
   418→            var bestVideoBitrate = 0L
   419→            var bestAudioBitrate = 0L
   420→
   421→            if (adaptiveFormats != null) {
   422→                for (i in 0 until adaptiveFormats.length()) {
   423→                    val fmt = adaptiveFormats.getJSONObject(i)
   424→                    val mimeType = fmt.optString("mimeType", "")
   425→                    val fmtUrl = fmt.optString("url")
   426→                    val bitrate = fmt.optLong("bitrate", 0)
   427→
   428→                    if (fmtUrl.isEmpty()) continue // 跳过加密的流
   429→
   430→                    if (mimeType.startsWith("video/") && bitrate > bestVideoBitrate) {
   431→                        bestVideoBitrate = bitrate
   432→                        bestVideoUrl = fmtUrl
   433→                    } else if (mimeType.startsWith("audio/") && bitrate > bestAudioBitrate) {
   434→                        bestAudioBitrate = bitrate
   435→                        bestAudioUrl = fmtUrl
   436→                    }
   437→                }
   438→            }
   439→
   440→            // 如果 adaptiveFormats 没有可用的，回退到 formats（混合流）
   441→            if (bestVideoUrl == null) {
   442→                val formats = streamingData?.optJSONArray("formats")
   443→                if (formats != null && formats.length() > 0) {
   444→                    // 选择最高画质的混合流
   445→                    var bestBitrate = 0L
   446→                    for (i in 0 until formats.length()) {
   447→                        val fmt = formats.getJSONObject(i)
   448→                        val fmtUrl = fmt.optString("url")
   449→                        val bitrate = fmt.optLong("bitrate", 0)
   450→                        if (fmtUrl.isNotEmpty() && bitrate > bestBitrate) {
   451→                            bestBitrate = bitrate
   452→                            bestVideoUrl = fmtUrl
   453→                        }
   454→                    }
   455→                }
   456→            }
   457→
   458→            val thumbnails = videoDetails?.optJSONObject("thumbnail")?.optJSONArray("thumbnails")
   459→            val cover = thumbnails?.let {
   460→                if (it.length() > 0) it.getJSONObject(it.length() - 1).optString("url") else null
   461→            }
   462→
   463→            if (bestVideoUrl != null) {
   464→                return@withContext Result.success(VideoDownloadInfo(
   465→                    title = title,
   466→                    downloadUrl = bestVideoUrl,
   467→                    cover = cover,
   468→                    duration = lengthSeconds,
   469→                    author = author,
   470→                    platform = VideoPlatform.YOUTUBE,
   471→                    audioUrl = bestAudioUrl
   472→                ))
   473→            }
   474→
   475→            Result.failure(Exception("无法获取视频流（可能需要签名解密）"))
   476→
   477→        } catch (e: Exception) {
   478→            Log.e(TAG, "InnerTube API error", e)
   479→            Result.failure(e)
   480→        }
   481→    }

   732→    /**
   733→     * 使用 B站官方 API 解析
   734→     * 参考 bilibili-api (Kotlin): https://github.com/czp3009/bilibili-api
   735→     *
   736→     * API 端点：
   737→     * - 视频信息: https://api.bilibili.com/x/web-interface/view?bvid={bvid}
   738→     * - 播放地址: https://api.bilibili.com/x/player/playurl?bvid={bvid}&cid={cid}
   739→     */
   740→    private suspend fun fetchBilibiliOfficialAPI(url: String): Result<VideoDownloadInfo> = withContext(Dispatchers.IO) {
   741→        try {
   742→            // 1. 如果是 b23.tv 短链接，先解析真实 URL
   743→            var realUrl = url
   744→            if (url.contains("b23.tv")) {
   745→                val resolved = resolveB23ShortLink(url)
   746→                if (resolved != null) {
   747→                    realUrl = resolved
   748→                    Log.d(TAG, "Resolved b23.tv -> $realUrl")
   749→                } else {
   750→                    return@withContext Result.failure(Exception("无法解析b23.tv短链接"))
   751→                }
   752→            }
   753→
   754→            // 2. 提取视频标识（BV号或av号）
   755→            val videoId = extractBVID(realUrl)
   756→            if (videoId == null) {
   757→                return@withContext Result.failure(Exception("无法提取BV号或av号"))
   758→            }
   759→            Log.d(TAG, "Video ID: $videoId")
   760→
   761→            // 获取Cookie（如果有配置）
   762→            val cookie = getCookieForPlatform(url)
   763→            if (cookie != null) {
   764→                Log.d(TAG, "Using configured cookies for Bilibili API")
   765→            }
   766→
   767→            // 3. 获取视频信息（获取 cid）- 支持 BV 号和 av 号
   768→            val infoUrl = if (videoId.startsWith("av")) {
   769→                "https://api.bilibili.com/x/web-interface/view?aid=${videoId.removePrefix("av")}"
   770→            } else {
   771→                "https://api.bilibili.com/x/web-interface/view?bvid=$videoId"
   772→            }
   773→            val infoRequestBuilder = Request.Builder()
   774→                .url(infoUrl)
   775→                .header("User-Agent", getUserAgent())
   776→                .header("Referer", "https://www.bilibili.com/")
   777→
   778→            // 添加Cookie
   779→            if (cookie != null) {
   780→                infoRequestBuilder.header("Cookie", cookie)
   781→            }
   782→
   783→            val infoRequest = infoRequestBuilder.build()
   784→
   785→            val infoResponse = client.newCall(infoRequest).execute()
   786→            val infoBody = infoResponse.body?.string() ?: return@withContext Result.failure(IOException("空响应"))
   787→            val infoJson = JSONObject(infoBody)
   788→
   789→            if (infoJson.optInt("code", -1) != 0) {
   790→                val errorMsg = infoJson.optString("message")
   791→                // 检查是否是需要Cookie的错误
   792→                if (errorMsg.contains("登录") || errorMsg.contains("会员") || errorMsg.contains("权限")) {
   793→                    return@withContext Result.failure(
   794→                        Exception("❌ 需要登录或会员权限\n\n错误信息: $errorMsg\n\n解决方案：\n• 在设置中配置B站Cookie（SESSDATA）\n• 或使用「导入本地视频」功能")
   795→                    )
   796→                }
   797→                return@withContext Result.failure(Exception("API错误: $errorMsg"))
   798→            }
   799→
   800→            val data = infoJson.getJSONObject("data")
   801→            val cid = data.optLong("cid")
   802→            val title = data.optString("title")
   803→            val cover = data.optString("pic").replace("\\/\\\\/", "/")
   804→            val duration = data.optInt("duration")
   805→            val author = data.optJSONObject("owner")?.optString("name") ?: "未知UP主"
   806→
   807→            Log.d(TAG, "Video info: $title, cid=$cid")
   808→
   809→            // 4. 获取播放URL - 使用DASH格式（fnval=4048支持全部格式）
   810→            val bvParam = if (videoId.startsWith("av")) "avid=${videoId.removePrefix("av")}" else "bvid=$videoId"
   811→            val playUrl = "https://api.bilibili.com/x/player/playurl?$bvParam&cid=$cid&qn=116&fnval=4048&fnver=0&fourk=1"
   812→            val playRequestBuilder = Request.Builder()
   813→                .url(playUrl)
   814→                .header("User-Agent", getUserAgent())
   815→                .header("Referer", "https://www.bilibili.com/")
   816→
   817→            if (cookie != null) {
   818→                playRequestBuilder.header("Cookie", cookie)
   819→            }
   820→
   821→            val playRequest = playRequestBuilder.build()
   822→            val playResponse = client.newCall(playRequest).execute()
   823→            val playBody = playResponse.body?.string() ?: return@withContext Result.failure(IOException("空响应"))
   824→            val playJson = JSONObject(playBody)
   825→
   826→            if (playJson.optInt("code", -1) != 0) {
   827→                val errorMsg = playJson.optString("message")
   828→                if (errorMsg.contains("登录") || errorMsg.contains("会员") || errorMsg.contains("权限")) {
   829→                    return@withContext Result.failure(
   830→                        Exception("❌ 需要登录或会员权限\n\n错误信息: $errorMsg\n\n解决方案：\n• 在设置中配置B站Cookie（SESSDATA）\n• 或使用「导入本地视频」功能")
   831→                    )
   832→                }
   833→                return@withContext Result.failure(Exception("获取播放URL失败: $errorMsg"))
   834→            }
   835→
   836→            val playData = playJson.getJSONObject("data")
   837→            val dash = playData.optJSONObject("dash")
   838→            val durl = playData.optJSONArray("durl")
   839→
   840→            var videoUrl: String? = null
   841→            var audioUrl: String? = null
   842→
   843→            if (durl != null && durl.length() > 0) {
   844→                // MP4格式（包含音视频）
   845→                videoUrl = durl.getJSONObject(0).optString("url")
   846→            } else if (dash != null) {
   847→                // DASH格式 - 选择最佳画质的视频流
   848→                val dashVideo = dash.optJSONArray("video")
   849→                val dashAudio = dash.optJSONArray("audio")
   850→
   851→                if (dashVideo != null && dashVideo.length() > 0) {
   852→                    // 按bandwidth降序选择最佳画质（优先选择高码率）
   853→                    var bestBandwidth = 0L
   854→                    for (i in 0 until dashVideo.length()) {
   855→                        val v = dashVideo.getJSONObject(i)
   856→                        val bw = v.optLong("bandwidth", 0)
   857→                        if (bw > bestBandwidth) {
   858→                            bestBandwidth = bw
   859→                            videoUrl = v.optString("baseUrl").takeIf { it.isNotEmpty() }
   860→                                ?: v.optString("base_url")
   861→                        }
   862→                    }
   863→                    // 选择最佳音频质量
   864→                    if (dashAudio != null && dashAudio.length() > 0) {
   865→                        var bestAudioBw = 0L
   866→                        for (i in 0 until dashAudio.length()) {
   867→                            val a = dashAudio.getJSONObject(i)
   868→                            val bw = a.optLong("bandwidth", 0)
   869→                            if (bw > bestAudioBw) {
   870→                                bestAudioBw = bw
   871→                                audioUrl = a.optString("baseUrl").takeIf { it.isNotEmpty() }
   872→                                    ?: a.optString("base_url")
   873→                            }
   874→                        }
   875→                    }
   876→                    Log.d(TAG, "DASH format: best video bandwidth=$bestBandwidth, hasAudio=${audioUrl != null}")
   877→                }
   878→            }
   879→
   880→            if (videoUrl != null) {
   881→                Log.d(TAG, "Got video URL from official API, cookies used: ${cookie != null}, hasAudio: ${audioUrl != null}")
   882→                return@withContext Result.success(VideoDownloadInfo(
   883→                    title = title,
   884→                    downloadUrl = videoUrl,
   885→                    cover = cover,
   886→                    duration = duration.toLong(),
   887→                    author = author,
   888→                    cookies = cookie,
   889→                    platform = VideoPlatform.BILIBILI,
   890→                    audioUrl = audioUrl
   891→                ))
   892→            }
   893→
   894→            Result.failure(Exception("无法获取播放地址"))
   895→
   896→        } catch (e: Exception) {
   897→            Log.e(TAG, "Official API error", e)
   898→            Result.failure(e)
   899→        }
   900→    }
