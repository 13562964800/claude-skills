     1→package com.multilanglearner.data.network
     2→
     3→import android.util.Log
     4→import com.multilanglearner.domain.model.AIConfig
     5→import com.multilanglearner.domain.model.WordTimestamp
     6→import kotlinx.serialization.Serializable
     7→import kotlinx.serialization.json.*
     8→import okhttp3.*
     9→import okhttp3.MediaType.Companion.toMediaType
    10→import okhttp3.RequestBody.Companion.asRequestBody
    11→import okhttp3.RequestBody.Companion.toRequestBody
    12→import java.io.File
    13→import java.util.concurrent.TimeUnit
    14→import javax.inject.Inject
    15→import javax.inject.Singleton
    16→
    17→@Serializable
    18→data class WhisperResult(
    19→    val success: Boolean,
    20→    val text: String = "",
    21→    val words: List<WordTimestamp> = emptyList(),
    22→    val detectedLang: String = "",
    23→    val error: String = ""
    24→)
    25→
    26→@Singleton
    27→class WhisperService @Inject constructor() {
    28→    private val client = OkHttpClient.Builder()
    29→        .connectTimeout(30, TimeUnit.SECONDS)
    30→        .readTimeout(300, TimeUnit.SECONDS)
    31→        .writeTimeout(60, TimeUnit.SECONDS)
    32→        .build()
    33→
    34→    private val json = Json { ignoreUnknownKeys = true }
    35→
    36→    // Groq API 文件大小限制（增加到 100MB，Groq 实际支持更大）
    37→    private val MAX_FILE_SIZE = 100 * 1024 * 1024L
    38→    private val MIN_FILE_SIZE = 1024L // 最小 1KB，避免空文件或太小的文件
    39→
    40→    // 最大重试次数
    41→    private val MAX_RETRIES = 3
    42→
    43→    /**
    44→     * Transcribe audio via Groq Whisper API
    45→     * 优先使用 groqApiKey，其次使用 provider 为 groq 时的 apiKey
    46→     * 添加自动重试机制
    47→     */
    48→    suspend fun transcribeGroq(audioFile: File, config: AIConfig, lang: String? = null): WhisperResult? {
    49→        // 自动重试逻辑
    50→        var lastError: Exception? = null
    51→        repeat(MAX_RETRIES) { attempt ->
    52→            if (attempt > 0) {
    53→                Log.d("WhisperService", "Retry $attempt/$MAX_RETRIES")
    54→                kotlinx.coroutines.delay(2000L * attempt) // 递增延迟
    55→            }
    56→
    57→            val result = transcribeGroqInternal(audioFile, config, lang)
    58→            if (result?.success == true) {
    59→                return result
    60→            }
    61→            lastError = Exception(result?.error ?: "Unknown error")
    62→        }
    63→
    64→        // 所有重试都失败
    65→        return WhisperResult(
    66→            success = false,
    67→            error = "❌ 转录失败（已重试 $MAX_RETRIES 次）\n\n${lastError?.message ?: "未知错误"}\n\n建议：检查网络连接和 API Key 配置"
    68→        )
    69→    }
    70→
    71→    /**
    72→     * 内部转录实现
    73→     */
    74→    private suspend fun transcribeGroqInternal(audioFile: File, config: AIConfig, lang: String? = null): WhisperResult? {
    75→        // 获取 Groq API Key：优先使用专门的 groqApiKey，其次检查 provider 是否为 groq
    76→        val apiKey = config.groqApiKey.takeIf { it.isNotBlank() }
    77→            ?: (config.apiKey.takeIf { config.provider == "groq" && it.isNotBlank() })
    78→            ?: return WhisperResult(success = false, error = "❌ 未配置 Groq API Key\n\n请前往「AI 设置」配置 Groq API Key。\n获取 API Key：https://console.groq.com/keys")
    79→
    80→        val whisperModel = config.whisperModel.ifBlank { "whisper-large-v3" }
    81→
    82→        // 检查文件大小
    83→        val fileSize = audioFile.length()
    84→        Log.d("WhisperService", "File size: $fileSize bytes, max: $MAX_FILE_SIZE")
    85→
    86→        // 检查最小文件大小
    87→        if (fileSize < MIN_FILE_SIZE) {
    88→            Log.e("WhisperService", "File too small: $fileSize bytes (min: $MIN_FILE_SIZE)")
    89→            return WhisperResult(
    90→                success = false,
    91→                error = "❌ 视频文件过小\n\n文件大小: ${fileSize / 1024} KB\n最小要求: 1 KB\n\n建议：请使用包含音频内容的有效视频文件"
    92→            )
    93→        }
    94→
    95→        if (fileSize > MAX_FILE_SIZE) {
    96→            val sizeMB = String.format("%.2f", fileSize / 1024.0 / 1024.0)
    97→            val maxMB = String.format("%.2f", MAX_FILE_SIZE / 1024.0 / 1024.0)
    98→            Log.e("WhisperService", "File too large: $sizeMB MB (max: $maxMB MB)")
    99→            return WhisperResult(
   100→                success = false,
   101→                error = "❌ 视频文件过大\n\n文件大小: $sizeMB MB\nGroq 限制: $maxMB MB\n\n建议解决方案：\n1. 使用较短的视频片段\n2. 在电脑上用 ffmpeg 压缩视频\n3. 在电脑上使用 yt-dlp 只下载音频"
   102→            )
   103→        }
   104→
   105→        Log.d("WhisperService", "Starting transcription with Groq Whisper: model=$whisperModel, file=${audioFile.name}, size=$fileSize")
   106→
   107→        val bodyBuilder = MultipartBody.Builder().setType(MultipartBody.FORM)
   108→            .addFormDataPart("file", audioFile.name, audioFile.asRequestBody("application/octet-stream".toMediaType()))
   109→            .addFormDataPart("model", whisperModel)
   110→            .addFormDataPart("response_format", "verbose_json")
   111→            .addFormDataPart("timestamp_granularities[]", "word")
   112→            .addFormDataPart("timestamp_granularities[]", "segment")
   113→            .addFormDataPart("temperature", "0")
   114→            .addFormDataPart("prompt", "Transcribe accurately with proper punctuation and capitalization.")
   115→
   116→        lang?.let { bodyBuilder.addFormDataPart("language", it) }
   117→
   118→        val request = Request.Builder()
   119→            .url("https://api.groq.com/openai/v1/audio/transcriptions")
   120→            .addHeader("Authorization", "Bearer $apiKey")
   121→            .post(bodyBuilder.build())
   122→            .build()
   123→
   124→        return try {
   125→            Log.d("WhisperService", "Sending request to Groq API...")
   126→            val response = client.newCall(request).execute()
   127→
   128→            Log.d("WhisperService", "Response code: ${response.code}")
   129→
   130→            if (response.code != 200) {
   131→                val errorBody = response.body?.string() ?: "Unknown error"
   132→                Log.e("WhisperService", "API error: $errorBody")
   133→
   134→                // 解析错误信息
   135→                val errorMsg = try {
   136→                    val errorJson = json.parseToJsonElement(errorBody).jsonObject
   137→                    val errObj = errorJson["error"]?.jsonObject
   138→                    val errMsg = errObj?.get("message")?.jsonPrimitive?.contentOrNull
   139→                    val errType = errObj?.get("type")?.jsonPrimitive?.contentOrNull
   140→                    val errCode = errObj?.get("code")?.jsonPrimitive?.contentOrNull
   141→
   142→                    when {
   143→                        errMsg?.contains("invalid_api_key", ignoreCase = true) == true ||
   144→                        errMsg?.contains("Unauthorized", ignoreCase = true) == true ||
   145→                        response.code == 401 ->
   146→                            "❌ API Key 无效\n\n请检查 Groq API Key 是否正确配置\n\n获取 API Key：https://console.groq.com/keys"
   147→                        errMsg?.contains("quota", ignoreCase = true) == true ||
   148→                        errMsg?.contains("limit", ignoreCase = true) == true ||
   149→                        errMsg?.contains("rate_limit", ignoreCase = true) == true ||
   150→                        response.code == 429 ->
   151→                            "❌ API 配额已用完\n\nGroq API 使用次数已达限制\n\n建议：\n1. 等待配额重置\n2. 更换 API Key\n3. 升级 Groq 账户"
   152→                        errMsg?.contains("forbidden", ignoreCase = true) == true ||
   153→                        errMsg?.contains("banned", ignoreCase = true) == true ||
   154→                        response.code == 403 ->
   155→                            "❌ 转录被禁止\n\n${errMsg ?: "访问被限制"}\n\n可能原因：\n1. API Key 被禁用\n2. 账户存在违规行为\n3. 内容不符合使用条款"
   156→                        else -> "❌ 转录失败 ($response.code)\n\n${errMsg ?: errorBody}"
   157→                    }
   158→                } catch (e: Exception) {
   159→                    "❌ 转录失败\n\n${errorBody.take(200)}"
   160→                }
   161→
   162→                return WhisperResult(success = false, error = errorMsg)
   163→            }
   164→
   165→            val body = response.body?.string() ?: return WhisperResult(success = false, error = "空响应")
   166→            Log.d("WhisperService", "Response length: ${body.length}")
   167→            Log.d("WhisperService", "Response body preview: ${body.take(500)}")
   168→
   169→            // 检查响应是否为有效的 JSON
   170→            if (!body.trim().startsWith("{") && !body.trim().startsWith("[")) {
   171→                Log.e("WhisperService", "Response is not valid JSON: ${body.take(200)}")
   172→                return WhisperResult(
   173→                    success = false,
   174→                    error = "❌ API 返回格式错误\n\nGroq API 返回了非 JSON 格式的响应。\n\n响应内容：\n${body.take(500)}"
   175→                )
   176→            }
   177→
   178→            try {
   179→                val jsonObj = json.parseToJsonElement(body).jsonObject
   180→                val text = jsonObj["text"]?.jsonPrimitive?.contentOrNull ?: ""
   181→                val detectedLang = jsonObj["language"]?.jsonPrimitive?.contentOrNull ?: ""
   182→
   183→                Log.d("WhisperService", "Transcribed text length: ${text.length}, detected language: $detectedLang")
   184→
   185→                val words = jsonObj["words"]?.jsonArray?.map { w ->
   186→                    val obj = w.jsonObject
   187→                    WordTimestamp(
   188→                        w = obj["word"]?.jsonPrimitive?.contentOrNull ?: "",
   189→                        s = (obj["start"]?.jsonPrimitive?.doubleOrNull ?: 0.0),
   190→                        e = (obj["end"]?.jsonPrimitive?.doubleOrNull ?: 0.0)
   191→                    )
   192→                } ?: emptyList()
   193→
   194→                Log.d("WhisperService", "Word timestamps count: ${words.size}")
   195→
   196→                WhisperResult(
   197→                    success = true,
   198→                    text = text,
   199→                    words = words,
   200→                    detectedLang = detectedLang
   201→                )
   202→            } catch (e: Exception) {
   203→                Log.e("WhisperService", "JSON parsing error", e)
   204→                Log.e("WhisperService", "Response body: $body")
   205→                WhisperResult(
   206→                    success = false,
   207→                    error = "❌ 响应解析失败\n\nGroq API 返回了无法解析的响应。\n\n错误: ${e.message}\n\n响应内容：\n${body.take(500)}"
   208→                )
   209→            }
   210→        } catch (e: Exception) {
   211→            Log.e("WhisperService", "Transcription error", e)
   212→            val errorMsg = when {
   213→                e.message?.contains("timeout", ignoreCase = true) == true ->
   214→                    "❌ 请求超时\n\n网络连接超时，请检查网络连接后重试。"
   215→                e.message?.contains("Unable to resolve host", ignoreCase = true) == true ||
   216→                e.message?.contains("UnknownHost", ignoreCase = true) == true ->
   217→                    "❌ 网络连接失败\n\n无法连接到 Groq API 服务器，请检查网络连接。"
   218→                else -> "❌ 转录失败\n\n${e.message ?: "未知错误"}\n\n请检查网络连接和 API Key 配置后重试。"
   219→            }
   220→            WhisperResult(success = false, error = errorMsg)
   221→        }
   222→    }
   223→}
