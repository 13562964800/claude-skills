   391→/**
   392→ * 下载进度卡片组件
   393→ */
   394→@Composable
   395→private fun DownloadProgressCard(
   396→    task: DownloadTask,
   397→    onCancel: () -> Unit,
   398→    formatFileSize: (Long) -> String,
   399→    formatSpeed: (Long) -> String,
   400→    formatRemainingTime: (Long) -> String,
   401→    getStatusText: (DownloadStatus) -> String
   402→) {
   403→    Card(
   404→        modifier = Modifier.fillMaxWidth(),
   405→        colors = CardDefaults.cardColors(
   406→            containerColor = when (task.status) {
   407→                DownloadStatus.FAILED, DownloadStatus.CANCELLED -> MaterialTheme.colorScheme.errorContainer
   408→                DownloadStatus.WAITING_FOR_AUTH -> MaterialTheme.colorScheme.tertiaryContainer
   409→                else -> MaterialTheme.colorScheme.primaryContainer
   410→            }
   411→        )
   412→    ) {
   413→        Column(modifier = Modifier.padding(16.dp)) {
   414→            // 标题和状态
   415→            Row(
   416→                modifier = Modifier.fillMaxWidth(),
   417→                horizontalArrangement = Arrangement.SpaceBetween,
   418→                verticalAlignment = Alignment.CenterVertically
   419→            ) {
   420→                Column(Modifier.weight(1f)) {
   421→                    Text(
   422→                        task.title,
   423→                        style = MaterialTheme.typography.titleSmall,
   424→                        maxLines = 1,
   425→                        overflow = TextOverflow.Ellipsis
   426→                    )
   427→                    Text(
   428→                        getStatusText(task.status),
   429→                        style = MaterialTheme.typography.bodySmall,
   430→                        color = when (task.status) {
   431→                            DownloadStatus.FAILED, DownloadStatus.CANCELLED -> MaterialTheme.colorScheme.error
   432→                            DownloadStatus.WAITING_FOR_AUTH -> MaterialTheme.colorScheme.onTertiaryContainer
   433→                            else -> MaterialTheme.colorScheme.onPrimaryContainer
   434→                        }
   435→                    )
   436→                }
   437→
   438→                // 取消按钮
   439→                if (task.status == DownloadStatus.DOWNLOADING || task.status == DownloadStatus.FETCHING_INFO) {
   440→                    IconButton(onClick = onCancel) {
   441→                        Icon(Icons.Default.Close, "取消", tint = MaterialTheme.colorScheme.error)
   442→                    }
   443→                }
   444→            }
   445→
   446→            Spacer(Modifier.height(12.dp))
   447→
   448→            // 进度条
   449→            if (task.status == DownloadStatus.DOWNLOADING) {
   450→                Column {
   451→                    Row(
   452→                        modifier = Modifier.fillMaxWidth(),
   453→                        horizontalArrangement = Arrangement.SpaceBetween
   454→                    ) {
   455→                        Text(
   456→                            "${task.progress}%",
   457→                            style = MaterialTheme.typography.bodySmall
   458→                        )
   459→                        if (task.speed > 0) {
   460→                            Text(
   461→                                formatSpeed(task.speed),
   462→                                style = MaterialTheme.typography.bodySmall,
   463→                                color = MaterialTheme.colorScheme.primary
   464→                            )
   465→                        }
   466→                    }
   467→
   468→                    Spacer(Modifier.height(4.dp))
   469→
   470→                    LinearProgressIndicator(
   471→                        progress = { task.progress / 100f },
   472→                        modifier = Modifier.fillMaxWidth(),
   473→                        color = MaterialTheme.colorScheme.primary
   474→                    )
   475→
   476→                    // 文件大小和剩余时间
   477→                    if (task.totalBytes > 0) {
   478→                        Spacer(Modifier.height(4.dp))
   479→                        Row(
   480→                            modifier = Modifier.fillMaxWidth(),
   481→                            horizontalArrangement = Arrangement.SpaceBetween
   482→                        ) {
   483→                            Text(
   484→                                "${formatFileSize(task.bytesDownloaded)} / ${formatFileSize(task.totalBytes)}",
   485→                                style = MaterialTheme.typography.bodySmall,
   486→                                color = MaterialTheme.colorScheme.onPrimaryContainer
   487→                            )
   488→                            if (task.remainingTime > 0 && task.speed > 0) {
   489→                                Text(
   490→                                    "剩余 ${formatRemainingTime(task.remainingTime)}",
   491→                                    style = MaterialTheme.typography.bodySmall,
   492→                                    color = MaterialTheme.colorScheme.onPrimaryContainer
   493→                                )
   494→                            }
   495→                        }
   496→                    }
   497→                }
   498→            } else if (task.status == DownloadStatus.FETCHING_INFO) {
   499→                Row(
   500→                    modifier = Modifier.fillMaxWidth(),
   501→                    verticalAlignment = Alignment.CenterVertically
   502→                ) {
   503→                    CircularProgressIndicator(
   504→                        modifier = Modifier.size(20.dp),
   505→                        strokeWidth = 2.dp
   506→                    )
   507→                    Spacer(Modifier.width(8.dp))
   508→                    Text(
   509→                        "正在解析视频信息...",
   510→                        style = MaterialTheme.typography.bodySmall,
   511→                        color = MaterialTheme.colorScheme.onPrimaryContainer
   512→                    )
   513→                }
   514→            } else if (task.status == DownloadStatus.FAILED) {
   515→                Row(
   516→                    modifier = Modifier.fillMaxWidth(),
   517→                    verticalAlignment = Alignment.CenterVertically
   518→                ) {
   519→                    Icon(
   520→                        Icons.Default.Error,
   521→                        contentDescription = null,
   522→                        tint = MaterialTheme.colorScheme.error,
   523→                        modifier = Modifier.size(20.dp)
   524→                    )
   525→                    Spacer(Modifier.width(8.dp))
   526→                    Text(
   527→                        task.errorMessage ?: "下载失败",
   528→                        style = MaterialTheme.typography.bodySmall,
   529→                        color = MaterialTheme.colorScheme.error
   530→                    )
   531→                }
   532→            } else if (task.status == DownloadStatus.COMPLETED) {
   533→                Row(
   534→                    modifier = Modifier.fillMaxWidth(),
   535→                    verticalAlignment = Alignment.CenterVertically
   536→                ) {
   537→                    Icon(
   538→                        Icons.Default.CheckCircle,
   539→                        contentDescription = null,
   540→                        tint = MaterialTheme.colorScheme.primary,
   541→                        modifier = Modifier.size(20.dp)
   542→                    )
   543→                    Spacer(Modifier.width(8.dp))
   544→                    Text(
   545→                        "下载完成",
   546→                        style = MaterialTheme.typography.bodySmall,
   547→                        color = MaterialTheme.colorScheme.primary
   548→                    )
   549→                }
   550→            } else if (task.status == DownloadStatus.WAITING_FOR_AUTH) {
   551→                Row(
   552→                    modifier = Modifier.fillMaxWidth(),
   553→                    verticalAlignment = Alignment.CenterVertically
   554→                ) {
