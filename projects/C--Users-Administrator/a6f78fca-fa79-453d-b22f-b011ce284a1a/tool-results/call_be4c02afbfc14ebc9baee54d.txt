    83→    fun loadSubtitles(videoPath: String, subtitlePath: String?) {
    84→        viewModelScope.launch(Dispatchers.IO) {
    85→            // Try loading VTT subtitle
    86→            val vttPath = subtitlePath ?: findExistingSubtitle(videoPath)
    87→            if (vttPath != null) {
    88→                val file = File(vttPath)
    89→                if (file.exists()) {
    90→                    val content = file.readText()
    91→                    val subs = VttParser.parse(content)
    92→                    if (subs.isNotEmpty()) {
    93→                        _uiState.update { it.copy(
    94→                            subtitles = subs,
    95→                            statusText = "已加载 ${subs.size} 条字幕"
    96→                        )}
    97→                        // Load word timestamps if available
    98→                        loadWordTimestamps(vttPath)
    99→                        return@launch
   100→                    }
   101→                }
   102→            }
   103→            _uiState.update { it.copy(statusText = "无字幕文件，请在 AI 设置中配置 Groq Whisper 后重新转录") }
   104→        }
   105→    }

   107→    private fun findExistingSubtitle(videoPath: String): String? {
   108→        val base = videoPath.substringBeforeLast(".")
   109→        for (ext in listOf(".vtt", ".srt")) {
   110→            val f = File(base + ext)
   111→            if (f.exists()) return f.absolutePath
   112→        }
   113→        // Check with language suffix
   114→        val dir = File(videoPath).parentFile ?: return null
   115→        val name = File(videoPath).nameWithoutExtension
   116→        dir.listFiles()?.forEach { f ->
   117→            if (f.name.startsWith(name) && f.extension == "vtt") return f.absolutePath
   118→        }
   119→        return null
   120→    }

   122→    private fun loadWordTimestamps(vttPath: String) {
   123→        val wtsPath = vttPath.replace(Regex("\\.vtt$"), ".wordts.json")
   124→        val wtsFile = File(wtsPath)
   125→        if (wtsFile.exists()) {
   126→            try {
   127→                val timings: List<WordTimestamp> = json.decodeFromString(wtsFile.readText())
   128→                _uiState.update { it.copy(wordTimings = timings) }
   129→            } catch (_: Exception) {}
   130→        }
   131→    }

   133→    /** Called every 30ms from the player to sync subtitles */
   134→    fun updatePosition(positionMs: Long) {
   135→        _uiState.update { it.copy(currentPositionMs = positionMs) }
   136→        val subs = _uiState.value.subtitles
   137→        if (subs.isEmpty()) return
   138→
   139→        // Binary search for current cue (ported from renderer.js)
   140→        var lo = 0; var hi = subs.size - 1; var idx = -1
   141→        while (lo <= hi) {
   142→            val mid = (lo + hi) ushr 1
   143→            when {
   144→                positionMs < subs[mid].startMs -> hi = mid - 1
   145→                positionMs > subs[mid].endMs -> lo = mid + 1
   146→                else -> { idx = mid; break }
   147→            }
   148→        }
   149→        // Gap handling: find next cue within 0.5s
   150→        if (idx < 0) {
   151→            for (i in subs.indices) {
   152→                if (subs[i].startMs > positionMs) {
   153→                    idx = if (subs[i].startMs - positionMs < 500) i else if (i > 0) i - 1 else 0
   154→                    break
   155→                }
   156→            }
   157→            if (idx < 0) idx = subs.size - 1
   158→        }
   159→
   160→        if (idx != _uiState.value.currentCueIndex) {
   161→            _uiState.update { it.copy(
   162→                currentCueIndex = idx,
   163→                currentSentence = subs[idx].text,
   164→                showWordCard = false,
   165→                aiAnalysisResult = null
   166→            )}
   167→        }
   168→    }

   170→    /** Retranscribe - 使用前台服务支持后台转录 */
   171→    fun retranscribe(videoPath: String) {
   172→        val context = getApplication<Application>()
   173→        val videoName = File(videoPath).nameWithoutExtension
   174→
   175→        // 保存旧字幕状态
   176→        val oldSubtitles = _uiState.value.subtitles
   177→        val oldWordTimings = _uiState.value.wordTimings
   178→
   179→        _uiState.update { it.copy(
   180→            subtitles = emptyList(),
   181→            wordTimings = emptyList(),
   182→            currentCueIndex = -1,
   183→            isTranscribing = true,
   184→            transcribeSegment = 0,
   185→            transcribeTotalSegments = 0,
   186→            transcribeProgress = 0,
   187→            statusText = "正在准备转录..."
   188→        )}
   189→
   190→        // 启动前台服务进行转录
   191→        val intent = Intent(context, TranscriptionService::class.java).apply {
   192→            action = TranscriptionService.ACTION_START
   193→            putExtra(TranscriptionService.EXTRA_VIDEO_PATH, videoPath)
   194→            putExtra(TranscriptionService.EXTRA_VIDEO_NAME, videoName)
   195→        }
   196→        context.startForegroundService(intent)
   197→
   198→        // 保存旧字幕以备失败时恢复
   199→        _oldSubtitles = oldSubtitles
   200→        _oldWordTimings = oldWordTimings
   201→    }

   260→    /** Look up a word - ported from renderer.js showWordInline() */
   261→    fun lookupWord(word: String) {
   262→        _uiState.update { it.copy(showWordCard = true, wordCardData = WordCardData(word = word, translation = "查询中...")) }
   263→        speak(word)
   264→
   265→        viewModelScope.launch(Dispatchers.IO) {
   266→            val lang = _uiState.value.currentLang
   267→            val context = _uiState.value.currentSentence
   268→
   269→            // Parallel lookups
   270→            val aiConfig = aiConfigStore.config.first()
   271→            val aiTransPromise = launch { aiTranslateWord(word, context, lang, aiConfig) }
   272→
   273→            val cardData = when (lang) {
   274→                "en" -> {
   275→                    val online = dictionaryService.lookupEnglish(word)
   276→                    val translation = dictionaryService.translate(word, "en")
   277→                    val aiTrans = aiTranslateWord(word, context, lang, aiConfig)
   278→                    WordCardData(
   279→                        word = word,
   280→                        phonetic = online?.phonetic ?: "",
   281→                        translation = aiTrans ?: translation ?: online?.translation ?: "未找到释义",
   282→                        definitions = online?.definitions ?: emptyList(),
   283→                        examples = online?.examples ?: emptyList()
   284→                    )
   285→                }
   286→                "ja" -> {
   287→                    val online = dictionaryService.lookupJapanese(word)
   288→                    val translation = dictionaryService.translate(word, "ja")
   289→                    val aiTrans = aiTranslateWord(word, context, lang, aiConfig)
   290→                    WordCardData(
   291→                        word = word,
   292→                        phonetic = online?.phonetic ?: "",
   293→                        translation = aiTrans ?: translation ?: "未找到释义",
   294→                        definitions = online?.definitions ?: emptyList()
   295→                    )
   296→                }
   297→                else -> {
   298→                    val translation = dictionaryService.translate(word, lang)
   299→                    val aiTrans = aiTranslateWord(word, context, lang, aiConfig)
   300→                    WordCardData(
   301→                        word = word,
   302→                        translation = aiTrans ?: translation ?: "未找到释义"
   303→                    )
   304→                }
   305→            }
   306→            _uiState.update { it.copy(wordCardData = cardData) }
   307→        }
   308→    }

   310→    private suspend fun aiTranslateWord(word: String, context: String, lang: String, config: AIConfig): String? {
   311→        val langName = LANGUAGES.find { it.code == lang }?.name ?: "英语"
   312→        val prompt = "将${langName}「${word}」翻译为中文。${if (context.isNotBlank()) "语境:\"$context\"。" else ""}只回复中文释义，多义项用/分隔，不要音译。"
   313→        return aiService.chat(config, prompt)
   314→    }

   367→    /** Get aligned word timings for current cue */
   368→    fun getAlignedTimings(words: List<String>): List<WordTimestamp?> {
   369→        val state = _uiState.value
   370→        val cueIdx = state.currentCueIndex
   371→        if (cueIdx < 0 || cueIdx >= state.subtitles.size || state.wordTimings.isEmpty()) {
   372→            return List(words.size) { null }
   373→        }
   374→        val cue = state.subtitles[cueIdx]
   375→        return WordAligner.alignWords(words, state.wordTimings, cue.startMs / 1000.0, cue.endMs / 1000.0)
   376→    }

   378→    /** Format remaining time for display */
   379→    private fun formatRemainingTime(seconds: Long): String {
   380→        return when {
   381→            seconds < 60 -> "${seconds}秒"
   382→            seconds < 3600 -> "${seconds / 60}分${seconds % 60}秒"
   383→            else -> "${seconds / 3600}小时${(seconds % 3600) / 60}分"
   384→        }
   385→    }
