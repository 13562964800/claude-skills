    51→/**

    54→@Serializable
    55→enum class DownloadStatus {
    56→    PENDING,         // 等待中
    57→    FETCHING_INFO,   // 获取信息中
    58→    DOWNLOADING,     // 下载中
    59→    PAUSED,          // 暂停
    60→    COMPLETED,       // 完成
    61→    FAILED,          // 失败
    62→    CANCELLED,       // 已取消
    63→    WAITING_FOR_AUTH // 等待认证（需要登录）
    64→}

   176→    fun downloadVideo(onComplete: (String, String?) -> Unit) {
   177→        val url = _uiState.value.url.trim()
   178→
   179→        if (url.isBlank()) {
   180→            updateCurrentDownload(DownloadTask(url = url, title = "", errorMessage = "请输入视频链接"))
   181→            return
   182→        }
   183→
   184→        // 检查是否已存在相同视频（通过 URL 匹配）
   185→        val existingVideo = _uiState.value.videos.find { it.sourceUrl == url }
   186→        if (existingVideo != null) {
   187→            // 显示重复下载确认对话框
   188→            _uiState.update { it.copy(
   189→                showDuplicateDownloadConfirm = true,
   190→                duplicateVideoTitle = existingVideo.title
   191→            )}
   192→            return
   193→        }
   194→
   195→        // 如果已达最大并发数，加入队列
   196→        if (activeDownloadTasks.size >= MAX_CONCURRENT_DOWNLOADS) {
   197→            addToQueue(url)
   198→            return
   199→        }
   200→
   201→        startDownload(url, onComplete)
   202→    }

   230→    /**
   231→     * 开始下载
   232→     */
   233→    private fun startDownload(url: String, onComplete: (String, String?) -> Unit) {
   234→        lateinit var job: Job
   235→        job = viewModelScope.launch(Dispatchers.IO) {
   236→            // 创建下载任务
   237→            val task = DownloadTask(
   238→                url = url,
   239→                title = "正在解析...",
   240→                status = DownloadStatus.FETCHING_INFO
   241→            )
   242→
   243→            try {
   244→                // 添加到活跃下载
   245→                activeDownloadTasks[task.id] = task
   246→                // 保存当前 job 到 Map（需要在协程启动后）
   247→                activeDownloadJobs[task.id] = job
   248→                updateActiveDownloads()
   249→
   250→                // 获取视频信息
   251→                val result = videoDownloadService.getVideoInfo(url)
   252→
   253→                result.onSuccess { videoInfo ->
   254→                    // 检查是否需要Cookie但未配置
   255→                    val needsCookie = when (videoInfo.platform) {
   256→                        VideoDownloadService.VideoPlatform.BILIBILI -> true
   257→                        VideoDownloadService.VideoPlatform.YOUTUBE -> false  // YouTube通常不需要Cookie
   258→                        VideoDownloadService.VideoPlatform.DOUYIN -> false  // 抖音使用API解析
   259→                        VideoDownloadService.VideoPlatform.TIKTOK -> false  // TikTok使用API解析
   260→                        else -> false
   261→                    }
   262→
   263→                    val hasCookie = when (videoInfo.platform) {
   264→                        VideoDownloadService.VideoPlatform.BILIBILI -> videoDownloadService.hasBilibiliCookie()
   265→                        VideoDownloadService.VideoPlatform.YOUTUBE -> videoDownloadService.hasYouTubeCookie()
   266→                        VideoDownloadService.VideoPlatform.DOUYIN -> videoDownloadService.hasDouyinCookie()
   267→                        VideoDownloadService.VideoPlatform.TIKTOK -> videoDownloadService.hasTiktokCookie()
   268→                        else -> true
   269→                    }
   270→
   271→                    if (needsCookie && !hasCookie) {
   272→                        // 需要登录获取Cookie，根据平台显示不同的登录对话框
   273→                        when (videoInfo.platform) {
   274→                            VideoDownloadService.VideoPlatform.BILIBILI -> {
   275→                                _uiState.update {
   276→                                    it.copy(
   277→                                        showBilibiliLogin = true,
   278→                                        pendingDownloadUrl = url,
   279→                                        currentDownload = task.copy(
   280→                                            title = "需要登录",
   281→                                            status = DownloadStatus.WAITING_FOR_AUTH
   282→                                        )
   283→                                    )
   284→                                }
   285→                            }
   286→                            VideoDownloadService.VideoPlatform.YOUTUBE -> {
   287→                                _uiState.update {
   288→                                    it.copy(
   289→                                        showYouTubeLogin = true,
   290→                                        pendingDownloadUrl = url,
   291→                                        currentDownload = task.copy(
   292→                                            title = "需要登录",
   293→                                            status = DownloadStatus.WAITING_FOR_AUTH
   294→                                        )
   295→                                    )
   296→                                }
   297→                            }
   298→                            VideoDownloadService.VideoPlatform.DOUYIN -> {
   299→                                _uiState.update {
   300→                                    it.copy(
   301→                                        showDouyinLogin = true,
   302→                                        pendingDownloadUrl = url,
   303→                                        currentDownload = task.copy(
   304→                                            title = "需要登录",
   305→                                            status = DownloadStatus.WAITING_FOR_AUTH
   306→                                        )
   307→                                    )
   308→                                }
   309→                            }
   310→                            VideoDownloadService.VideoPlatform.TIKTOK -> {
   311→                                _uiState.update {
   312→                                    it.copy(
   313→                                        showTiktokLogin = true,
   314→                                        pendingDownloadUrl = url,
   315→                                        currentDownload = task.copy(
   316→                                            title = "需要登录",
   317→                                            status = DownloadStatus.WAITING_FOR_AUTH
   318→                                        )
   319→                                    )
   320→                                }
   321→                            }
   322→                            else -> { /* 不应该到达这里 */ }
   323→                        }
   324→                        return@launch
   325→                    }
   326→
   327→                    // 更新任务信息
   328→                    val updatedTask = task.copy(
   329→                        title = videoInfo.title,
   330→                        status = DownloadStatus.DOWNLOADING
   331→                    )
   332→                    updateCurrentDownload(updatedTask)
   333→
   334→                    // 生成文件名
   335→                    val safeTitle = videoInfo.title.replace(Regex("[\\\\/:*?\"<>|]"), "_").take(50)
   336→                    val outputFile = File(downloadsDir, "${safeTitle}_${System.currentTimeMillis()}.mp4")
   337→
   338→                    // 使用分段下载器下载，传递Cookie
   339→                    downloadWithSegmentedDownloader(url, videoInfo, safeTitle, outputFile, onComplete)
   340→                }.onFailure { error ->
   341→                    val failedTask = task.copy(
   342→                        status = DownloadStatus.FAILED,
   343→                        errorMessage = "解析失败: ${error.message}\n\n建议：使用导入功能添加本地视频",
   344→                        errorCode = error.javaClass.simpleName
   345→                    )
   346→                    onDownloadFailed(task.id, failedTask, failedTask.errorMessage ?: "解析失败")
   347→                }
   348→
   349→            } catch (e: Exception) {
   350→                android.util.Log.e("HomeViewModel", "Download failed", e)
   351→                val failedTask = task.copy(
   352→                    status = DownloadStatus.FAILED,
   353→                    errorMessage = "下载失败: ${e.message}",
   354→                    errorCode = e.javaClass.simpleName
   355→                )
   356→                onDownloadFailed(task.id, failedTask, failedTask.errorMessage ?: "下载失败")
   357→            }
   358→        }
   359→    }

   361→    /**
   362→     * 使用分段下载器下载视频
   363→     */
   364→    private suspend fun downloadWithSegmentedDownloader(
   365→        url: String,
   366→        videoInfo: com.multilanglearner.data.network.VideoDownloadService.VideoDownloadInfo,
   367→        title: String,
   368→        outputFile: File,
   369→        onComplete: (String, String?) -> Unit
   370→    ) {
   371→        val startTime = System.currentTimeMillis()
   372→        var lastUpdate = startTime
   373→
   374→        try {
   375→            val config = SegmentedDownloader.DownloadConfig(
   376→                url = videoInfo.downloadUrl,
   377→                outputFile = outputFile,
   378→                enableResume = true,
   379→                // 使用当前配置的Cookie（如果可用）
   380→                cookies = currentBilibiliCookie ?: videoInfo.cookies,
   381→                // 音频流URL（用于DASH格式）
   382→                audioUrl = videoInfo.audioUrl,
   383→                onProgress = { progress ->
   384→                    val now = System.currentTimeMillis()
   385→                    // 限制更新频率（每200ms更新一次）
   386→                    if (now - lastUpdate > 200) {
   387→                        val elapsedSeconds = (now - startTime) / 1000.0
   388→                        val speed = progress.speed
   389→                        val remainingBytes = progress.totalBytes - progress.bytesDownloaded
   390→                        val remainingTime = if (speed > 0) remainingBytes / speed else 0L
   391→
   392→                        // 查找对应的任务ID
   393→                        val taskId = activeDownloadTasks.entries.find { it.value.url == url }?.key
   394→                        val currentTask = taskId?.let { activeDownloadTasks[it] }
   395→                        if (taskId != null && currentTask != null) {
   396→                            val updatedTask = currentTask.copy(
   397→                                status = DownloadStatus.DOWNLOADING,
   398→                                progress = progress.percent,
   399→                                bytesDownloaded = progress.bytesDownloaded,
   400→                                totalBytes = progress.totalBytes,
   401→                                speed = speed,
   402→                                remainingTime = remainingTime,
   403→                                title = currentTask.title ?: title
   404→                            )
   405→                            updateActiveDownloadTask(taskId, updatedTask)
   406→                        }
   407→                        lastUpdate = now
   408→                    }
   409→                }
   410→            )
   411→
   412→            val result = segmentedDownloader.download(config)
   413→
   414→            result.onSuccess { file ->
   415→                // 获取任务ID和任务
   416→                val taskId = activeDownloadTasks.entries.find { it.value.url == url }?.key
   417→                val task = taskId?.let { activeDownloadTasks[it] }
   418→
   419→                if (taskId != null && task != null) {
   420→                    val completedTask = task.copy(
   421→                        status = DownloadStatus.COMPLETED,
   422→                        progress = 100,
   423→                        completedAt = System.currentTimeMillis()
   424→                    )
   425→                    onDownloadCompleted(taskId, completedTask, file, onComplete)
   426→                }
   427→            }.onFailure { error ->
   428→                val taskId = activeDownloadTasks.entries.find { it.value.url == url }?.key
   429→                val task = taskId?.let { activeDownloadTasks[it] }
   430→                if (taskId != null && task != null) {
   431→                    onDownloadFailed(taskId, task, "下载失败: ${error.message}")
   432→                }
   433→            }
   434→
   435→        } catch (e: Exception) {
   436→            android.util.Log.e("HomeViewModel", "Segmented download failed", e)
   437→            // 这里不需要额外处理，因为 onFailure 已经处理了
   438→        }
   439→    }

   441→    /**
   442→     * 取消当前下载
   443→     */
   444→    fun cancelDownload(taskId: String? = null) {
   445→        if (taskId != null) {
   446→            // 取消特定任务
   447→            activeDownloadJobs[taskId]?.cancel()
   448→            activeDownloadTasks[taskId]?.let { task ->
   449→                val cancelledTask = task.copy(
   450→                    status = DownloadStatus.CANCELLED,
   451→                    completedAt = System.currentTimeMillis()
   452→                )
   453→                activeDownloadTasks.remove(taskId)
   454→                activeDownloadJobs.remove(taskId)
   455→                updateActiveDownloads()
   456→                updateCurrentDownload(cancelledTask)
   457→                addToHistory(cancelledTask)
   458→
   459→                // 清理临时文件
   460→                if (task.totalBytes > 0) {
   461→                    val safeTitle = task.title.replace(Regex("[\\\\/:*?\"<>|]"), "_").take(50)
   462→                    val outputFile = File(downloadsDir, "${safeTitle}_${task.createdAt}.mp4")
   463→                    segmentedDownloader.cancelDownload(outputFile)
   464→                }
   465→            }
   466→            // 处理队列中的下一个任务
   467→            processQueue()
   468→        } else {
   469→            // 取消所有任务
   470→            activeDownloadJobs.values.forEach { it.cancel() }
   471→            activeDownloadTasks.values.forEach { task ->
   472→                val cancelledTask = task.copy(
   473→                    status = DownloadStatus.CANCELLED,
   474→                    completedAt = System.currentTimeMillis()
   475→                )
   476→                addToHistory(cancelledTask)
   477→
   478→                // 清理临时文件
   479→                if (task.totalBytes > 0) {
   480→                    val safeTitle = task.title.replace(Regex("[\\\\/:*?\"<>|]"), "_").take(50)
   481→                    val outputFile = File(downloadsDir, "${safeTitle}_${task.createdAt}.mp4")
   482→                    segmentedDownloader.cancelDownload(outputFile)
   483→                }
   484→            }
   485→            activeDownloadJobs.clear()
   486→            activeDownloadTasks.clear()
   487→            updateActiveDownloads()
   488→            updateCurrentDownload(null)
   489→        }
   490→    }

   516→    fun deleteFromHistoryByPath(filePath: String) {
   517→        viewModelScope.launch(Dispatchers.IO) {
   518→            // 查找历史记录
   519→            val history = _downloadHistory.value.find { it.filePath == filePath }
   520→            if (history != null) {
   521→                // 删除视频文件
   522→                try {
   523→                    val videoFile = File(history.filePath)
   524→                    if (videoFile.exists()) {
   525→                        videoFile.delete()
   526→                        android.util.Log.d("HomeViewModel", "Deleted video file: ${history.filePath}")
   527→                    }
   528→
   529→                    // 删除相关字幕文件
   530→                    val base = history.filePath.substringBeforeLast(".")
   531→                    listOf(".vtt", ".wordts.json").forEach { ext ->
   532→                        val file = File("$base$ext")
   533→                        if (file.exists()) {
   534→                            file.delete()
   535→                        }
   536→                    }
   537→                } catch (e: Exception) {
   538→                    android.util.Log.e("HomeViewModel", "Error deleting video file", e)
   539→                }
   540→
   541→                // 从历史记录中移除
   542→                _downloadHistory.update { it.filter { h -> h.id != history.id } }
   543→                saveDownloadHistory()
   544→            }
   545→        }
   546→    }

  1048→    private fun getFileName(context: Context, uri: Uri): String? {
  1049→        var result: String? = null
  1050→        if (uri.scheme == "content") {
  1051→            val cursor = context.contentResolver.query(uri, null, null, null, null)
  1052→            cursor?.use {
  1053→                if (it.moveToFirst()) {
  1054→                    val index = it.getColumnIndex(android.provider.OpenableColumns.DISPLAY_NAME)
  1055→                    if (index >= 0) result = it.getString(index)
  1056→                }
  1057→            }
  1058→        }
  1059→        if (result == null) {
  1060→            result = uri.path
  1061→            val cut = result?.lastIndexOf('/')
  1062→            if (cut != -1) {
  1063→                result = result?.substring(cut!! + 1)
  1064→            }
  1065→        }
  1066→        return result
  1067→    }
