   214→private fun parseContentDispositionFileName(contentDisposition: String?): String? {
   215→    if (contentDisposition == null) return null
   216→    Regex("filename\\*=(?:UTF-8''|utf-8'')(.+)").find(contentDisposition)?.let {
   217→        return try { java.net.URLDecoder.decode(it.groupValues[1], "UTF-8") } catch (_: Exception) { null }
   218→    }
   219→    Regex("filename=\"?([^\"]+)\"?").find(contentDisposition)?.let {
   220→        return it.groupValues[1].trim()
   221→    }
   222→    return null
   223→}
