    41→        viewModelScope.launch {
    42→            _syncStatus.value = "同步中..."

    44→                val count = withContext(Dispatchers.IO) {
    45→                    val latestTime = db.smsDao().getLatestTime() ?: 0

    50→                    for (r in records) {

    53→                        Log.d(TAG, "syncSms: extracted ${urls.size} urls: $urls")

    55→                        if (id > 0) {
    56→                            c++
    57→                            SmsProcessor.process(getApplication(), db, r.copy(id = id))
    58→                        }
    59→                    }

    61→                }
