     1→package com.multilanglearner.data.network
     2→
     3→import android.util.Log
     4→import com.multilanglearner.domain.model.AIConfig
     5→import com.multilanglearner.domain.model.WordTimestamp
     6→import kotlinx.serialization.Serializable
     7→import kotlinx.serialization.json.*
     8→import okhttp3.*
     9→import okhttp3.MediaType.Companion.toMediaType
    10→import okhttp3.RequestBody.Companion.asRequestBody
    11→import okhttp3.RequestBody.Companion.toRequestBody
    12→import java.io.File
    13→import java.util.concurrent.TimeUnit
    14→import javax.inject.Inject
    15→import javax.inject.Singleton
    16→
    17→@Serializable
    18→data class WhisperResult(
    19→    val success: Boolean,
    20→    val text: String = "",
    21→    val words: List<WordTimestamp> = emptyList(),
    22→    val detectedLang: String = "",
    23→    val error: String = ""
    24→)
    25→
    26→@Singleton
    27→class WhisperService @Inject constructor() {
    28→    private val client = OkHttpClient.Builder()
    29→        .connectTimeout(15, TimeUnit.SECONDS)  // 减少连接超时，加快重试
    30→        .readTimeout(600, TimeUnit.SECONDS)    // 增加读取超时，支持长音频
    31→        .writeTimeout(120, TimeUnit.SECONDS)   // 增加写入超时
    32→        .retryOnConnectionFailure(true)        // 启用自动重试
    33→        .build()
    34→
    35→    private val json = Json { ignoreUnknownKeys = true }
    36→
    37→    // Groq API 文件大小限制（官方限制 25MB，设置安全值）
    38→    // 参考：https://console.groq.com/docs/speech-to-text
    39→    private val MAX_FILE_SIZE = 25 * 1024 * 1024L
    40→    private val MIN_FILE_SIZE = 1024L // 最小 1KB，避免空文件或太小的文件
    41→
    42→    // 最大重试次数
    43→    private val MAX_RETRIES = 3
    44→
    45→    /**
    46→     * Transcribe audio via Groq Whisper API
    47→     * 优先使用 groqApiKey，其次使用 provider 为 groq 时的 apiKey
    48→     * 添加自动重试机制，包括 API 配额限制处理
    49→     */
    50→    suspend fun transcribeGroq(audioFile: File, config: AIConfig, lang: String? = null): WhisperResult? {
    51→        // 自动重试逻辑
    52→        var lastError: Exception? = null
    53→        repeat(MAX_RETRIES) { attempt ->
    54→            if (attempt > 0) {
    55→                Log.d("WhisperService", "Retry $attempt/$MAX_RETRIES")
    56→                kotlinx.coroutines.delay(2000L * attempt) // 递增延迟
    57→            }
    58→
    59→            val result = transcribeGroqInternal(audioFile, config, lang)
    60→            if (result?.success == true) {
    61→                return result
    62→            }
    63→            lastError = Exception(result?.error ?: "Unknown error")
    64→        }
    65→
    66→        // 所有重试都失败
    67→        return WhisperResult(
    68→            success = false,
    69→            error = "❌ 转录失败（已重试 $MAX_RETRIES 次）\n\n${lastError?.message ?: "未知错误"}\n\n建议：检查网络连接和 API Key 配置"
    70→        )
    71→    }
    72→
    73→    /**
    74→     * 检查是否是速率限制错误（429）
    75→     * 从错误消息中提取需要等待的时间（秒）
    76→     */
    77→    private fun isRateLimitError(errorMsg: String): Pair<Boolean, Int> {
    78→        val rateLimitPattern = Regex("Please try again in (\\d+)s")
    79→        val match = rateLimitPattern.find(errorMsg)
    80→        return if (match != null) {
    81→            val seconds = match.groupValues[1].toIntOrNull() ?: 0
    82→            Pair(true, seconds)
    83→        } else {
    84→            Pair(false, 0)
    85→        }
    86→    }
    87→
    88→    /**
    89→     * 内部转录实现
    90→     */
    91→    private suspend fun transcribeGroqInternal(audioFile: File, config: AIConfig, lang: String? = null): WhisperResult? {
    92→        // 获取 Groq API Key：优先使用专门的 groqApiKey，其次检查 provider 是否为 groq
    93→        val apiKey = config.groqApiKey.takeIf { it.isNotBlank() }
    94→            ?: (config.apiKey.takeIf { config.provider == "groq" && it.isNotBlank() })
    95→            ?: return WhisperResult(success = false, error = "❌ 未配置 Groq API Key\n\n请前往「AI 设置」配置 Groq API Key。\n获取 API Key：https://console.groq.com/keys")
    96→
    97→        val whisperModel = config.whisperModel.ifBlank { "whisper-large-v3-turbo" }
    98→
    99→        // 检查文件大小
   100→        val fileSize = audioFile.length()
   101→        Log.d("WhisperService", "File size: $fileSize bytes, max: $MAX_FILE_SIZE")
   102→
   103→        // 检查最小文件大小
   104→        if (fileSize < MIN_FILE_SIZE) {
   105→            Log.e("WhisperService", "File too small: $fileSize bytes (min: $MIN_FILE_SIZE)")
   106→            return WhisperResult(
   107→                success = false,
   108→                error = "❌ 视频文件过小\n\n文件大小: ${fileSize / 1024} KB\n最小要求: 1 KB\n\n建议：请使用包含音频内容的有效视频文件"
   109→            )
   110→        }
   111→
   112→        if (fileSize > MAX_FILE_SIZE) {
   113→            val sizeMB = String.format("%.2f", fileSize / 1024.0 / 1024.0)
   114→            Log.e("WhisperService", "File too large: $sizeMB MB (max: 25 MB)")
   115→            return WhisperResult(
   116→                success = false,
   117→                error = "❌ 视频文件过大\n\n文件大小: $sizeMB MB\nGroq 限制: 25 MB\n\n建议解决方案：\n1. 应用会自动分段处理大文件\n2. 如果仍然失败，请使用较短的视频\n3. 或在设置中切换到其他 AI 提供商"
   118→            )
   119→        }
   120→
   121→        Log.d("WhisperService", "Starting transcription with Groq Whisper: model=$whisperModel, file=${audioFile.name}, size=$fileSize")
   122→
   123→        // 根据语言设置合适的 prompt，引导 Whisper 更准确地识别
   124→        val prompt = when (lang) {
   125→            "en" -> "Transcribe accurately with proper punctuation and capitalization."
   126→            "ja" -> "正確に句読点を付けて書き起こしてください。"
   127→            "ko" -> "정확하게 구두점을 포함하여 전사하세요."
   128→            "ru" -> "Транскрибируйте точно с правильной пунктуацией."
   129→            "th" -> "ถอดความอย่างถูกต้องพร้อมเครื่องหมายวรรคตอนที่เหมาะสม"
   130→            "vi" -> "Phiên âm chính xác với dấu câu đúng."
   131→            "ms" -> "Transkripsi dengan tepat dan tanda baca yang betul."
   132→            "id" -> "Transkripsikan dengan akurat dan tanda baca yang benar."
   133→            "ar" -> "انسخ بدقة مع علامات الترقيم الصحيحة."
   134→            "pt" -> "Transcreva com precisão e pontuação correta."
   135→            "es" -> "Transcribe con precisión y puntuación correcta."
   136→            "fr" -> "Transcrire avec précision et ponctuation correcte."
   137→            "de" -> "Transkribieren Sie genau mit korrekter Zeichensetzung."
   138→            "hi" -> "सटीक विराम चिह्नों के साथ सही ढंग से लिप्यंतरण करें।"
   139→            "zh" -> "准确转录，使用正确的标点符号。"
   140→            else -> "Transcribe accurately with proper punctuation and capitalization."
   141→        }
   142→
   143→        val bodyBuilder = MultipartBody.Builder().setType(MultipartBody.FORM)
   144→            .addFormDataPart("file", audioFile.name, audioFile.asRequestBody("application/octet-stream".toMediaType()))
   145→            .addFormDataPart("model", whisperModel)
   146→            .addFormDataPart("response_format", "verbose_json")
   147→            .addFormDataPart("timestamp_granularities[]", "word")
   148→            .addFormDataPart("timestamp_granularities[]", "segment")
   149→            .addFormDataPart("temperature", "0")
   150→            .addFormDataPart("prompt", prompt)
   151→
   152→        lang?.let { bodyBuilder.addFormDataPart("language", it) }
   153→
   154→        val request = Request.Builder()
   155→            .url("https://api.groq.com/openai/v1/audio/transcriptions")
   156→            .addHeader("Authorization", "Bearer $apiKey")
   157→            .post(bodyBuilder.build())
   158→            .build()
   159→
   160→        return try {
   161→            Log.d("WhisperService", "Sending request to Groq API...")
   162→            val response = client.newCall(request).execute()
   163→
   164→            Log.d("WhisperService", "Response code: ${response.code}")
   165→
   166→            if (response.code != 200) {
   167→                val errorBody = response.body?.string() ?: "Unknown error"
   168→                Log.e("WhisperService", "API error: $errorBody")
   169→
   170→                // 解析错误信息
   171→                val errorMsg = try {
   172→                    val errorJson = json.parseToJsonElement(errorBody).jsonObject
   173→                    val errObj = errorJson["error"]?.jsonObject
   174→                    val errMsg = errObj?.get("message")?.jsonPrimitive?.contentOrNull
   175→                    val errType = errObj?.get("type")?.jsonPrimitive?.contentOrNull
   176→                    val errCode = errObj?.get("code")?.jsonPrimitive?.contentOrNull
   177→
   178→                    when {
   179→                        errMsg?.contains("invalid_api_key", ignoreCase = true) == true ||
   180→                        errMsg?.contains("Unauthorized", ignoreCase = true) == true ||
   181→                        response.code == 401 ->
   182→                            "❌ API Key 无效\n\n请检查 Groq API Key 是否正确配置\n\n获取 API Key：https://console.groq.com/keys"
   183→                        errMsg?.contains("quota", ignoreCase = true) == true ||
   184→                        errMsg?.contains("limit", ignoreCase = true) == true ||
   185→                        errMsg?.contains("rate_limit", ignoreCase = true) == true ||
   186→                        response.code == 429 ->
   187→                            "❌ API 配额已用完\n\nGroq API 使用次数已达限制\n\n建议：\n1. 等待配额重置\n2. 更换 API Key\n3. 升级 Groq 账户"
   188→                        errMsg?.contains("forbidden", ignoreCase = true) == true ||
   189→                        errMsg?.contains("banned", ignoreCase = true) == true ||
   190→                        response.code == 403 ->
   191→                            "❌ 转录被禁止\n\n${errMsg ?: "访问被限制"}\n\n可能原因：\n1. API Key 被禁用\n2. 账户存在违规行为\n3. 内容不符合使用条款"
   192→                        else -> "❌ 转录失败 ($response.code)\n\n${errMsg ?: errorBody}"
   193→                    }
   194→                } catch (e: Exception) {
   195→                    "❌ 转录失败\n\n${errorBody.take(200)}"
   196→                }
   197→
   198→                return WhisperResult(success = false, error = errorMsg)
   199→            }
   200→
   201→            val body = response.body?.string() ?: return WhisperResult(success = false, error = "空响应")
   202→            Log.d("WhisperService", "Response length: ${body.length}")
   203→            Log.d("WhisperService", "Response body preview: ${body.take(500)}")
   204→
   205→            // 检查响应是否为有效的 JSON
   206→            if (!body.trim().startsWith("{") && !body.trim().startsWith("[")) {
   207→                Log.e("WhisperService", "Response is not valid JSON: ${body.take(200)}")
   208→                return WhisperResult(
   209→                    success = false,
   210→                    error = "❌ API 返回格式错误\n\nGroq API 返回了非 JSON 格式的响应。\n\n响应内容：\n${body.take(500)}"
   211→                )
   212→            }
   213→
   214→            try {
   215→                val jsonObj = json.parseToJsonElement(body).jsonObject
   216→                val text = jsonObj["text"]?.jsonPrimitive?.contentOrNull ?: ""
   217→                val detectedLang = jsonObj["language"]?.jsonPrimitive?.contentOrNull ?: ""
   218→
   219→                Log.d("WhisperService", "Transcribed text length: ${text.length}, detected language: $detectedLang")
   220→
   221→                val words = jsonObj["words"]?.jsonArray?.map { w ->
   222→                    val obj = w.jsonObject
   223→                    WordTimestamp(
   224→                        w = obj["word"]?.jsonPrimitive?.contentOrNull ?: "",
   225→                        s = (obj["start"]?.jsonPrimitive?.doubleOrNull ?: 0.0),
   226→                        e = (obj["end"]?.jsonPrimitive?.doubleOrNull ?: 0.0)
   227→                    )
   228→                } ?: emptyList()
   229→
   230→                Log.d("WhisperService", "Word timestamps count: ${words.size}")
   231→
   232→                WhisperResult(
   233→                    success = true,
   234→                    text = text,
   235→                    words = words,
   236→                    detectedLang = detectedLang
   237→                )
   238→            } catch (e: Exception) {
   239→                Log.e("WhisperService", "JSON parsing error", e)
   240→                Log.e("WhisperService", "Response body: $body")
   241→                WhisperResult(
   242→                    success = false,
   243→                    error = "❌ 响应解析失败\n\nGroq API 返回了无法解析的响应。\n\n错误: ${e.message}\n\n响应内容：\n${body.take(500)}"
   244→                )
   245→            }
   246→        } catch (e: Exception) {
   247→            Log.e("WhisperService", "Transcription error", e)
   248→            val errorMsg = when {
   249→                e.message?.contains("timeout", ignoreCase = true) == true ->
   250→                    "❌ 请求超时\n\n网络连接超时，请检查网络连接后重试。"
   251→                e.message?.contains("Unable to resolve host", ignoreCase = true) == true ||
   252→                e.message?.contains("UnknownHost", ignoreCase = true) == true ->
   253→                    "❌ 网络连接失败\n\n无法连接到 Groq API 服务器，请检查网络连接。"
   254→                else -> "❌ 转录失败\n\n${e.message ?: "未知错误"}\n\n请检查网络连接和 API Key 配置后重试。"
   255→            }
   256→            WhisperResult(success = false, error = errorMsg)
   257→        }
   258→    }
   259→}
