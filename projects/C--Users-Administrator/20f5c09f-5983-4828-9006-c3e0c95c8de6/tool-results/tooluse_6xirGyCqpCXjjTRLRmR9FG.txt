   119→    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
   120→        when (intent?.action) {
   121→            ACTION_START -> {
   122→                val videoPath = intent.getStringExtra(EXTRA_VIDEO_PATH)
   123→                val videoName = intent.getStringExtra(EXTRA_VIDEO_NAME) ?: "视频"
   124→
   125→                // 检查是否已有正在进行的转录
   126→                if (transcriptionStatus == 1) {
   127→                    Log.w(TAG, "Transcription already in progress, ignoring start request")
   128→                    return START_NOT_STICKY
   129→                }
   130→
   131→                if (videoPath != null) {
   132→                    startTranscription(videoPath, videoName)
   133→                }
   134→            }
   135→            ACTION_CANCEL -> {
   136→                cancelTranscription()
   137→            }
   138→        }
   139→        return START_STICKY
   140→    }

   142→    /**
   143→     * 开始转录
   144→     */
   145→    private fun startTranscription(videoPath: String, videoName: String) {
   146→        Log.d(TAG, "Starting transcription for: $videoPath")
   147→
   148→        val transcriptionId = System.currentTimeMillis().toString()
   149→
   150→        // 只有用户明确触发才执行转录
   151→        userTriggered = true
   152→
   153→        // 更新转录状态为转录中
   154→        transcriptionStatus = 1
   155→
   156→        // 直接并行执行，不再排队
   157→        executeTranscription(transcriptionId, videoPath, videoName)
   158→    }

   164→        userTriggered = false
   165→        // 完成或取消后，状态改为空闲
   166→        transcriptionStatus = 0
   167→    }

   176→    /**
   177→     * 执行转录任务
   178→     */
   179→    private fun executeTranscription(
   180→        transcriptionId: String,
   181→        videoPath: String,
   182→        videoName: String
   183→    ) {
   184→        currentTranscriptionId = transcriptionId
   185→        isRunning = true
   186→        currentVideoName = videoName
   187→        currentProgress = 0
   188→        currentMessage = "正在准备..."
   189→
   190→        // 启动前台服务
   191→        startForeground(NOTIFICATION_ID, createProgressNotification(videoName, 0, 0, "正在准备..."))
   192→
   193→        val job = serviceScope.launch {
   194→            try {
   195→                _serviceState.value = ServiceState.Transcribing(0, 0, 0, "正在准备...")
   196→
   197→                val config = aiConfigStore.config.first()
   198→                val videoFile = File(videoPath)
   199→
   200→                // 检查 API Key
   201→                val hasGroqKey = config.groqApiKey.isNotBlank() ||
   202→                                 (config.provider == "groq" && config.apiKey.isNotBlank())
   203→
   204→                if (!hasGroqKey) {
   205→                    sendErrorBroadcast("❌ 请先配置 Groq API Key\n\n获取 API Key：https://console.groq.com/keys")
   206→                    _serviceState.value = ServiceState.Error("请先配置 Groq API Key")
   207→                    stopForeground(STOP_FOREGROUND_REMOVE)
   208→                    stopSelf()
   209→                    return@launch
   210→                }
   211→
   212→                // 执行转录 - 使用用户指定的转录语言（如果设置了）
   213→                val transcribeLang = config.transcribeLanguage.takeIf { it.isNotBlank() }
   214→                val result = ffmpegSegmentedService.transcribeWithSegments(
   215→                    videoFile,
   216→                    config,
   217→                    transcribeLang,
   218→                    onProgress = { segment, total, progress, message, remainingSeconds ->
   219→                        updateProgress(videoName, segment, total, progress, message)
   220→                        sendProgressBroadcast(segment, total, progress, message, remainingSeconds)
   221→                        _serviceState.value = ServiceState.Transcribing(
   222→                            segment, total, progress, message
   223→                        )
   224→                    }
   225→                )
   226→
   227→                // 转录完成
   228→                if (result != null && result.success) {
   229→                    val base = videoPath.substringBeforeLast(".")
   230→                    val vttPath = base + ".vtt"
   231→                    val wtsPath = base + ".wordts.json"
   232→
   233→                    // 如果字幕文件还不存在（可能是其他转录任务创建的），保存它们
   234→                    if (!File(vttPath).exists()) {
   235→                        saveSubtitles(result, vttPath, wtsPath)
   236→                    }
   237→
   238→                    // 保存完成后立即停止服务，避免自动继续
   239→                    sendCompletedBroadcast(true, result.detectedLang, result.text, videoPath, vttPath)
   240→                    _serviceState.value = ServiceState.Completed(
   241→                        result.detectedLang,
   242→                        result.text
   243→                    )
   244→
   245→                    updateNotificationComplete(videoName)
   246→                    delay(3000) // 显示完成通知3秒
   247→
   248→                    // 重置用户触发标志，服务将停止
   249→                    resetUserTrigger()
   250→
   251→                    // 移除已完成的任务
   252→                    activeJobs.remove(transcriptionId)
   253→                    if (activeJobs.isEmpty()) {
   254→                        isRunning = false
   255→                        currentVideoName = null
   256→                        stopForeground(STOP_FOREGROUND_REMOVE)
   257→                        stopSelf()
   258→                    }
   259→
   260→                } else {
   261→                    val error = result?.error ?: "未知错误"
   262→                    sendErrorBroadcast(error)
   263→                    _serviceState.value = ServiceState.Error(error)
   264→
   265→                    // 出错后也重置用户触发标志
   266→                    resetUserTrigger()
   267→
   268→                    activeJobs.remove(transcriptionId)
   269→                    if (activeJobs.isEmpty()) {
   270→                        isRunning = false
   271→                        currentVideoName = null
   272→                        stopForeground(STOP_FOREGROUND_REMOVE)
   273→                        stopSelf()
   274→                    }
   275→                }
   276→
   277→            } catch (e: Exception) {
   278→                Log.e(TAG, "Transcription error", e)
   279→                val error = "转录失败: ${e.message}"
   280→                sendErrorBroadcast(error)
   281→                _serviceState.value = ServiceState.Error(error)
   282→
   283→                activeJobs.remove(transcriptionId)
   284→                if (activeJobs.isEmpty()) {
   285→                    isRunning = false
   286→                    currentVideoName = null
   287→                    stopForeground(STOP_FOREGROUND_REMOVE)
   288→                    stopSelf()
   289→                }
   290→            }
   291→        }
   292→        activeJobs[transcriptionId] = job
   293→    }

   295→    /**
   296→     * 取消转录
   297→     */
   298→    private fun cancelTranscription() {
   299→        activeJobs.values.forEach { it.cancel() }
   300→        activeJobs.clear()
   301→        isRunning = false
   302→        currentVideoName = null
   303→        currentProgress = 0
   304→        currentMessage = ""
   305→        _serviceState.value = ServiceState.Idle
   306→
   307→        // 重置用户触发标志
   308→        resetUserTrigger()
   309→
   310→        stopForeground(STOP_FOREGROUND_REMOVE)
   311→        stopSelf()
   312→        Log.d(TAG, "All transcriptions cancelled")
   313→    }

   315→    /**
   316→     * 保存字幕文件
   317→     */
   318→    private suspend fun saveSubtitles(
   319→        result: WhisperResult,
   320→        vttPath: String,
   321→        wtsPath: String
   322→    ) = withContext(Dispatchers.IO) {
   323→        val vttBuilder = StringBuilder("WEBVTT\n\n")
   324→        val allWordTimings = mutableListOf<WordTimestamp>()
   325→
   326→        if (result.words.isNotEmpty()) {
   327→            var curWords = mutableListOf<WordTimestamp>()
   328→            for (wt in result.words) {
   329→                curWords.add(wt)
   330→                if (wt.w.trim().matches(Regex(".*[.!?。！？]$")) && curWords.isNotEmpty()) {
   331→                    val sentText = curWords.joinToString(" ") { it.w }
   332→                    val sStart = curWords.first().s
   333→                    val sEnd = curWords.last().e
   334→                    vttBuilder.append("${VttParser.formatVtt(sStart)} --> ${VttParser.formatVtt(sEnd)}\n$sentText\n\n")
   335→                    allWordTimings.addAll(curWords)
   336→                    curWords = mutableListOf()
   337→                }
   338→            }
   339→            if (curWords.isNotEmpty()) {
   340→                val sentText = curWords.joinToString(" ") { it.w }
   341→                val sStart = curWords.first().s
   342→                val sEnd = curWords.last().e
   343→                vttBuilder.append("${VttParser.formatVtt(sStart)} --> ${VttParser.formatVtt(sEnd)}\n$sentText\n\n")
   344→                allWordTimings.addAll(curWords)
   345→            }
   346→        }
   347→
   348→        File(vttPath).writeText(vttBuilder.toString())
   349→        if (allWordTimings.isNotEmpty()) {
   350→            val json = Json { ignoreUnknownKeys = true }
   351→            File(wtsPath).writeText(json.encodeToString(ListSerializer(WordTimestamp.serializer()), allWordTimings))
   352→        }
   353→
   354→        Log.d(TAG, "Subtitles saved: ${allWordTimings.size} words")
   355→    }

   357→    private fun updateProgress(videoName: String, segment: Int, total: Int, progress: Int, message: String) {
   358→        val notification = createProgressNotification(videoName, segment, total, message)
   359→        val notificationManager = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
   360→        notificationManager.notify(NOTIFICATION_ID, notification)
   361→    }
   362→
   363→    private fun updateNotificationComplete(videoName: String) {
   364→        val notification = createCompleteNotification(videoName)
   365→        val notificationManager = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
   366→        notificationManager.notify(NOTIFICATION_ID, notification)
   367→    }

   369→    private fun sendProgressBroadcast(segment: Int, total: Int, progress: Int, message: String, remaining: Long) {
   370→        // 更新静态状态供页面恢复时查询
   371→        currentProgress = progress
   372→        currentMessage = message
   373→
   374→        val intent = Intent(ACTION_PROGRESS).apply {
   375→            putExtra(EXTRA_SEGMENT, segment)
   376→            putExtra(EXTRA_TOTAL, total)
   377→            putExtra(EXTRA_PROGRESS, progress)
   378→            putExtra(EXTRA_MESSAGE, message)
   379→            putExtra(EXTRA_REMAINING, remaining)
   380→        }
   381→        sendBroadcast(intent)
   382→    }
   383→
   384→    private fun sendCompletedBroadcast(success: Boolean, detectedLang: String, text: String, videoPath: String? = null, vttPath: String? = null) {
   385→        val intent = Intent(ACTION_COMPLETED).apply {
   386→            putExtra(EXTRA_SUCCESS, success)
   387→            putExtra(EXTRA_DETECTED_LANG, detectedLang)
   388→            // 计算字幕数量（粗略估计：按句子分割）
   389→            val subtitleCount = text.split(Regex("[.!?。！？]")).filter { it.isNotBlank() }.size
   390→            putExtra(EXTRA_SUBTITLE_COUNT, subtitleCount)
   391→            videoPath?.let { putExtra(EXTRA_VIDEO_PATH, it) }
   392→            vttPath?.let { putExtra(EXTRA_VTT_PATH, it) }
   393→        }
   394→        sendBroadcast(intent)
   395→    }
   396→
   397→    private fun sendErrorBroadcast(error: String) {
   398→        val intent = Intent(ACTION_COMPLETED).apply {
   399→            putExtra(EXTRA_SUCCESS, false)
   400→            putExtra(EXTRA_ERROR, error)
   401→        }
   402→        sendBroadcast(intent)
   403→    }

   423→    /**
   424→     * 创建进度通知
   425→     */
   426→    private fun createProgressNotification(
   427→        videoName: String,
   428→        segment: Int,
   429→        total: Int,
   430→        message: String
   431→    ): Notification {
   432→        val openIntent = Intent(this, com.multilanglearner.MainActivity::class.java).apply {
   433→            action = ACTION_OPEN
   434→            flags = Intent.FLAG_ACTIVITY_NEW_TASK or Intent.FLAG_ACTIVITY_CLEAR_TOP
   435→        }
   436→        val openPendingIntent = PendingIntent.getActivity(
   437→            this, 0, openIntent,
   438→            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
   439→        )
   440→
   441→        val cancelIntent = Intent(this, TranscriptionService::class.java).apply {
   442→            action = ACTION_CANCEL
   443→        }
   444→        val cancelPendingIntent = PendingIntent.getService(
   445→            this, 0, cancelIntent,
   446→            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
   447→        )
   448→
   449→        val progressText = if (total > 1) {
   450→            "分段 $segment/$total ($message)"
   451→        } else {
   452→            message
   453→        }
   454→
   455→        return NotificationCompat.Builder(this, CHANNEL_ID)
   456→            .setContentTitle("正在转录: $videoName")
   457→            .setContentText(progressText)
   458→            .setSmallIcon(android.R.drawable.ic_dialog_info)
   459→            .setOngoing(true)
   460→            .setAutoCancel(false)
   461→            .setContentIntent(openPendingIntent)
   462→            .addAction(android.R.drawable.ic_menu_close_clear_cancel, "取消", cancelPendingIntent)
   463→            .setProgress(total, segment, false)
   464→            .build()
   465→    }

   508→        val message: String

   515→        val error: String
