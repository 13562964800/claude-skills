   185→     * 下载视频 - 支持 B站、YouTube 等平台
   186→     * 支持最多3个并发下载
   187→     */
   188→    fun downloadVideo(onComplete: (String, String?) -> Unit) {
   189→        val url = _uiState.value.url.trim()
   190→
   191→        if (url.isBlank()) {
   192→            // 不再使用 currentDownload，直接在 activeDownloads 中添加错误任务
   193→            return
   194→        }
   195→
   196→        // 检查是否已存在相同视频（通过 URL 匹配）
   197→        val existingVideo = _uiState.value.videos.find { it.sourceUrl == url }
   198→        if (existingVideo != null) {
   199→            // 显示重复下载确认对话框
   200→            _uiState.update { it.copy(
   201→                showDuplicateDownloadConfirm = true,
   202→                duplicateVideoTitle = existingVideo.title
   203→            )}
   204→            return
   205→        }
   206→
   207→        // 如果已达最大并发数，加入队列
   208→        if (activeDownloadTasks.size >= MAX_CONCURRENT_DOWNLOADS) {
   209→            addToQueue(url)
   210→            return
   211→        }
   212→
   213→        startDownload(url, onComplete)
   214→    }

   242→    /**
   243→     * 开始下载
   244→     */
   245→    private fun startDownload(url: String, onComplete: (String, String?) -> Unit) {
   246→        lateinit var job: Job
   247→        job = viewModelScope.launch(Dispatchers.IO) {
   248→            // 创建下载任务
   249→            val task = DownloadTask(
   250→                url = url,
   251→                title = "正在解析...",
   252→                status = DownloadStatus.FETCHING_INFO
   253→            )
   254→
   255→            try {
   256→                // 添加到活跃下载
   257→                activeDownloadTasks[task.id] = task
   258→                // 保存当前 job 到 Map（需要在协程启动后）
   259→                activeDownloadJobs[task.id] = job
   260→                updateActiveDownloads()
   261→
   262→                // 获取视频信息
   263→                val result = videoDownloadService.getVideoInfo(url)
   264→
   265→                result.onSuccess { videoInfo ->
   266→                    // 检查是否需要Cookie但未配置
   267→                    val needsCookie = when (videoInfo.platform) {
   268→                        VideoDownloadService.VideoPlatform.BILIBILI -> true
   269→                        VideoDownloadService.VideoPlatform.YOUTUBE -> false  // YouTube通常不需要Cookie
   270→                        VideoDownloadService.VideoPlatform.DOUYIN -> false  // 抖音使用API解析
   271→                        VideoDownloadService.VideoPlatform.TIKTOK -> false  // TikTok使用API解析
   272→                        else -> false
   273→                    }
   274→
   275→                    val hasCookie = when (videoInfo.platform) {
   276→                        VideoDownloadService.VideoPlatform.BILIBILI -> videoDownloadService.hasBilibiliCookie()
   277→                        VideoDownloadService.VideoPlatform.YOUTUBE -> videoDownloadService.hasYouTubeCookie()
   278→                        VideoDownloadService.VideoPlatform.DOUYIN -> videoDownloadService.hasDouyinCookie()
   279→                        VideoDownloadService.VideoPlatform.TIKTOK -> videoDownloadService.hasTiktokCookie()
   280→                        else -> true
   281→                    }
   282→
   283→                    if (needsCookie && !hasCookie) {
   284→                        // 需要登录获取Cookie，根据平台显示不同的登录对话框
   285→                        // 先添加一个等待认证的任务到 activeDownloads
   286→                        val waitingTask = task.copy(
   287→                            title = "需要登录",
   288→                            status = DownloadStatus.WAITING_FOR_AUTH
   289→                        )
   290→                        activeDownloadTasks[task.id] = waitingTask
   291→                        updateActiveDownloads()
   292→
   293→                        when (videoInfo.platform) {
   294→                            VideoDownloadService.VideoPlatform.BILIBILI -> {
   295→                                _uiState.update {
   296→                                    it.copy(
   297→                                        showBilibiliLogin = true,
   298→                                        pendingDownloadUrl = url
   299→                                    )
   300→                                }
   301→                            }
   302→                            VideoDownloadService.VideoPlatform.YOUTUBE -> {
   303→                                _uiState.update {
   304→                                    it.copy(
   305→                                        showYouTubeLogin = true,
   306→                                        pendingDownloadUrl = url
   307→                                    )
   308→                                }
   309→                            }
   310→                            VideoDownloadService.VideoPlatform.DOUYIN -> {
   311→                                _uiState.update {
   312→                                    it.copy(
   313→                                        showDouyinLogin = true,
   314→                                        pendingDownloadUrl = url
   315→                                    )
   316→                                }
   317→                            }
   318→                            VideoDownloadService.VideoPlatform.TIKTOK -> {
   319→                                _uiState.update {
   320→                                    it.copy(
   321→                                        showTiktokLogin = true,
   322→                                        pendingDownloadUrl = url
   323→                                    )
   324→                                }
   325→                            }
   326→                            else -> { /* 不应该到达这里 */ }
   327→                        }
   328→                        return@launch
   329→                    }
   330→
   331→                    // 更新任务信息并显示在 activeDownloads 中
   332→                    val updatedTask = task.copy(
   333→                        title = videoInfo.title,
   334→                        status = DownloadStatus.DOWNLOADING
   335→                    )
   336→                    activeDownloadTasks[task.id] = updatedTask
   337→                    updateActiveDownloads()
   338→
   339→                    // 生成文件名
   340→                    val safeTitle = videoInfo.title.replace(Regex("[\\\\/:*?\"<>|]"), "_").take(50)
   341→                    val outputFile = File(downloadsDir, "${safeTitle}_${System.currentTimeMillis()}.mp4")
   342→
   343→                    // 使用分段下载器下载，传递Cookie和taskId
   344→                    downloadWithSegmentedDownloader(task.id, url, videoInfo, safeTitle, outputFile, onComplete)
   345→                }.onFailure { error ->
   346→                    val failedTask = task.copy(
   347→                        status = DownloadStatus.FAILED,
   348→                        errorMessage = "解析失败: ${error.message}\n\n建议：使用导入功能添加本地视频",
   349→                        errorCode = error.javaClass.simpleName
   350→                    )
   351→                    onDownloadFailed(task.id, failedTask, failedTask.errorMessage ?: "解析失败")
   352→                }
   353→
   354→            } catch (e: Throwable) {
   355→                android.util.Log.e("HomeViewModel", "Download failed", e)
   356→                val failedTask = task.copy(
   357→                    status = DownloadStatus.FAILED,
   358→                    errorMessage = "下载失败: ${e.message}",
   359→                    errorCode = e.javaClass.simpleName
   360→                )
   361→                onDownloadFailed(task.id, failedTask, failedTask.errorMessage ?: "下载失败")
   362→            }
   363→        }
   364→    }

   366→    /**
   367→     * 使用分段下载器下载视频
   368→     */
   369→    private suspend fun downloadWithSegmentedDownloader(
   370→        taskId: String,
   371→        url: String,
   372→        videoInfo: com.multilanglearner.data.network.VideoDownloadService.VideoDownloadInfo,
   373→        title: String,
   374→        outputFile: File,
   375→        onComplete: (String, String?) -> Unit
   376→    ) {
   377→        val startTime = System.currentTimeMillis()
   378→        var lastUpdate = startTime
   379→
   380→        try {
   381→            val config = SegmentedDownloader.DownloadConfig(
   382→                url = videoInfo.downloadUrl,
   383→                outputFile = outputFile,
   384→                enableResume = true,
   385→                // 使用当前配置的Cookie（如果可用）
   386→                cookies = currentBilibiliCookie ?: videoInfo.cookies,
   387→                // 音频流URL（用于DASH格式）
   388→                audioUrl = videoInfo.audioUrl,
   389→                onProgress = { progress ->
   390→                    val now = System.currentTimeMillis()
   391→                    // 限制更新频率（每200ms更新一次）
   392→                    if (now - lastUpdate > 200) {
   393→                        val elapsedSeconds = (now - startTime) / 1000.0
   394→                        val speed = progress.speed
   395→                        val remainingBytes = progress.totalBytes - progress.bytesDownloaded
   396→                        val remainingTime = if (speed > 0) remainingBytes / speed else 0L
   397→
   398→                        // 直接使用 taskId 更新对应的任务进度
   399→                        val currentTask = activeDownloadTasks[taskId]
   400→                        if (currentTask != null) {
   401→                            val updatedTask = currentTask.copy(
   402→                                status = DownloadStatus.DOWNLOADING,
   403→                                progress = progress.percent,
   404→                                bytesDownloaded = progress.bytesDownloaded,
   405→                                totalBytes = progress.totalBytes,
   406→                                speed = speed,
   407→                                remainingTime = remainingTime,
   408→                                title = currentTask.title ?: title
   409→                            )
   410→                            updateActiveDownloadTask(taskId, updatedTask)
   411→                        }
   412→                        lastUpdate = now
   413→                    }
   414→                }
   415→            )
   416→
   417→            val result = segmentedDownloader.download(config)
   418→
   419→            result.onSuccess { file ->
   420→                // 直接使用传入的 taskId
   421→                val currentTask = activeDownloadTasks[taskId]
   422→                if (currentTask != null) {
   423→                    val completedTask = currentTask.copy(
   424→                        status = DownloadStatus.COMPLETED,
   425→                        progress = 100,
   426→                        completedAt = System.currentTimeMillis()
   427→                    )
   428→                    onDownloadCompleted(taskId, completedTask, file, onComplete)
   429→                }
   430→            }.onFailure { error ->
   431→                // 直接使用传入的 taskId
   432→                val currentTask = activeDownloadTasks[taskId]
   433→                if (currentTask != null) {
   434→                    onDownloadFailed(taskId, currentTask, "下载失败: ${error.message}")
   435→                }
   436→            }
   437→
   438→        } catch (e: Exception) {
   439→            android.util.Log.e("HomeViewModel", "Segmented download failed", e)
   440→            // 这里不需要额外处理，因为 onFailure 已经处理了
   441→        }
   442→    }

   444→    /**
   445→     * 取消当前下载
   446→     */
   447→    fun cancelDownload(taskId: String? = null) {
   448→        if (taskId != null) {
   449→            // 取消特定任务 - 直接删除，不保留记录
   450→            activeDownloadJobs[taskId]?.cancel()
   451→            activeDownloadTasks[taskId]?.let { task ->
   452→                activeDownloadTasks.remove(taskId)
   453→                activeDownloadJobs.remove(taskId)
   454→                updateActiveDownloads()
   455→
   456→                // 清理临时文件
   457→                if (task.totalBytes > 0) {
   458→                    val safeTitle = task.title.replace(Regex("[\\\\/:*?\"<>|]"), "_").take(50)
   459→                    val outputFile = File(downloadsDir, "${safeTitle}_${task.createdAt}.mp4")
   460→                    segmentedDownloader.cancelDownload(outputFile)
   461→                }
   462→            }
   463→            // 处理队列中的下一个任务
   464→            processQueue()
   465→        } else {
   466→            // 取消所有任务 - 直接删除，不保留记录
   467→            activeDownloadJobs.values.forEach { it.cancel() }
   468→            activeDownloadTasks.values.forEach { task ->
   469→                // 清理临时文件
   470→                if (task.totalBytes > 0) {
   471→                    val safeTitle = task.title.replace(Regex("[\\\\/:*?\"<>|]"), "_").take(50)
   472→                    val outputFile = File(downloadsDir, "${safeTitle}_${task.createdAt}.mp4")
   473→                    segmentedDownloader.cancelDownload(outputFile)
   474→                }
   475→            }
   476→            activeDownloadJobs.clear()
   477→            activeDownloadTasks.clear()
   478→            updateActiveDownloads()
   479→        }
   480→    }

   504→     * 从历史记录中删除条目并删除视频文件（通过 filePath）
   505→     */
   506→    fun deleteFromHistoryByPath(filePath: String) {
   507→        viewModelScope.launch(Dispatchers.IO) {
   508→            // 查找历史记录
   509→            val history = _downloadHistory.value.find { it.filePath == filePath }
   510→            if (history != null) {
   511→                // 删除视频文件
   512→                try {
   513→                    val videoFile = File(history.filePath)
   514→                    if (videoFile.exists()) {
   515→                        videoFile.delete()
   516→                        android.util.Log.d("HomeViewModel", "Deleted video file: ${history.filePath}")
   517→                    }
   518→
   519→                    // 删除相关字幕文件
   520→                    val base = history.filePath.substringBeforeLast(".")
   521→                    listOf(".vtt", ".wordts.json").forEach { ext ->
   522→                        val file = File("$base$ext")
   523→                        if (file.exists()) {
   524→                            file.delete()
   525→                        }
   526→                    }
   527→                } catch (e: Exception) {
   528→                    android.util.Log.e("HomeViewModel", "Error deleting video file", e)
   529→                }
   530→
   531→                // 从历史记录中移除
   532→                _downloadHistory.update { it.filter { h -> h.id != history.id } }
   533→                saveDownloadHistory()
   534→            }
   535→        }
   536→    }

   625→

   627→                    var opened = false
   628→
   629→                    // 方式1：用 DocumentsUI 打开目录（兼容性最好）
   630→                    if (!opened) {
   631→                        try {
   632→                            val relativePath = parentDir.absolutePath.removePrefix("/storage/emulated/0/")

   634→                            val uri = android.net.Uri.parse("content://com.android.externalstorage.documents/document/primary:$encodedPath")
   635→                            val intent = Intent(Intent.ACTION_VIEW).apply {
   636→                                setDataAndType(uri, "vnd.android.document/directory")
   637→                                addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   638→                            }
   639→                            context.startActivity(intent)
   640→                            opened = true
   641→                        } catch (_: Exception) {}
   642→                    }
   643→
   644→                    // 方式2：用系统文件管理器 BROWSE 动作
   645→                    if (!opened) {
   646→                        try {
   647→                            val intent = Intent("android.provider.action.BROWSE").apply {
   648→                                data = android.net.Uri.parse("content://com.android.externalstorage.documents/document/primary:${parentDir.absolutePath.removePrefix("/storage/emulated/0/")}")
   649→                                addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   650→                            }
   651→                            context.startActivity(intent)
   652→                            opened = true
   653→                        } catch (_: Exception) {}
   654→                    }
   655→
   656→                    // 方式3：直接分享视频文件
   657→                    if (!opened) {
   658→                        try {
   659→                            val uri = androidx.core.content.FileProvider.getUriForFile(
   660→                                context, "${context.packageName}.fileprovider", videoFile
   661→                            )
   662→                            val intent = Intent(Intent.ACTION_VIEW).apply {
   663→                                setDataAndType(uri, "video/*")
   664→                                addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
   665→                                addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   666→                            }
   667→                            context.startActivity(Intent.createChooser(intent, "打开视频文件"))

   669→                        } catch (_: Exception) {}
   670→                    }

   675→                        clipboard.setPrimaryClip(android.content.ClipData.newPlainText("文件路径", parentDir.absolutePath))

   682→                }
   683→            } catch (e: Exception) {

   685→                withContext(Dispatchers.Main) {
   686→                    android.widget.Toast.makeText(context, "打开失败: ${e.message}", android.widget.Toast.LENGTH_LONG).show()
   687→                }
   688→            }

   882→        if (activeDownloadTasks.size >= MAX_CONCURRENT_DOWNLOADS) return
   883→

   887→        updateQueueList()
   888→
   889→        // 开始下载
   890→        startDownload(task.url) { _, _ -> }
   891→    }

  1068→    private fun getFileName(context: Context, uri: Uri): String? {
  1069→        var result: String? = null
  1070→        if (uri.scheme == "content") {
  1071→            val cursor = context.contentResolver.query(uri, null, null, null, null)
  1072→            cursor?.use {
  1073→                if (it.moveToFirst()) {
  1074→                    val index = it.getColumnIndex(android.provider.OpenableColumns.DISPLAY_NAME)
  1075→                    if (index >= 0) result = it.getString(index)
  1076→                }
  1077→            }
  1078→        }
  1079→        if (result == null) {
  1080→            result = uri.path
  1081→            val cut = result?.lastIndexOf('/')
  1082→            if (cut != -1) {
  1083→                result = result?.substring(cut!! + 1)
  1084→            }
  1085→        }
  1086→        return result
  1087→    }

  1093→    fun onBilibiliCookiesExtracted(

  1097→        fullCookie: String = "",
  1098→        onComplete: (String, String?) -> Unit

  1107→                    biliJct?.let { append("bili_jct=$it; ") }
  1108→                    dedeUserID?.let { append("DedeUserID=$it; ") }

  1110→            }
  1111→
  1112→            // 保存Cookie到配置

  1122→

  1329→    fun transcriptionCompleted(videoPath: String? = null, vttPath: String? = null) {
  1330→        val isSuccess = videoPath != null && vttPath != null
  1331→        _uiState.update {
  1332→            it.copy(
  1333→                transcribingVideo = null,
  1334→                transcriptionQueue = emptyList(),
  1335→                transcriptionProgress = 0,
  1336→                transcriptionMessage = "",
  1337→                snackbarMessage = if (isSuccess) "转录完成，可以开始学习了" else "转录失败，请检查网络和 API Key",
  1338→                switchToTab = if (isSuccess) 2 else null
  1339→            )
  1340→        }
  1341→        // 更新数据库中的 vttPath
  1342→        if (videoPath != null && vttPath != null) {
  1343→            viewModelScope.launch(Dispatchers.IO) {
  1344→                try {
  1345→                    val video = videoRepository.getByPath(videoPath)
  1346→                    if (video != null) {
  1347→                        val wtsPath = vttPath.substringBeforeLast(".") + ".wordts.json"
  1348→                        videoRepository.update(video.copy(
  1349→                            vttPath = vttPath,
  1350→                            wordTsPath = if (java.io.File(wtsPath).exists()) wtsPath else null
  1351→                        ))
  1352→                        android.util.Log.d("HomeViewModel", "Updated vttPath for: ${video.title}")
  1353→                    }
  1354→                } catch (e: Exception) {
  1355→                    android.util.Log.e("HomeViewModel", "Failed to update vttPath", e)
  1356→                }
  1357→            }
  1358→        }
  1359→    }

  1365→    fun startTranscription(video: VideoEntity, context: Context) {

  1368→                // 启动转录服务
  1369→                val intent = Intent(context, com.multilanglearner.service.TranscriptionService::class.java).apply {
  1370→                    action = com.multilanglearner.service.TranscriptionService.ACTION_START
  1371→                    putExtra(com.multilanglearner.service.TranscriptionService.EXTRA_VIDEO_PATH, video.filePath)
  1372→                    putExtra(com.multilanglearner.service.TranscriptionService.EXTRA_VIDEO_NAME, video.title)
  1373→                }
  1374→                context.startForegroundService(intent)
  1375→
  1376→                _uiState.update { it.copy(
  1377→                    snackbarMessage = "开始转录「${video.title}」",
  1378→                    transcribingVideo = video.title,
  1379→                    transcriptionProgress = 0,
  1380→                    transcriptionMessage = "准备中..."
  1381→                )}
  1382→                android.util.Log.d("HomeViewModel", "Started transcription for: ${video.title}")
  1383→            } catch (e: Exception) {
  1384→                android.util.Log.e("HomeViewModel", "Failed to start transcription", e)
  1385→                _uiState.update { it.copy(snackbarMessage = "启动转录失败: ${e.message}") }
